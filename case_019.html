案例 19：shared_ptr/unique_ptr

练习 C++11 智能指针的使用：unique_ptr 独占所有权、shared_ptr 共享所有权、weak_ptr 打破循环引用。

核心知识点：智能指针、所有权模型、循环引用、自定义删除器

完整代码（含逐行注释）

// 案例 19：shared_ptr / unique_ptr
// 理解两种智能指针的所有权模型
#include <iostream>
#include <memory>    // shared_ptr, unique_ptr, weak_ptr, make_shared
#include <string>
using namespace std;

class Resource {
    string name;
public:
    Resource(const string& n) : name(n) {
        cout << "[" << name << "] 创建" << endl;
    }
    ~Resource() { cout << "[" << name << "] 销毁" << endl; }
    void use() { cout << "使用 [" << name << "]" << endl; }
};

// 演示 shared_ptr 循环引用问题
struct Node {
    string name;
    shared_ptr<Node> next;   // shared_ptr 会导致循环引用
    weak_ptr<Node> prev;     // weak_ptr 打破循环引用
    Node(const string& n) : name(n) {}
    ~Node() { cout << "Node [" << name << "] 销毁" << endl; }
};

int main() {
    // === unique_ptr：独占所有权，不可拷贝，可移动 ===
    {
        unique_ptr<Resource> u1 = make_unique<Resource>("独占资源");
        u1->use();

        // unique_ptr<Resource> u2 = u1; // 编译错误！不可拷贝
        unique_ptr<Resource> u2 = move(u1); // 移动转移所有权
        // u1 现在为空（nullptr）
        if (!u1) cout << "u1 已为空" << endl;
    } // u2 离开作用域，自动释放

    cout << "---" << endl;

    // === shared_ptr：共享所有权，引用计数 ===
    {
        shared_ptr<Resource> s1 = make_shared<Resource>("共享资源");
        {
            shared_ptr<Resource> s2 = s1; // 引用计数 +1
            cout << "引用计数 = " << s1.use_count() << endl;
            s2->use();
        } // s2 离开作用域，引用计数 -1
        cout << "引用计数 = " << s1.use_count() << endl;
    } // 引用计数为 0，释放资源

    cout << "---" << endl;

    // === weak_ptr 打破循环引用 ===
    {
        auto node1 = make_shared<Node>("Node1");
        auto node2 = make_shared<Node>("Node2");

        node1->next = node2;   // shared_ptr，形成引用
        node2->prev = node1;   // weak_ptr，不增加引用计数

        // 没有循环引用，离开作用域后都正确析构
    }

    return 0;
}

引用计数 = 1
