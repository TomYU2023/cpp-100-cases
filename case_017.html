案例 17：Lambda表达式练手

深入练习 Lambda 表达式的各种捕获方式和用法，包括值捕获、引用捕获、泛型 Lambda。

核心知识点：Lambda 语法、捕获列表、泛型 Lambda、函数对象

完整代码（含逐行注释）

// 案例 17：Lambda 表达式练手
// 练习各种捕获方式和 Lambda 用法
#include <iostream>
#include <vector>
#include <algorithm>
#include <functional>
using namespace std;

int main() {
    int a = 10, b = 20, c = 30;

    // 1. 值捕获：拷贝外部变量，Lambda 内修改不影响外部
    auto f1 = [a, b]() { return a + b; };
    cout << "[a,b] 值捕获 sum = " << f1() << endl; // 30

    // 2. 引用捕获：可修改外部变量
    auto f2 = [&a]() { a += 5; };
    f2();
    cout << "引用捕获后 a = " << a << endl; // 15

    // 3. 可变 Lambda：值捕获但允许在 Lambda 内修改（不影响外部）
    auto f3 = [a]() mutable { a += 100; return a; };
    cout << "[a] mutable 内修改 = " << f3() << endl; // 115
    cout << "原始 a = " << a << endl;                // 15

    // 4. [=] 全部值捕获，[&] 全部引用捕获
    auto f4 = [=]() { return a + b + c; };
    auto f5 = [&]() { a++; b++; c++; };

    // 5. 泛型 Lambda（C++14）：auto 参数
    auto genericAdd = [](auto x, auto y) { return x + y; };
    cout << "泛型 add(3, 4.5) = " << genericAdd(3, 4.5) << endl;

    // 6. Lambda 作为 sort 比较器
    vector<int> nums = {5, 2, 8, 1, 9};
    sort(nums.begin(), nums.end(),
         [](int x, int y) { return x > y; }); // 降序
    cout << "降序排序: ";
    for (int n : nums) cout << n << " ";
    cout << endl;

    // 7. function 包装 Lambda（适合存储和传递）
    function<int(int, int)> op = [](int x, int y) { return x * y; };
    cout << "3 * 5 = " << op(3, 5) << endl;

    return 0;
}

运行结果

[a,b] 值捕获 sum = 30
引用捕获后 a = 15
[a] mutable 内修改 = 115
原始 a = 15
泛型 add(3, 4.5) = 7.5
降序排序: 9 8 5 2 1 
3 * 5 = 15
