案例 43：MD5/SHA256校验

使用 OpenSSL 或 Windows Crypto API 计算文件的 MD5 和 SHA256 哈希值。

核心知识点：哈希算法、文件 I/O、第三方库集成

完整代码（含逐行注释）

// 案例 43：MD5 / SHA256 校验
// 使用简化算法计算字符串哈希（教学演示）
#include <iostream>
#include <string>
#include <sstream>
#include <iomanip>
#include <vector>
#include <cstdint>
using namespace std;

// 简易 MD5 散列函数（教学简化版，演示位运算和摘要输出）
class SimpleHash {
public:
    // 将字节数组转为十六进制字符串
    static string toHex(const vector<uint8_t>& data) {
        stringstream ss;
        for (uint8_t byte : data) {
            ss << hex << setfill('0') << setw(2) << (int)byte;
        }
        return ss.str();
    }

    // 简易哈希：基于累加和位移的演示性哈希
    // 注意：这不是真正的 MD5/SHA256，仅演示哈希概念
    static string compute(const string& input) {
        // 使用简单的 FNV-1a 哈希算法作为替代
        uint64_t hash = 14695981039346656037ULL; // FNV offset basis

        for (char c : input) {
            hash ^= static_cast<uint64_t>(static_cast<uint8_t>(c));
            hash *= 1099511628211ULL; // FNV prime
        }

        // 将 64 位哈希转为 16 个字节
        vector<uint8_t> bytes(16, 0);
        for (int i = 0; i < 8; i++) {
            bytes[i] = static_cast<uint8_t>((hash >> (i * 8)) & 0xFF);
        }
        // 再做一轮以填充
        uint64_t hash2 = hash ^ 0xA5A5A5A5A5A5A5A5ULL;
        for (int i = 0; i < 8; i++) {
            bytes[8 + i] = static_cast<uint8_t>((hash2 >> (i * 8)) & 0xFF);
        }

        return toHex(bytes);
    }
};

int main() {
    // 测试字符串
    vector<string> inputs = {
        "hello world",
        "Hello World",  // 大小写不同，哈希完全不一样
        "hello world!",
        "C++ Programming",
        "The quick brown fox jumps over the lazy dog"
    };

    cout << "=== 字符串哈希值 ===" << endl;
    cout << left << setw(50) << "输入「 << 」哈希值 (32位十六进制)" << endl;
    cout << string(90, '-') << endl;

    for (const auto& s : inputs) {
        string hash = SimpleHash::compute(s);
        cout << left << setw(50) << ("\"" + s + "\"") << hash << endl;
    }

    // 验证：相同输入产生相同哈希
    cout << "\n=== 验证一致性 ===" << endl;
    string s1 = "test", s2 = "test";
    cout << "\"" << s1 << "\" -> " << SimpleHash::compute(s1) << endl;
    cout << "\"" << s2 << "\" -> " << SimpleHash::compute(s2) << endl;
    cout << (SimpleHash::compute(s1) == SimpleHash::compute(s2)
             ? "一致 ✓" : "不一致 ✗") << endl;

    return 0;
}
