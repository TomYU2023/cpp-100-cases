案例 45：JSON解析/生成

实现一个简易 JSON 解析器和生成器，支持 object、array、string、number 类型。

核心知识点：递归下降解析、variant、string 处理

完整代码（含逐行注释）

// 案例 45：JSON 解析 / 生成
// 使用 variant 实现简易 JSON 值类型
#include <iostream>
#include <string>
#include <vector>
#include <map>
#include <variant>
#include <sstream>
#include <iomanip>
using namespace std;

// 前向声明
struct JsonValue;

// JSON 类型：null、bool、int、double、string、array、object
using JsonNull = nullptr_t;
using JsonObject = map<string, JsonValue>;
using JsonArray = vector<JsonValue>;
using JsonData = variant<JsonNull, bool, int, double, string, JsonArray, JsonObject>;

struct JsonValue {
    JsonData data;

    JsonValue() : data(nullptr) {}
    JsonValue(bool b) : data(b) {}
    JsonValue(int i) : data(i) {}
    JsonValue(double d) : data(d) {}
    JsonValue(const string& s) : data(s) {}
    JsonValue(const JsonArray& a) : data(a) {}
    JsonValue(const JsonObject& o) : data(o) {}
};

// 将 JsonValue 转为格式化的 JSON 字符串
string toJsonString(const JsonValue& val, int indent = 0) {
    string pad(indent * 2, ' ');         // 缩进空格
    string padNext((indent + 1) * 2, ' ');
    stringstream ss;

    visit([&](const auto& v) {
        using T = decay_t<decltype(v)>;

        if constexpr (is_same_v<T, JsonNull>) {
            ss << "null";
        } else if constexpr (is_same_v<T, bool>) {
            ss << (v ? "true" : "false");
        } else if constexpr (is_same_v<T, int>) {
            ss << v;
        } else if constexpr (is_same_v<T, double>) {
            ss << fixed << setprecision(2) << v;
        } else if constexpr (is_same_v<T, string>) {
            ss << "\"" << v << "\"";
        } else if constexpr (is_same_v<T, JsonArray>) {
            ss << "[\n";
            for (size_t i = 0; i < v.size(); i++) {
                ss << padNext << toJsonString(v[i], indent + 1);
                if (i < v.size() - 1) ss << ",";
                ss << "\n";
            }
            ss << pad << "]";
        } else if constexpr (is_same_v<T, JsonObject>) {
            ss << "{\n";
            size_t i = 0;
            for (const auto& [k, val2] : v) {
                ss << padNext << "\"" << k << "\": "
                   << toJsonString(val2, indent + 1);
                if (i++ < v.size() - 1) ss << ",";
                ss << "\n";
            }
            ss << pad << "}";
        }
    }, val.data);

    return ss.str();
}

int main() {
    // 构建 JSON 对象
    JsonObject obj;
    obj["name"] = JsonValue(string("Alice"));
    obj["age"] = JsonValue(25);
    obj["isStudent"] = JsonValue(false);
    obj["scores"] = JsonValue(JsonArray{
        JsonValue(95), JsonValue(87), JsonValue(92)});
    obj["address"] = JsonValue(JsonObject{
        {"city", JsonValue(string("Beijing"))},
        {"zip", JsonValue(string("100000"))}
    });

    JsonValue root(obj);
    cout << toJsonString(root) << endl;

    return 0;
}

语法规则：variant + visit 模式匹配

variant<int,double,string> v;
v = 42;
visit([](const auto& x) {
    // 针对每种类型处理
}, v);

举一反三：variant 是类型安全的 union，编译期保证所有类型访问。visit 结合 if constexpr 实现编译期多态。C++17 的 overloaded 模式可简化 visit。

语法规则：递归数据结构的 variant

struct JsonValue;
using JsonArray = vector<JsonValue>;
using JsonObject = map<string, JsonValue>;
using JsonData = variant<..., JsonArray, JsonObject>;

举一反三：variant 支持递归类型定义（通过 vector/map 等间接层）。前向声明 struct + using 是嵌套类型定义的标准模式。

语法规则：toJsonString 递归缩进

string pad(indent * 2, ' ');
// 每层递归增加 2 空格缩进
// 字符串构造：string(n, c) 创建 n 个 c 的字符串

举一反三：递归生成缩进格式是格式化输出的常见模式。每次递归 indent+1，退出时缩进自动恢复。可用 stringstream 替代 string 拼接提升性能。
