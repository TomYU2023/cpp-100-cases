案例 21：vector 自定义分配器

为 vector 编写自定义内存分配器，跟踪内存分配/释放行为，理解 STL 分配器机制。

核心知识点：自定义分配器、allocator 接口、内存跟踪

完整代码（含逐行注释）

// 案例 21：vector 自定义分配器
// 实现一个带日志的自定义分配器
#include <iostream>
#include <vector>
#include <memory>
using namespace std;

// 自定义分配器：记录每次分配和释放
template <typename T>
struct LoggingAllocator {
    using value_type = T;

    LoggingAllocator() = default;
    template <typename U>
    LoggingAllocator(const LoggingAllocator<U>&) {}

    // 分配 n 个 T 类型的内存
    T* allocate(size_t n) {
        cout << "分配 " << n << " 个元素 ("
             << n * sizeof(T) << " bytes)" << endl;
        return static_cast<T*>(::operator new(n * sizeof(T)));
    }

    // 释放内存
    void deallocate(T* p, size_t n) noexcept {
        cout << "释放 " << n << " 个元素" << endl;
        ::operator delete(p);
    }
};

// 分配器相等比较（同类分配器始终相等）
template <typename T, typename U>
bool operator==(const LoggingAllocator<T>&, const LoggingAllocator<U>&) {
    return true;
}
template <typename T, typename U>
bool operator!=(const LoggingAllocator<T>&, const LoggingAllocator<U>&) {
    return false;
}

int main() {
    // 使用自定义分配器创建 vector
    vector<int, LoggingAllocator<int>> v;

    cout << "--- push_back 测试 ---" << endl;
    for (int i = 1; i <= 10; i++) {
        v.push_back(i);
    }

    cout << "\nvector 内容: ";
    for (int n : v) cout << n << " ";
    cout << endl;

    cout << "size=" << v.size()
         << " capacity=" << v.capacity() << endl;
    return 0;
}
