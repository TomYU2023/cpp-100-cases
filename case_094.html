案例 94：std::mem_fn 成员函数包装

使用 mem_fn 将成员函数包装为可调用对象，用于算法和回调。

核心知识点：mem_fn、成员指针、bind、可调用对象

完整代码（含逐行注释）

// 案例 94：std::mem_fn 成员函数包装
// 将成员函数转为可调用对象
#include <iostream>
#include <functional>
#include <vector>
#include <algorithm>
#include <string>
using namespace std;

class Student {
private:
    string name;
    double score;
public:
    Student(string n, double s) : name(move(n)), score(s) {}

    double getScore() const { return score; }
    const string& getName() const { return name; }
    bool isPass() const { return score >= 60.0; }
    void print() const {
        cout << name << ": " << score << endl;
    }
};

int main() {
    vector<Student> students = {
        {"Alice", 85.5},
        {"Bob", 55.0},
        {"Charlie", 92.0},
        {"David", 58.5},
        {"Eve", 73.0}
    };

    // 1. mem_fn 包装成员函数用于算法
    cout << "=== 所有学生 ===" << endl;
    for_each(students.begin(), students.end(),
             mem_fn(&Student::print));

    // 2. 用 mem_fn 作为排序依据
    sort(students.begin(), students.end(),
         [](const Student& a, const Student& b) {
             return a.getScore() > b.getScore();
         });

    cout << "\n=== 按分数排序 ===" << endl;
    for_each(students.begin(), students.end(),
             mem_fn(&Student::print));

    // 3. 用 mem_fn 做筛选
    auto passCount = count_if(students.begin(), students.end(),
                              mem_fn(&Student::isPass));
    cout << "\n及格人数: " << passCount
         << " / " << students.size() << endl;

    // 4. mem_fn vs lambda vs bind
    // mem_fn: mem_fn(&Student::getScore)
    // lambda: [](const Student& s) { return s.getScore(); }
    // bind:   bind(&Student::getScore, placeholders::_1)

    // 用 mem_fn 基于成员函数返回值查找
    auto it = find_if(students.begin(), students.end(),
                      [](const Student& s) {
                          return s.getScore() > 90.0;
                      });
    if (it != students.end()) {
        cout << "\n第一个超 90 分: " << it->getName()
             << " (" << it->getScore() << ")" << endl;
    }

    return 0;
}
