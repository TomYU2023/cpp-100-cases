案例 94：简易数据库

实现 SQL 子集解析（SELECT/FROM/WHERE）、内存表存储、B-Tree 索引和查询执行引擎。

核心知识点：SQL 解析、std::istringstream、B-Tree 索引（map 模拟）、enum class、查询执行

完整代码（含逐行注释）

// 案例 94：简易数据库 —— SQL 解析 + 执行引擎
// 实现 SELECT/FROM/WHERE 子集，包含内存表存储、B-Tree 索引雏形、查询执行。

#include <iostream>
#include <vector>
#include <string>
#include <map>
#include <sstream>
#include <algorithm>
#include <cctype>
#include <functional>
#include <memory>

// --- 数据类型 ---
enum class DataType { INT, STRING };

struct Value {
    DataType type;
    int intVal = 0;
    std::string strVal;
};

// --- 记录（一行）---
struct Row {
    std::map<std::string, Value> columns;
};

// --- 表 ---
struct Table {
    std::string name;
    std::vector<std::string> columnNames;
    std::vector<Row> rows;
    // 简易 B-Tree 索引: 列名 -> (值 -> 行号列表)
    std::map<std::string, std::map<int, std::vector<int>>> index;

    void insert(const std::vector<Value>& values) {
        Row r;
        for (size_t i = 0; i < columnNames.size() && i < values.size(); i++)
            r.columns[columnNames[i]] = values[i];
        rows.push_back(r);
        int rowIdx = (int)rows.size() - 1;
        for (size_t i = 0; i < columnNames.size() && i < values.size(); i++)
            if (values[i].type == DataType::INT)
                index[columnNames[i]][values[i].intVal].push_back(rowIdx);
    }
};

// --- 数据库 ---
std::map<std::string, Table> database;

void createTable(const std::string& name, const std::vector<std::string>& cols) {
    Table t;
    t.name = name;
    t.columnNames = cols;
    database[name] = t;
    std::cout << "表 " << name << " 已创建 (" << cols.size() << " 列)。" << std::endl;
}

// --- SQL 解析器 ---
struct ParsedQuery {
    std::string tableName;
    std::vector<std::string> selectCols;
    std::string whereCol;
    std::string whereOp;   // =, <, >
    std::string whereVal;
};

ParsedQuery parseSQL(const std::string& sql) {
    ParsedQuery q;
    std::istringstream ss(sql);
    std::string token;
    ss >> token; // SELECT
    ss >> token;
    if (token == "*") q.selectCols.clear();
    else {
        q.selectCols.push_back(token);
        while (ss >> token && token != "FROM") {
            if (token.back() == ',') token.pop_back();
            q.selectCols.push_back(token);
        }
    }
    if (token != "FROM") ss >> token;
    ss >> q.tableName;
    if (ss >> token && token == "WHERE") {
        ss >> q.whereCol >> q.whereOp >> q.whereVal;
    }
    return q;
}

// --- 查询执行器 ---
void executeQuery(const ParsedQuery& q) {
    auto it = database.find(q.tableName);
    if (it == database.end()) {
        std::cout << "表 " << q.tableName << " 不存在。" << std::endl;
        return;
    }
    Table& table = it->second;
    std::vector<int> candidates;

    // 优先使用索引（等值查询 O(log n)）
    if (!q.whereCol.empty() && q.whereOp == "=") {
        auto idxIt = table.index.find(q.whereCol);
        if (idxIt != table.index.end()) {
            int val = std::stoi(q.whereVal);
            auto valIt = idxIt->second.find(val);
            if (valIt != idxIt->second.end())
                candidates = valIt->second;
        }
    }
    // 无索引则全表扫描
    if (candidates.empty()) {
        for (int i = 0; i < (int)table.rows.size(); i++)
            candidates.push_back(i);
    }

    auto& outCols = q.selectCols.empty() ? table.columnNames : q.selectCols;
    for (const auto& col : outCols) std::cout << col << "\t";
    std::cout << "\n" << std::string(outCols.size() * 10, '-') << std::endl;

    int count = 0;
    for (int idx : candidates) {
        Row& row = table.rows[idx];
        if (!q.whereCol.empty() && q.whereOp != "=") {
            int rowVal = row.columns.at(q.whereCol).intVal;
            int condVal = std::stoi(q.whereVal);
            if (q.whereOp == "<" && !(rowVal < condVal)) continue;
            if (q.whereOp == ">" && !(rowVal > condVal)) continue;
        }
        for (const auto& col : outCols) {
            auto& v = row.columns.at(col);
            if (v.type == DataType::INT) std::cout << v.intVal;
            else std::cout << v.strVal;
            std::cout << "\t";
        }
        std::cout << std::endl;
        count++;
    }
    std::cout << count << " 行。" << std::endl;
}

int main() {
    std::cout << "=== 简易数据库 ===" << std::endl;
    createTable("users", {"id", "name", "age"});
    database["users"].insert({Value{DataType::INT, 1}, Value{DataType::STRING,0,"Alice"}, Value{DataType::INT, 25}});
    database["users"].insert({Value{DataType::INT, 2}, Value{DataType::STRING,0,"Bob"}, Value{DataType::INT, 30}});
    database["users"].insert({Value{DataType::INT, 3}, Value{DataType::STRING,0,"Charlie"}, Value{DataType::INT, 22}});

    while (true) {
        std::cout << "\nSQL> ";
        std::string sql;
        std::getline(std::cin, sql);
        if (sql == "quit" || sql == "exit") break;
        if (sql.find("SELECT") == 0) {
            auto q = parseSQL(sql);
            executeQuery(q);
        } else {
            std::cout << "仅支持 SELECT ... FROM ... WHERE ..." << std::endl;
        }
    }
    return 0;
}

语法规则：std::istringstream 字符串流解析

#include <sstream>
std::string sql = "SELECT * FROM users WHERE id = 1";
std::istringstream ss(sql);
std::string token;
while (ss >> token) {
    // token 依次为 SELECT, *, FROM, users, WHERE, id, =, 1
}

举一反三：istringstream 将字符串当作输入流处理，>> 按空白分割。这是写词法分析器和简单解析器的最快方式。进阶用 std::regex 或手写状态机处理引号内的空格。

语法规则：B-Tree 索引（用 map 模拟）

// 索引结构: 列名 -> (值 -> 行号列表)
std::map<std::string, std::map<int, std::vector<int>>> index;
// 插入时维护索引
index[colName][intValue].push_back(rowIndex);
// 查询时 O(log n) 定位
auto it = index[colName].find(targetValue);

举一反三：真正的 B-Tree 索引会将数据按页组织在磁盘上，每页 4KB/16KB。内存中用 std::map 模拟原理。当数据量超过内存时，B-Tree 的优势（减少磁盘 IO 次数）才会体现。

语法规则：enum class 强类型枚举

enum class DataType { INT, STRING };
DataType t = DataType::INT;
// int n = t;           // 编译错误！不能隐式转换
int n = (int)t;         // 必须显式转换

// 对比传统 enum（弱类型，容易污染命名空间）
enum OldEnum { A, B };   // A 直接暴露在外部作用域

举一反三：enum class 是 C++11 引入的强类型枚举，作用域限定（DataType::INT），不隐式转换，更安全。传统 enum 的值直接泄漏到外层作用域，容易命名冲突。
