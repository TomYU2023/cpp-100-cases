案例 48：单例模式(线程安全)

实现一个线程安全的单例模式，使用 C++11 的静态局部变量保证安全。

核心知识点：单例模式、静态局部变量、线程安全

完整代码（含逐行注释）

// 案例 48：单例模式（线程安全）
// C++11 静态局部变量保证线程安全的初始化
#include <iostream>
#include <string>
#include <thread>
#include <vector>
using namespace std;

// 数据库连接管理器（单例）
class DatabaseManager {
private:
    string connectionString;
    bool connected;

    // 私有构造函数
    DatabaseManager() : connectionString("localhost:3306"), connected(false) {
        cout << "DatabaseManager 构造" << endl;
    }

    // 禁止拷贝和赋值
    DatabaseManager(const DatabaseManager&) = delete;
    DatabaseManager& operator=(const DatabaseManager&) = delete;

public:
    // 获取单例实例（C++11 保证线程安全）
    static DatabaseManager& getInstance() {
        // 静态局部变量的初始化在 C++11+ 是线程安全的
        static DatabaseManager instance;
        return instance;
    }

    void connect() {
        if (!connected) {
            // 模拟连接
            cout << "连接数据库: " << connectionString << endl;
            connected = true;
        }
    }

    void query(const string& sql) {
        if (!connected) connect();
        cout << "执行: " << sql << endl;
    }

    ~DatabaseManager() {
        cout << "DatabaseManager 析构" << endl;
    }
};

// 配置管理器（另一个单例）
class ConfigManager {
private:
    ConfigManager() { cout << "ConfigManager 构造" << endl; }
    ConfigManager(const ConfigManager&) = delete;
    ConfigManager& operator=(const ConfigManager&) = delete;

public:
    static ConfigManager& getInstance() {
        static ConfigManager instance;
        return instance;
    }

    string getValue(const string& key) {
        return "value_of_" + key;
    }
};

int main() {
    cout << "=== 线程安全单例 ===" << endl;

    // 多线程同时获取单例
    vector<thread> threads;
    for (int i = 0; i < 5; i++) {
        threads.emplace_back([i]() {
            auto& db = DatabaseManager::getInstance();
            db.query("SELECT * FROM users WHERE id=" + to_string(i));
        });
    }
    for (auto& t : threads) t.join();

    // 验证：多次获取返回同一个实例
    auto& db1 = DatabaseManager::getInstance();
    auto& db2 = DatabaseManager::getInstance();
    cout << "&db1 == &db2: "
         << (&db1 == &db2 ? "true (同一实例)" : "false") << endl;

    return 0;
}
