案例 18：移动语义演练

完整代码（带逐行注释）

1 #include <iostream>
2 #include <vector>
3 #include <chrono> 计时
4 using namespace std;
5
6 class Buffer { 模拟大资源类
7 private:
8 int* data_; 动态大数组
9 size_t size_;
10 public:
11 Buffer(size_t n) : size_(n) {
12 data_ = new int[n]; 分配资源
13 cout << "构造函数（分配 " << n << " 个 int）" << endl;
14 }
15
16 Buffer(const Buffer& other) : size_(other.size_) { 拷贝构造
17 data_ = new int[size_]; 深拷贝：分配新内存
18 copy(other.data_, other.data_ + size_, data_); 逐个复制元素
19 cout << "拷贝构造（深拷贝 " << size_ << " 个元素）" << endl;
20 }
21
22 Buffer(Buffer&& other) noexcept 移动构造：&& 表示右值引用，noexcept 重要！
23 : data_(other.data_), size_(other.size_) { 直接接管资源（指针赋值）
24 other.data_ = nullptr; 将源对象置空
25 other.size_ = 0; 防止析构时释放已转移的资源
26 cout << "移动构造（零拷贝！）" << endl;
27 }
28
29 Buffer& operator=(Buffer&& other) noexcept { 移动赋值
30 if (this == &other) return *this; 自赋值检查
31 delete[] data_; 释放当前资源
32 data_ = other.data_;
33 size_ = other.size_;
34 other.data_ = nullptr;
35 other.size_ = 0;
36 cout << "移动赋值" << endl;
37 return *this;
38 }
39
40 ~Buffer() {
41 cout << "析构" << endl;
42 delete[] data_; delete 空指针安全
43 }
44 };
45
46 int main() {
47 Buffer b1(1000000); 分配 100w 个 int（约 4MB）
48
49 cout << endl << \"--- 拷贝 vs 移动 ---\" << endl;
50 Buffer b2 = b1; 调用拷贝构造
51
52 Buffer b3 = move(b1); move 将 b1 转为右值，触发移动构造
53 cout << "b1 已被移动: size=\" << b1.size_ << endl; b1 大小变为 0
54
55 cout << endl << \"--- vector push_back 对比 ---\" << endl;
56 vector<Buffer> v;
57 v.reserve(3); 预留空间，避免扩容干扰
58 v.push_back(Buffer(100)); 临时对象 → 移动
59 cout << "---" << endl;
60 Buffer tmp(100);
61 v.push_back(tmp); 左值 → 拷贝
62 cout << "---" << endl;
63 v.push_back(move(tmp)); 显式移动
64
65 return 0;
66 }

新语法讲解

右值引用 (&&) 与移动语义

T&& 是右值引用，只能绑定到临时对象或 move() 转换后的对象。移动构造/赋值直接接管资源而非复制，大幅提升性能。
string s1 = "hello";
string s2 = move(s1); // 移动，s1 变为空
// 规则：只对即将销毁的变量使用 move()

noexcept 的重要性

移动操作应声明为 noexcept。这对 vector 的扩容至关重要：如果移动构造不是 noexcept，vector 扩容时会退化到拷贝以确保异常安全。
// 检查你的类型
static_assert(is_nothrow_move_constructible_v<MyType>);
// 确保 vector 扩容时使用移动

举一反三

编写 swap 用移动优化

代码：template void fastSwap(T& a, T& b) { T tmp=move(a); a=move(b); b=move(tmp); }

思路：零拷贝交换两个对象

返回值优化 (RVO)

代码：Buffer create() { return Buffer(1000); } // 编译器自动优化，不调用移动

思路：现代编译器会自动消除临时对象

完美转发 forward

代码：template void wrapper(T&& arg) { target(forward(arg)); }

思路：保持参数的值类别不变
