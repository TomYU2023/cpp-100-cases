案例 18：移动语义演练

通过实现一个带资源管理的 Buffer 类，深入理解移动构造和移动赋值的原理和用法。

核心知识点：移动构造、移动赋值、std::move、右值引用

完整代码（含逐行注释）

// 案例 18：移动语义演练
// 通过资源管理类理解移动构造和移动赋值
#include <iostream>
#include <vector>
#include <chrono>
using namespace std;

class Buffer {
private:
    size_t size;
    int* data;

public:
    explicit Buffer(size_t sz) : size(sz), data(new int[sz]) {
        cout << "构造: size=" << size << endl;
    }

    // 拷贝构造（深拷贝，昂贵！）
    Buffer(const Buffer& other) : size(other.size), data(new int[other.size]) {
        copy(other.data, other.data + size, data);
        cout << "拷贝构造: size=" << size << endl;
    }

    // 移动构造（窃取资源，O(1)！）
    Buffer(Buffer&& other) noexcept
        : size(other.size), data(other.data) {
        other.size = 0;      // 置空源对象
        other.data = nullptr;
        cout << "移动构造: size=" << size << endl;
    }

    // 移动赋值
    Buffer& operator=(Buffer&& other) noexcept {
        if (this != &other) {
            delete[] data;           // 释放当前资源
            size = other.size;
            data = other.data;       // 窃取
            other.size = 0;
            other.data = nullptr;
        }
        cout << "移动赋值: size=" << size << endl;
        return *this;
    }

    ~Buffer() {
        if (data) cout << "析构: size=" << size << endl;
        delete[] data;
    }
};

// 工厂函数：返回临时对象（触发移动语义）
Buffer createBuffer(size_t sz) {
    Buffer buf(sz);
    return buf; // NRVO / 移动构造
}

int main() {
    // 1. 从工厂函数接收（移动构造，不拷贝）
    Buffer b1 = createBuffer(1000);

    // 2. 显式使用 std::move 触发移动
    Buffer b2(500);
    Buffer b3 = move(b2); // b2 资源被窃取，b2 不再有效
    // 警告：此时不应再使用 b2（其 data 已被置空）

    // 3. vector 扩容时自动移动（如果移动构造 noexcept）
    vector<Buffer> buffers;
    buffers.reserve(3);
    buffers.push_back(Buffer(100));
    buffers.push_back(Buffer(200));
    buffers.push_back(Buffer(300));

    return 0;
}
