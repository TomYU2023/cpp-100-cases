案例 67：REST API 客户端

封装一个轻量级 HTTP 客户端类，支持 GET 和 POST 请求，包含 DNS 解析和 JSON POST 示例。

核心知识点：HTTP 客户端、getaddrinfo DNS 解析、原始字符串字面量、ostringstream

完整代码（含逐行注释）

// 案例 67：REST API 客户端（封装 HTTP GET/POST）
// 用原生 socket 封装一个轻量级 HTTP 客户端，支持 GET 和 POST 请求

#include <iostream>
#include <string>
#include <sstream>
#include <cstring>
#include <map>

#ifdef _WIN32
    #include <winsock2.h>
    #include <ws2tcpip.h>
    #pragma comment(lib, "ws2_32.lib")
    using socklen_t = int;
    #define CLOSE_SOCK closesocket
#else
    #include <sys/socket.h>
    #include <netinet/in.h>
    #include <arpa/inet.h>
    #include <netdb.h>       // gethostbyname
    #include <unistd.h>
    #define CLOSE_SOCK close
#endif

/* ── 语法规则：DNS 解析 getaddrinfo ──
   int getaddrinfo(node, service, hints, res)
     node: 主机名（如 "www.baidu.com"）或 IP
     service: 端口号字符串（如 "80"）
     hints: addrinfo 结构体，设置期望的地址族/协议等
     res: 输出，addrinfo 链表头指针
     返回：0 成功，非 0 失败 → gai_strerror(code) 获取错误信息

   示例：
     addrinfo hints{};
     hints.ai_family = AF_INET;     // IPv4
     hints.ai_socktype = SOCK_STREAM; // TCP
     addrinfo* result;
     getaddrinfo("www.example.com", "80", &hints, &result);
     // result 中 ai_addr 就是可用的 sockaddr
     freeaddrinfo(result);

   举一反三：getaddrinfo 替代过时的 gethostbyname，支持 IPv6。
   几乎所有网络程序建立连接的第一步就是 DNS 解析。
*/

class HttpClient {
    std::string host;
    int port;
    int sockFd;
    static const int BUFFER_SIZE = 8192;

    // 连接到服务器
    bool connect() {
        // 1. 创建 socket
        sockFd = socket(AF_INET, SOCK_STREAM, 0);
        if (sockFd < 0) return false;

        // 2. DNS 解析
        addrinfo hints{}, *result;
        hints.ai_family = AF_INET;
        hints.ai_socktype = SOCK_STREAM;

        std::string portStr = std::to_string(port);
        if (getaddrinfo(host.c_str(), portStr.c_str(), &hints, &result) != 0) {
            CLOSE_SOCK(sockFd);
            return false;
        }

        // 3. 连接服务器
        int ret = ::connect(sockFd, result->ai_addr, result->ai_addrlen);
        freeaddrinfo(result);  // 释放 DNS 结果内存

        if (ret < 0) {
            CLOSE_SOCK(sockFd);
            return false;
        }
        return true;
    }

    // 发送请求并接收响应
    std::string sendRequest(const std::string& request) {
        if (!connect()) return "连接失败";

        // 发送请求
        ::send(sockFd, request.c_str(), request.size(), 0);

        // 接收响应
        char buffer[BUFFER_SIZE];
        std::string response;
        int bytes;
        while ((bytes = recv(sockFd, buffer, BUFFER_SIZE - 1, 0)) > 0) {
            buffer[bytes] = '\0';
            response += buffer;
        }

        CLOSE_SOCK(sockFd);
        return response;
    }

public:
    HttpClient(const std::string& h, int p = 80) : host(h), port(p), sockFd(-1) {}

    // GET 请求
    std::string get(const std::string& path) {
        std::ostringstream oss;
        oss << "GET " << path << " HTTP/1.1\r\n"
            << "Host: " << host << "\r\n"
            << "User-Agent: CppHttpClient/1.0\r\n"  // 自定义 User-Agent
            << "Accept: */*\r\n"
            << "Connection: close\r\n"              // 短连接
            << "\r\n";                              // 请求头结束
        return sendRequest(oss.str());
    }

    // POST 请求（JSON）
    std::string post(const std::string& path, const std::string& body) {
        std::ostringstream oss;
        oss << "POST " << path << " HTTP/1.1\r\n"
            << "Host: " << host << "\r\n"
            << "Content-Type: application/json\r\n"   // JSON 类型
            << "Content-Length: " << body.size() << "\r\n"
            << "Connection: close\r\n"
            << "\r\n"
            << body;                                     // 请求体
        return sendRequest(oss.str());
    }
};

int main() {
#ifdef _WIN32
    WSADATA wsa;
    WSAStartup(MAKEWORD(2,2), &wsa);
#endif

    // 测试 GET 请求（向 httpbin.org 发送测试请求）
    HttpClient client("httpbin.org", 80);

    std::cout << "=== GET /get?name=cpp ===\n";
    std::string getResp = client.get("/get?name=cpp");
    std::cout << getResp << "\n\n";

    // 测试 POST 请求
    std::cout << "=== POST /post ===\n";
    std::string postResp = client.post("/post", R"({"language":"C++","level":"intermediate"})");
    std::cout << postResp << std::endl;

#ifdef _WIN32
    WSACleanup();
#endif
    return 0;
}

语法规则：原始字符串字面量 R"(...)"

// C++11 原始字符串，括号内所有字符原样保留，无需转义
std::string json = R"({"key":"value"})";  // 错误：内引号需注意
std::string json = R"({"key":"value"})"; // 正确
// 推荐用分隔符：R"delim(...)delim"

💡 举一反三：写 JSON、正则表达式、HTML 模板时非常实用，避免大量反斜杠转义。delim 可以是任意字符串，只要首尾一致即可：R"JSON({"a":1})JSON"。

语法规则：ostringstream 字符串拼接

std::ostringstream oss;
oss << "Hello " << name << ", age " << age;
std::string result = oss.str();  // 获取拼接结果

💡 举一反三：比 + 号拼接效率高（避免多次临时对象分配），也方便格式化输出。类似的有 istringstream 用于字符串解析。
