案例 67：std::format 格式化

使用 C++20 format 库进行类型安全的字符串格式化，替代 printf 和 stringstream。

核心知识点：format、格式化占位符、类型安全、编译期检查

完整代码（含逐行注释）

// 案例 67：std::format 格式化
// 类型安全的字符串格式化（C++20）
#include <iostream>
#include <format>
#include <string>
#include <vector>
#include <chrono>
using namespace std;

int main() {
    // 1. 基本格式化
    cout << format("Hello, {}!", "World") << endl;
    cout << format("整数: {}, 浮点数: {:.2f}", 42, 3.14159) << endl;

    // 2. 位置参数
    cout << format("{1} 在 {0} 之前", " 」之后", " 」之前") << endl;

    // 3. 对齐、宽度、填充
    cout << format("|{:>10}|", "右对齐") << endl;   // 右对齐 10 字符宽
    cout << format("|{:<10}|", "左对齐") << endl;   // 左对齐
    cout << format("|{:^10}|", "居中") << endl;     // 居中
    cout << format("|{:*^10}|", "星号填充") << endl; // * 填充

    // 4. 进制和精度
    cout << format("十六进制: {:#x}, 八进制: {:#o}, 二进制: {:#b}",
                   255, 255, 255) << endl;
    cout << format("科学计数: {:.2e}", 12345.6789) << endl;
    cout << format("百分比: {:.1%}", 0.856) << endl;

    // 5. 格式化为字符串
    string msg = format("结果: x={}, y={:.2f}, name=\"{}\"",
                        10, 3.14159, "test");
    cout << msg << endl;

    // 6. 表格对齐
    cout << "\n=== 数据表格 ===" << endl;
    cout << format("{:<8} {:<6} {:>8}\n", "姓名", " 」年龄", " 」分数");
    cout << format("{:-<8} {:-<6} {:->8}\n", "", "", "");
    cout << format("{:<8} {:<6} {:>8.1f}\n", "Alice", 25, 95.5);
    cout << format("{:<8} {:<6} {:>8.1f}\n", "Bob", 30, 88.0);

    return 0;
}

运行结果

整数格式化：
  format("{}", 42)           -> "42"
  format("{:>10}", 42)       -> "        42"
  format("{:0>10}", 42)      -> "0000000042"
  format("{:#x}", 42)        -> "0x2a"

浮点数格式化：
  format("{:.2f}", 3.14159)  -> "3.14"
  format("{:.4e}", 3.14159)  -> "3.1416e+00"

字符串格式化：
  format("Hello, {}!", "World") -> "Hello, World!"
  format("{:*^20}", "center")   -> "*******center*******"

组合格式化：
  format("{} + {} = {}", 1, 2, 3) -> "1 + 2 = 3"

使用 format_to（输出到已有缓冲区）：
  buffer contents -> "Formatted: 42, 3.14"
