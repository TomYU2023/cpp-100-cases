案例 83：std::next_permutation 排列生成

使用 next_permutation 生成全排列，解决排列组合类算法问题。

核心知识点：next_permutation、全排列、字典序、排列遍历

完整代码（含逐行注释）

// 案例 83：std::next_permutation 排列生成
// 生成所有排列组合
#include <iostream>
#include <algorithm>
#include <vector>
#include <string>
using namespace std;

int main() {
    // 1. 基础：生成数字全排列
    vector<int> nums = {1, 2, 3};

    cout << "=== 全排列 (1, 2, 3) ===" << endl;
    int count = 0;
    do {
        cout << ++count << ": ";
        for (int x : nums) cout << x << " ";
        cout << endl;
    } while (next_permutation(nums.begin(), nums.end()));
    cout << "共 " << count << " 种排列" << endl;

    // 2. 必须从有序序列开始才能生成全部
    cout << "\n=== 从非有序开始会漏 ===" << endl;
    vector<int> unsorted = {3, 1, 2};
    count = 0;
    do {
        cout << ++count << ": ";
        for (int x : unsorted) cout << x << " ";
        cout << endl;
    } while (next_permutation(unsorted.begin(), unsorted.end()));
    cout << "仅 " << count << " 种（漏了前面的）" << endl;

    // 3. 字符串排列
    cout << "\n=== 字符串排列 ===" << endl;
    string s = "ABC";
    sort(s.begin(), s.end()); // 先排序
    do {
        cout << s << endl;
    } while (next_permutation(s.begin(), s.end()));

    // 4. prev_permutation：递减排列
    cout << "\n=== 递减排列 ===" << endl;
    vector<int> desc = {4, 3, 2, 1};
    do {
        for (int x : desc) cout << x << " ";
        cout << endl;
    } while (prev_permutation(desc.begin(), desc.end()));

    return 0;
}
