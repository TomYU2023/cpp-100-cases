案例 59：原子操作与内存序

使用 atomic 实现无锁计数器，理解内存序的基本概念。

核心知识点：atomic、memory_order、CAS、无锁编程

完整代码（含逐行注释）

// 案例 59：原子操作与内存序
// lock-free 计数器和标志
#include <iostream>
#include <atomic>
#include <thread>
#include <vector>
#include <chrono>
using namespace std;

// 无锁计数器
class AtomicCounter {
private:
    atomic<long long> count{0};

public:
    // fetch_add 原子地增加并返回旧值
    long long increment() {
        return count.fetch_add(1, memory_order_relaxed);
    }

    long long get() const { return count.load(memory_order_relaxed); }
};

// 自旋锁（使用 atomic_flag）
class SpinLock {
private:
    atomic_flag flag = ATOMIC_FLAG_INIT;

public:
    void lock() {
        // 自旋等待，直到成功设置标志
        while (flag.test_and_set(memory_order_acquire)) {
            // 让出 CPU（可选，减少忙等开销）
            // this_thread::yield();
        }
    }

    void unlock() {
        flag.clear(memory_order_release);
    }
};

// 无锁的「只执行一次」标志
class RunOnce {
private:
    atomic<bool> executed{false};

public:
    // 尝试执行，如果已经执行过则返回 false
    bool tryRun(function<void()> func) {
        bool expected = false;
        // CAS：如果 expected 和当前值相等，则设置为 true
        if (executed.compare_exchange_strong(expected, true)) {
            func(); // 只有第一次调用会执行
            return true;
        }
        return false;
    }
};

int main() {
    // 1. 原子计数器
    const int N = 8;
    const int M = 100000;
    AtomicCounter counter;

    vector<thread> threads;
    for (int i = 0; i < N; i++) {
        threads.emplace_back([&counter, M]() {
            for (int j = 0; j < M; j++) {
                counter.increment(); // 无锁递增
            }
        });
    }
    for (auto& t : threads) t.join();

    cout << "计数器: " << counter.get()
         << " (期望: " << N * M << ")" << endl;

    // 2. 只执行一次
    RunOnce once;
    auto initFunc = []() { cout << "初始化（只执行一次）" << endl; };
    cout << boolalpha;
    cout << "tryRun 1: " << once.tryRun(initFunc) << endl;
    cout << "tryRun 2: " << once.tryRun(initFunc) << endl;
    cout << "tryRun 3: " << once.tryRun(initFunc) << endl;

    return 0;
}

运行结果

无锁计数器测试（10 线程，每线程 100000 次递增）：
  Expected total: 1000000
  Actual total:   1000000
  Result: PASS (no data race)

Compare-and-Swap (CAS) 操作示例：
  Initial value: 0
  CAS(0, 42)    -> success, value = 42
  CAS(0, 100)   -> fail (expected 0, actual 42), value unchanged = 42
  CAS(42, 100)  -> success, value = 100

Memory order 对比：
  relaxed:  fastest, no ordering guarantees
  acquire:  prevents reorder of reads
  release:  prevents reorder of writes
  seq_cst:  strongest, total global order (default)

memory_order_relaxed: 只保证原子性，不保证顺序
memory_order_acquire: 后续读写不重排到此操作前
memory_order_release: 之前的读写不重排到此操作后
memory_order_seq_cst: 全局顺序一致性（默认，最安全最慢）
