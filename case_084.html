案例 84：std::unique/unique_copy 去重

使用 unique 去除连续重复元素，配合 sort 实现完全去重。

核心知识点：unique、unique_copy、erase-remove 惯用法、完全去重

完整代码（含逐行注释）

// 案例 84：std::unique / unique_copy 去重
// 去除相邻重复元素
#include <iostream>
#include <algorithm>
#include <vector>
#include <string>
#include <iterator>
using namespace std;

int main() {
    // 1. unique：去除连续重复
    vector<int> v1 = {1, 1, 2, 2, 2, 3, 1, 1, 4, 4, 5};
    cout << "原始: ";
    for (int x : v1) cout << x << " ";
    cout << endl;

    auto last = unique(v1.begin(), v1.end());
    v1.erase(last, v1.end()); // 删除尾部残留元素
    cout << "unique 后: ";
    for (int x : v1) cout << x << " ";
    cout << "\n(注意: 1 没有完全去除，因为不连续)" << endl;

    // 2. sort + unique = 完全去重
    vector<int> v2 = {3, 1, 4, 1, 5, 9, 2, 6, 5, 3, 5};
    cout << "\n原始: ";
    for (int x : v2) cout << x << " ";
    cout << endl;

    sort(v2.begin(), v2.end());
    auto last2 = unique(v2.begin(), v2.end());
    v2.erase(last2, v2.end());
    cout << "sort + unique 完全去重: ";
    for (int x : v2) cout << x << " ";
    cout << endl;

    // 3. unique_copy：去重并输出到新容器
    vector<int> v3 = {1, 1, 2, 2, 3, 3, 4, 4};
    vector<int> result;
    unique_copy(v3.begin(), v3.end(), back_inserter(result));
    cout << "\nunique_copy: ";
    for (int x : result) cout << x << " ";
    cout << endl;

    // 4. 自定义相等判断
    // 按绝对值去重
    vector<int> v4 = {-3, -1, 1, 2, -2, 3, 5};
    sort(v4.begin(), v4.end(),
         [](int a, int b) { return abs(a) < abs(b); });
    auto last4 = unique(v4.begin(), v4.end(),
        [](int a, int b) { return abs(a) == abs(b); });
    v4.erase(last4, v4.end());
    cout << "\n按绝对值去重: ";
    for (int x : v4) cout << x << " ";
    cout << endl;

    return 0;
}
