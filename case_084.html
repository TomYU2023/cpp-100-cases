案例 84：连接池

实现一个线程安全的数据库连接池，支持连接复用、上限控制、超时等待和 RAII 自动归还。

核心知识点：shared_ptr、移动语义、condition_variable、RAII、operator bool、= delete

完整代码（含逐行注释）

#include <iostream>
#include <queue>
#include <memory>
#include <mutex>
#include <condition_variable>
#include <chrono>
#include <thread>

// ========== 模拟数据库连接 ==========
// 真实场景替换为 MySQL/PostgreSQL/Redis 连接对象
class DBConnection {
    int id_;
    bool inUse_{false};
public:
    explicit DBConnection(int id) : id_(id) {
        std::cout << "[连接" << id_ << "] 创建\n";
    }
    ~DBConnection() {
        std::cout << "[连接" << id_ << "] 销毁\n";
    }

    int id() const { return id_; }

    bool execute(const std::string& sql) {
        if (inUse_) {
            std::cerr << "[连接" << id_ << "] 已被占用！\n";
            return false;
        }
        std::cout << "[连接" << id_ << "] 执行: " << sql << "\n";
        std::this_thread::sleep_for(std::chrono::milliseconds(100)); // 模拟耗时
        return true;
    }
};

// ========== 连接池 ==========
// 语法规则：RAII 连接包装器 — 借用和归还自动化
// 获取连接时从池中取出，析构时自动归还
class ConnectionGuard {
    std::shared_ptr<DBConnection> conn_; // 共享指针，池也持有一份
    class ConnectionPool* pool_;          // 指向所属池的指针
    bool returned_{false};

public:
    ConnectionGuard(std::shared_ptr<DBConnection> conn, class ConnectionPool* pool)
        : conn_(conn), pool_(pool) {}

    // 禁止拷贝，只允许移动
    ConnectionGuard(const ConnectionGuard&) = delete;
    ConnectionGuard& operator=(const ConnectionGuard&) = delete;
    // 语法规则：移动构造函数 — 转移资源所有权
    // ConnectionGuard(ConnectionGuard&& other) noexcept
    // 移动后原对象状态有效但未定义，通常设为可析构状态
    ConnectionGuard(ConnectionGuard&& other) noexcept
        : conn_(std::move(other.conn_)), pool_(other.pool_) {
        other.pool_ = nullptr;
    }

    ~ConnectionGuard() { release(); }

    DBConnection* operator->() { return conn_.get(); }
    DBConnection& operator*() { return *conn_; }
    // operator-> 和 operator* 使 ConnectionGuard 可以像指针一样使用
    // guard->execute(sql) 等价于 (*guard).execute(sql)

    void release();
    explicit operator bool() const { return conn_ != nullptr; }
    // explicit operator bool：安全布尔转换，防止隐式转 int 等意外
};

class ConnectionPool {
    std::queue<std::shared_ptr<DBConnection>> pool_; // 连接队列
    std::mutex mtx_;
    std::condition_variable cv_;
    size_t maxSize_;
    size_t activeCount_{0};
    bool stopped_{false};

public:
    explicit ConnectionPool(size_t maxSize = 5) : maxSize_(maxSize) {}

    ~ConnectionPool() {
        stop();
    }

    void stop() {
        std::lock_guard<std::mutex> lock(mtx_);
        stopped_ = true;
        cv_.notify_all();
    }

    // 获取连接（阻塞等待直到有可用连接）
    // 语法规则：std::shared_ptr — 共享所有权的智能指针
    // 多个 shared_ptr 指向同一对象，引用计数为 0 时释放。
    // make_shared 一次分配对象+控制块，更高效。
    // use_count() 返回引用计数（调试用，不要用于逻辑判断）。
    ConnectionGuard acquire(std::chrono::milliseconds timeout = std::chrono::milliseconds(5000)) {
        std::unique_lock<std::mutex> lock(mtx_);

        // 等待直到有空闲连接或超时
        auto deadline = std::chrono::steady_clock::now() + timeout;
        while (pool_.empty() && !stopped_) {
            // 如果当前活跃连接数未达上限，创建新连接
            if (activeCount_ < maxSize_) {
                auto conn = std::make_shared<DBConnection>(static_cast<int>(activeCount_));
                pool_.push(conn);
                activeCount_++;
                break;
            }
            // 等待连接归还或超时
            if (cv_.wait_until(lock, deadline) == std::cv_status::timeout) {
                throw std::runtime_error("获取连接超时");
            }
        }

        if (stopped_) throw std::runtime_error("连接池已停止");

        auto conn = pool_.front();
        pool_.pop();
        return ConnectionGuard(conn, this);
    }

    // 归还连接到池中
    void release(std::shared_ptr<DBConnection> conn) {
        std::lock_guard<std::mutex> lock(mtx_);
        pool_.push(conn);
        cv_.notify_one(); // 通知等待的线程
    }

    size_t available() const {
        // 注意：这里简化了，实际需要 mutex
        return pool_.size();
    }
};

void ConnectionGuard::release() {
    if (!returned_ && conn_ && pool_) {
        pool_->release(conn_);
        returned_ = true;
    }
}

// ========== 使用演示 ==========
int main() {
    ConnectionPool pool(3); // 最多 3 个连接

    std::cout << "=== 连接池演示 ===\n\n";

    // 模拟多个线程并发获取连接
    // 语法规则：std::vector + emplace_back 存储线程
    // emplace_back 将参数直接转发给 thread 构造函数，避免拷贝
    std::vector<std::thread> workers;

    auto workerFunc = [&pool](int id) {
        try {
            std::cout << "[线程" << id << "] 请求连接...\n";
            auto guard = pool.acquire();

            // 使用连接
            guard->execute("SELECT * FROM users WHERE id = " + std::to_string(id));
            std::this_thread::sleep_for(std::chrono::milliseconds(200));

            std::cout << "[线程" << id << "] 释放连接\n";
        } catch (const std::exception& e) {
            std::cerr << "[线程" << id << "] 错误: " << e.what() << "\n";
        }
    };

    // 启动 5 个线程（超过池大小 3，部分会等待）
    for (int i = 0; i < 5; i++) {
        workers.emplace_back(workerFunc, i);
    }

    for (auto& t : workers) t.join();

    std::cout << "\n设计要点：\n"
              << "1. 连接复用减少创建/销毁开销\n"
              << "2. 连接数上限防止资源耗尽\n"
              << "3. RAII Guard 自动归还连接\n"
              << "4. 阻塞等待 + 超时机制\n";
    return 0;
}

语法规则：移动构造函数和移动语义

class Buffer {
    char* data_;
    size_t size_;
public:
    Buffer(size_t n) : data_(new char[n]), size_(n) {}
    // 移动构造："偷"走资源
    Buffer(Buffer&& other) noexcept
        : data_(other.data_), size_(other.size_) {
        other.data_ = nullptr; // 源对象置空
        other.size_ = 0;
    }
    // 禁止拷贝
    Buffer(const Buffer&) = delete;
    ~Buffer() { delete[] data_; }
};

💡 举一反三：移动语义避免了深拷贝的开销。移动后源对象应处于"有效但未定义"状态（通常设为 nullptr/0）。noexcept 标记很重要，STL 容器在 resize 时如果移动构造不是 noexcept 会退化到拷贝。

语法规则：operator bool 安全布尔转换

class OptionalInt {
    int val_;
    bool has_val_;
public:
    explicit operator bool() const { return has_val_; }
    // 有 explicit 就不能隐式转换：
    // OptionalInt oi;
    // int x = oi;      // 编译错误！
    // if (oi) { ... }   // OK：条件上下文允许 explicit bool
};

💡 举一反三：C++11 起用 explicit operator bool 代替 C++03 的 safe bool idiom。加上 explicit 防止隐式转换为 int/double 等，但 if/while/&&/||/!/?: 等上下文仍能正常使用。

语法规则：std::this_thread::sleep_for / sleep_until

// 绝对休眠（指定时长）
std::this_thread::sleep_for(std::chrono::seconds(1));
// 相对休眠（指定时间点）
auto wakeup = std::chrono::steady_clock::now() + std::chrono::seconds(1);
std::this_thread::sleep_until(wakeup);

💡 举一反三：sleep_for 适合"暂停 N 秒"场景，sleep_until 适合"等到某个时刻"场景（如定时任务）。注意 sleep_for 可能因系统调度而比指定时长略长。

语法规则：= delete 禁止函数

class NonCopyable {
public:
    NonCopyable() = default;                  // 使用默认构造
    NonCopyable(const NonCopyable&) = delete;  // 禁止拷贝
    NonCopyable& operator=(const NonCopyable&) = delete;
};

💡 举一反三：= delete 禁用一个函数，比 C++03 的 private 声明更明确。可以禁掉任何函数（不仅是编译器自动生成的），如 void f(int) = delete 阻止隐式转换。
