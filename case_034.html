案例 34：chrono 计时器

使用 chrono 库实现高精度计时器，测量函数执行时间，理解现代 C++ 时间库。

核心知识点：chrono、duration、time_point、高精度计时

完整代码（含逐行注释）

// 案例 34：chrono 计时器
// 测量代码执行时间
#include <iostream>
#include <chrono>
#include <thread>
#include <vector>
#include <algorithm>
#include <numeric>
using namespace std;
using namespace std::chrono;

// 简单计时器类
class Timer {
private:
    time_point<high_resolution_clock> start;
public:
    Timer() : start(high_resolution_clock::now()) {}

    // 返回经过的毫秒数
    double elapsed_ms() const {
        auto end = high_resolution_clock::now();
        return duration<double, milli>(end - start).count();
    }

    // 返回经过的微秒数
    double elapsed_us() const {
        auto end = high_resolution_clock::now();
        return duration<double, micro>(end - start).count();
    }
};

// 模拟耗时操作
void slowFunction() {
    // 做一些计算
    long long sum = 0;
    for (int i = 0; i < 1000000; i++) {
        sum += i;
    }
}

int main() {
    // 1. 简单计时
    Timer t;
    slowFunction();
    cout << "slowFunction 耗时: "
         << t.elapsed_ms() << " ms ("
         << t.elapsed_us() << " us)" << endl;

    // 2. 对比不同算法的性能
    vector<int> v(1000000);
    iota(v.begin(), v.end(), 0);

    // 算法 A：accumulate
    {
        Timer t2;
        long long s = accumulate(v.begin(), v.end(), 0LL);
        cout << "accumulate 耗时: " << t2.elapsed_ms() << " ms"
             << "  sum=" << s << endl;
    }

    // 算法 B：手动循环
    {
        Timer t2;
        long long s = 0;
        for (int x : v) s += x;
        cout << "手动循环 耗时: " << t2.elapsed_ms() << " ms"
             << "  sum=" << s << endl;
    }

    // 3. 时间单位转换
    auto one_sec = seconds(1);
    cout << "1秒 = " << duration_cast<milliseconds>(one_sec).count()
         << " ms" << endl;
    cout << "1秒 = " << duration_cast<microseconds>(one_sec).count()
         << " us" << endl;

    return 0;
}

运行结果

accumulate 耗时: 2.2236 ms  sum=499999500000
手动循环 耗时: 2.2373 ms  sum=499999500000
1秒 = 1000 ms
1秒 = 1000000 us
