案例 6：动态数组(造vector轮子)

手写一个简化版动态数组类，实现自动扩容、下标访问等功能，深入理解 vector 底层原理。

核心知识点：动态内存分配、RAII、拷贝构造、运算符重载

完整代码（含逐行注释）

// 案例 6：手写动态数组（造 vector 轮子）
// 理解动态扩容、RAII 和拷贝语义
#include <iostream>
using namespace std;

class MyVector {
private:
    int* data;       // 指向动态数组的指针
    size_t sz;       // 当前元素数量
    size_t cap;      // 当前容量

    // 扩容：容量翻倍
    void expand() {
        cap = (cap == 0) ? 1 : cap * 2;
        int* newData = new int[cap];            // 分配新内存
        for (size_t i = 0; i < sz; i++) {
            newData[i] = data[i];               // 拷贝旧数据
        }
        delete[] data;                          // 释放旧内存
        data = newData;
    }

public:
    // 构造函数
    MyVector() : data(nullptr), sz(0), cap(0) {}

    // 析构函数：释放内存
    ~MyVector() { delete[] data; }

    // 拷贝构造函数（深拷贝）
    MyVector(const MyVector& other) : sz(other.sz), cap(other.cap) {
        data = new int[cap];
        for (size_t i = 0; i < sz; i++) data[i] = other.data[i];
    }

    // 尾部添加元素
    void push_back(int val) {
        if (sz == cap) expand(); // 容量不足则扩容
        data[sz++] = val;
    }

    // 下标访问运算符重载
    int& operator[](size_t index) { return data[index]; }

    size_t size() const { return sz; }
    size_t capacity() const { return cap; }
};

int main() {
    MyVector v;
    for (int i = 1; i <= 10; i++) {
        v.push_back(i * 10);
    }

    for (size_t i = 0; i < v.size(); i++) {
        cout << v[i] << " ";
    }
    cout << endl;
    cout << "size=" << v.size() << " capacity=" << v.capacity() << endl;
    return 0;
}

语法规则：动态内存分配与 RAII

int* data = new int[cap]; // 分配
// ... 使用 ...
delete[] data;           // 释放

举一反三：new[] 必须搭配 delete[]，new 搭配 delete。RAII 将资源绑定到对象生命周期，构造时获取，析构时释放。

语法规则：拷贝构造函数

MyVector(const MyVector& other) : sz(other.sz), cap(other.cap) {
    data = new int[cap];
    // 深拷贝数据
}

举一反三：三大法则：若自定义了析构/拷贝构造/拷贝赋值中的一个，通常需要全部定义。C++11 增加移动构造和移动赋值。

语法规则：运算符重载

int& operator[](size_t index) {
    return data[index];
}

举一反三：返回引用允许读写元素（v[i] = 5）。const 版本返回 const int& 只读。不能重载不存在的运算符（如 **）。
