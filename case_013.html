案例 13：形状继承体系(多态)

构建 Shape 基类，派生 Circle、Rectangle、Triangle，通过虚函数实现多态，计算不同形状的面积。

核心知识点：继承、虚函数、多态、纯虚函数、override

完整代码（含逐行注释）

// 案例 13：形状继承体系（多态）
// 使用虚函数实现运行时多态
#include <iostream>
#include <vector>
#include <cmath>
using namespace std;

// 抽象基类：包含纯虚函数的类不能实例化
class Shape {
public:
    // 纯虚函数 = 0：派生类必须实现
    virtual double area() const = 0;
    virtual string name() const = 0;

    // 虚析构函数：确保通过基类指针删除派生对象时正确析构
    virtual ~Shape() {}
};

// Circle 派生类
class Circle : public Shape {
private:
    double radius;
public:
    Circle(double r) : radius(r) {}
    double area() const override { return M_PI * radius * radius; }
    string name() const override { return "圆形"; }
};

// Rectangle 派生类
class Rectangle : public Shape {
private:
    double width, height;
public:
    Rectangle(double w, double h) : width(w), height(h) {}
    double area() const override { return width * height; }
    string name() const override { return "矩形"; }
};

// Triangle 派生类
class Triangle : public Shape {
private:
    double base, height;
public:
    Triangle(double b, double h) : base(b), height(h) {}
    double area() const override { return 0.5 * base * height; }
    string name() const override { return "三角形"; }
};

int main() {
    // 基类指针容器，存储不同派生类对象
    vector<Shape*> shapes;
    shapes.push_back(new Circle(3.0));
    shapes.push_back(new Rectangle(4.0, 5.0));
    shapes.push_back(new Triangle(6.0, 4.0));

    // 多态：通过基类指针调用，运行时决定调用哪个派生类的实现
    double total = 0;
    for (const Shape* s : shapes) {
        cout << s->name() << " 面积 = " << s->area() << endl;
        total += s->area();
    }
    cout << "总面积 = " << total << endl;

    // 释放内存
    for (Shape* s : shapes) delete s;

    return 0;
}

语法规则：虚函数与多态

class Base {
public:
    virtual void draw(); // 虚函数
};
Base* p = new Derived();
p->draw(); // 调用 Derived::draw()

举一反三：virtual 关键字让函数调用在运行时动态绑定。非虚函数是静态绑定（编译时决定）。多态需通过指针或引用调用。

语法规则：纯虚函数与抽象类

class Shape {
public:
    virtual double area() const = 0; // 纯虚函数
};

举一反三：含纯虚函数的类为抽象类，不能实例化。派生类必须实现所有纯虚函数否则自身也是抽象类。相当于 Java 的 interface。

语法规则：override 关键字

double area() const override { /* ... */ }

举一反三：C++11 引入，明确表示此函数覆盖基类虚函数。编译器会检查签名是否匹配，防止因拼写错误而意外创建新函数。
