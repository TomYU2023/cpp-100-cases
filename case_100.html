案例 100：综合项目：命令行 TODO 工具

综合运用 STL 容器、算法、文件 I/O 和字符串处理，打造一个功能完整的命令行 TODO 管理工具。

核心知识点：文件持久化、命令解析、多模块组合、完整项目结构

完整代码（含逐行注释）

// 案例 100：命令行 TODO 工具
// 支持增删改查、标记完成、优先级和持久化
#include <iostream>
#include <fstream>
#include <vector>
#include <string>
#include <algorithm>
#include <iomanip>
#include <ctime>
#include <sstream>
using namespace std;

// 任务数据结构
struct TodoItem {
    int id;            // 任务 ID
    string title;      // 标题
    bool completed;    // 是否完成
    int priority;      // 优先级 (0=低, 1=中, 2=高)
    string createdAt;  // 创建时间

    // 获取优先级字符串
    string priorityStr() const {
        const char* labels[] = {"低", "中", "高"};
        return labels[priority];
    }
};

// TODO 管理器
class TodoManager {
private:
    vector<TodoItem> items;
    int nextId = 1;
    const string filename = "todos.txt";

    // 获取当前时间字符串
    string getTime() const {
        time_t now = time(nullptr);
        tm t;
        localtime_s(&t, &now);
        char buf[20];
        strftime(buf, sizeof(buf), "%Y-%m-%d %H:%M", &t);
        return buf;
    }

    // 查找任务位置
    int findIndex(int id) const {
        for (size_t i = 0; i < items.size(); i++) {
            if (items[i].id == id) return (int)i;
        }
        return -1;
    }

public:
    // 从文件加载
    void load() {
        ifstream file(filename);
        if (!file) return; // 文件不存在则跳过

        TodoItem item;
        while (file >> item.id >> item.completed >> item.priority) {
            file.ignore(); // 跳过空格
            getline(file, item.title);
            // 读取创建时间
            getline(file, item.createdAt);
            items.push_back(item);
            if (item.id >= nextId) nextId = item.id + 1;
        }
    }

    // 保存到文件
    void save() const {
        ofstream file(filename);
        for (const auto& item : items) {
            file << item.id << " " << item.completed << " "
                 << item.priority << " " << item.title << "\n"
                 << item.createdAt << "\n";
        }
    }

    // 添加任务
    void add(const string& title, int priority = 0) {
        items.push_back({nextId++, title, false,
                         clamp(priority, 0, 2), getTime()});
        cout << "添加任务 #" << items.back().id << ": "
             << title << endl;
    }

    // 删除任务
    void remove(int id) {
        int idx = findIndex(id);
        if (idx == -1) {
            cout << "任务 #" << id << " 不存在" << endl;
            return;
        }
        cout << "删除任务 #" << id << ": " << items[idx].title << endl;
        items.erase(items.begin() + idx);
    }

    // 标记完成
    void complete(int id) {
        int idx = findIndex(id);
        if (idx == -1) {
            cout << "任务 #" << id << " 不存在" << endl;
            return;
        }
        items[idx].completed = true;
        cout << "完成任务 #" << id << ": " << items[idx].title << endl;
    }

    // 列出所有任务
    void list(bool showAll = true) const {
        cout << "\n=== TODO 列表 ===" << endl;
        cout << setw(4) << "ID"
             << setw(6) << "状态"
             << setw(6) << "优先级"
             << setw(22) << "创建时间"
             << "  标题" << endl;
        cout << string(65, '-') << endl;

        for (const auto& item : items) {
            if (!showAll && item.completed) continue;
            cout << setw(4) << item.id
                 << setw(6) << (item.completed ? "完成「 : 」未完成")
                 << setw(6) << item.priorityStr()
                 << setw(22) << item.createdAt
                 << "  " << item.title << endl;
        }
        cout << "共 " << items.size() << " 项" << endl;
    }
};

// 命令解析
void showHelp() {
    cout << "\n命令列表:\n"
         << "  add <标题> -p <0-2>    添加任务 (可指定优先级)\n"
         << "  list                  列出全部任务\n"
         << "  list active           列出未完成任务\n"
         << "  done <id>             标记任务完成\n"
         << "  remove <id>           删除任务\n"
         << "  help                  显示帮助\n"
         << "  quit                  退出\n" << endl;
}

int main() {
    TodoManager manager;
    manager.load();

    cout << "命令行 TODO 工具 (输入 help 查看命令)" << endl;

    string line;
    while (true) {
        cout << "\n> ";
        getline(cin, line);
        istringstream iss(line);
        string cmd;
        iss >> cmd;

        if (cmd == "quit" || cmd == "exit") break;
        else if (cmd == "help") showHelp();
        else if (cmd == "list") {
            string sub;
            iss >> sub;
            manager.list(sub != "active");
        }
        else if (cmd == "add") {
            string rest;
            getline(iss, rest);
            // 去除前导空格
            size_t start = rest.find_first_not_of(" ");
            string title = (start != string::npos)
                           ? rest.substr(start) : "";

            int priority = 0;
            // 查找 -p 标志
            auto pPos = title.find(" -p ");
            if (pPos != string::npos) {
                priority = stoi(title.substr(pPos + 4));
                title = title.substr(0, pPos);
            }
            manager.add(title, priority);
        }
        else if (cmd == "done" || cmd == "complete") {
            int id;
            iss >> id;
            manager.complete(id);
        }
        else if (cmd == "remove" || cmd == "delete") {
            int id;
            iss >> id;
            manager.remove(id);
        }
        else if (!cmd.empty()) {
            cout << "未知命令: " << cmd << " (输入 help 查看帮助)" << endl;
        }

        manager.save(); // 每次操作后保存
    }

    cout << "再见!" << endl;
    return 0;
}

// 业务层: TodoManager 类（增删改查 + 持久化）

语法规则：文件 I/O 持久化

ifstream file(filename);
if (!file) return; // 文件不存在则跳过
// 与程序同目录下的文本文件做持久化
// save() 在每次修改后调用确保数据不丢

举一反三：简单的文本文件是最轻量的持久化方案。二进制文件更快但不可读。JSON 格式是更好的选择（可读性强 + 跨语言）。生产项目应考虑异常处理和数据完整性。

语法规则：命令行参数解析

istringstream iss(line);
iss >> cmd; // 提取第一个词为命令
// 后续词为参数，根据命令类型解析
// 移除了传统 C 的 argc/argv 复杂性

举一反三：istringstream 是轻量命令行解析的好工具。对更复杂的场景考虑 getopt（POSIX）或 CLI11 库。始终为未知命令提供帮助信息。工具可用性很大程度上取决于交互设计。
