案例 54：CRTP 静态多态

使用 CRTP 实现编译期多态，避免虚函数的运行时开销。

核心知识点：CRTP、模板继承、编译期多态

完整代码（含逐行注释）

// 案例 54：CRTP 静态多态
// 编译期多态，零虚函数开销
#include <iostream>
#include <vector>
using namespace std;

// CRTP 基类模板
template <typename Derived>
class Shape {
public:
    // 静态多态：通过 static_cast 调用派生类方法
    void draw() const {
        static_cast<const Derived*>(this)->drawImpl();
    }

    double area() const {
        return static_cast<const Derived*>(this)->areaImpl();
    }

    // 基类可以提供默认实现
    string name() const {
        return static_cast<const Derived*>(this)->nameImpl();
    }
};

// 派生类：Circle
class Circle : public Shape<Circle> {
    double radius;
public:
    explicit Circle(double r) : radius(r) {}

    void drawImpl() const {
        cout << "圆形(半径=" << radius << "), 面积=" << area() << endl;
    }

    double areaImpl() const {
        return 3.14159 * radius * radius;
    }

    string nameImpl() const { return "Circle"; }
};

// 派生类：Rectangle
class Rectangle : public Shape<Rectangle> {
    double w, h;
public:
    Rectangle(double width, double height) : w(width), h(height) {}

    void drawImpl() const {
        cout << "矩形(宽=" << w << " 高=" << h << "), 面积=" << area() << endl;
    }

    double areaImpl() const {
        return w * h;
    }

    string nameImpl() const { return "Rectangle"; }
};

// 使用模板函数而非虚函数
template <typename T>
void drawAndPrint(const Shape<T>& shape) {
    shape.draw();
}

int main() {
    Circle c(5.0);
    Rectangle r(4.0, 3.0);

    // 静态多态：编译时确定调用哪个函数
    c.draw();
    r.draw();

    // 通过模板函数统一处理
    drawAndPrint(c);
    drawAndPrint(r);

    return 0;
}

语法规则：CRTP 模式

template<typename Derived>
class Base {
    void interface() {
        static_cast<Derived*>(this)->impl();
    }
};
class Derived : public Base<Derived> {
    void impl() { /* 具体实现 */ }
};

举一反三：CRTP (Curiously Recurring Template Pattern) 将派生类作为基类的模板参数。基类通过 static_cast 调用派生类方法，编译期解析，零运行时开销。

语法规则：静态多态 vs 动态多态

// 动态多态：虚函数 → vtable 查找 → 运行时
// 静态多态：模板/CRTP → 编译期绑定 → 零开销
// 取舍：动态灵活可容器存异构对象，静态速度快但类型在编译时确定

举一反三：静态多态无法将不同类型对象存入同一个容器（如 vector）。两者选择：性能和编译隔离选静态，运行时灵活性和扩展性选动态。

语法规则：static_cast 向下转型

static_cast<const Derived*>(this)->impl();
// 在 CRTP 中，this 一定是 Derived* 类型
// 不是 dynamic_cast（运行时检查），而是 static_cast（编译期确定）

举一反三：CRTP 中 static_cast 是安全的，因为派生关系在编译期确定。dynamic_cast 有运行时开销，static_cast 则没有。但误用 static_cast 可能导致未定义行为。
