案例 54：CRTP 静态多态

使用 CRTP 实现编译期多态，避免虚函数的运行时开销。

核心知识点：CRTP、模板继承、编译期多态

完整代码（含逐行注释）

// 案例 54：CRTP 静态多态
// 编译期多态，零虚函数开销
#include <iostream>
#include <vector>
using namespace std;

// CRTP 基类模板
template <typename Derived>
class Shape {
public:
    // 静态多态：通过 static_cast 调用派生类方法
    void draw() const {
        static_cast<const Derived*>(this)->drawImpl();
    }

    double area() const {
        return static_cast<const Derived*>(this)->areaImpl();
    }

    // 基类可以提供默认实现
    string name() const {
        return static_cast<const Derived*>(this)->nameImpl();
    }
};

// 派生类：Circle
class Circle : public Shape<Circle> {
    double radius;
public:
    explicit Circle(double r) : radius(r) {}

    void drawImpl() const {
        cout << "圆形(半径=" << radius << "), 面积=" << area() << endl;
    }

    double areaImpl() const {
        return 3.14159 * radius * radius;
    }

    string nameImpl() const { return "Circle"; }
};

// 派生类：Rectangle
class Rectangle : public Shape<Rectangle> {
    double w, h;
public:
    Rectangle(double width, double height) : w(width), h(height) {}

    void drawImpl() const {
        cout << "矩形(宽=" << w << " 高=" << h << "), 面积=" << area() << endl;
    }

    double areaImpl() const {
        return w * h;
    }

    string nameImpl() const { return "Rectangle"; }
};

// 使用模板函数而非虚函数
template <typename T>
void drawAndPrint(const Shape<T>& shape) {
    shape.draw();
}

int main() {
    Circle c(5.0);
    Rectangle r(4.0, 3.0);

    // 静态多态：编译时确定调用哪个函数
    c.draw();
    r.draw();

    // 通过模板函数统一处理
    drawAndPrint(c);
    drawAndPrint(r);

    return 0;
}
