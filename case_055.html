案例 55：类型萃取 type_traits

使用 type_traits 实现编译期类型判断和条件编译，理解模板元编程基础。

核心知识点：type_traits、enable_if、SFINAE、编译期判断

完整代码（含逐行注释）

// 案例 55：类型萃取 type_traits
// 使用 enable_if 实现编译期条件编译
#include <iostream>
#include <type_traits>
#include <string>
#include <vector>
using namespace std;

// 使用 enable_if 限制模板：仅对整数类型生效
template <typename T>
typename enable_if<is_integral<T>::value, T>::type
add(T a, T b) {
    cout << "整数加法: ";
    return a + b;
}

// 仅对浮点类型生效
template <typename T>
typename enable_if<is_floating_point<T>::value, T>::type
add(T a, T b) {
    cout << "浮点加法: ";
    return a + b;
}

// 类型萃取辅助函数
template <typename T>
void printTypeInfo() {
    cout << "类型: " << typeid(T).name() << endl;
    cout << "  是整数? " << boolalpha << is_integral<T>::value << endl;
    cout << "  是浮点? " << is_floating_point<T>::value << endl;
    cout << "  是指针? " << is_pointer<T>::value << endl;
    cout << "  是 const? " << is_const<T>::value << endl;
    cout << "  sizeof = " << sizeof(T) << endl;
}

// 编译期判断：仅当 T 有 .size() 方法时才调用
template <typename T>
auto getSize(const T& container) -> decltype(container.size()) {
    return container.size();
}

int main() {
    // 根据类型自动选择不同的 add 重载
    cout << add(10, 20) << endl;       // 整数版本
    cout << add(3.14, 2.72) << endl;   // 浮点版本

    cout << "\n=== 类型信息 ===" << endl;
    printTypeInfo<int>();
    cout << "---" << endl;
    printTypeInfo<double>();
    cout << "---" << endl;
    printTypeInfo<string>();

    // 编译期条件判断
    if constexpr (is_integral<int>::value) {
        cout << "\nint 是整数类型，执行整数特有逻辑" << endl;
    }

    vector<int> v = {1, 2, 3, 4, 5};
    cout << "vector size = " << getSize(v) << endl;

    return 0;
}
