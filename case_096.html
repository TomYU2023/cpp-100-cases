案例 96：std::invoke 统一调用

使用 invoke 统一调用函数、成员函数、函数对象，理解 C++17 的统一调用语法。

核心知识点：invoke、统一调用、INVOKE 协议、is_invocable

完整代码（含逐行注释）

// 案例 96：std::invoke 统一调用
// 统一调用函数、成员函数、函数对象
#include <iostream>
#include <functional>
#include <string>
#include <vector>
using namespace std;

class Calculator {
public:
    int add(int a, int b) const {
        cout << "  Calculator::add(" << a << ", " << b << ")";
        return a + b;
    }

    static int multiply(int a, int b) {
        return a * b;
    }

    double factor = 1.5;
};

// 普通函数
int subtract(int a, int b) {
    return a - b;
}

int main() {
    // 1. invoke 调用普通函数
    auto result1 = invoke(subtract, 10, 3);
    cout << "invoke(普通函数): " << result1 << endl;

    // 2. invoke 调用 Lambda
    auto lambda = [](int x) { return x * x; };
    auto result2 = invoke(lambda, 7);
    cout << "invoke(Lambda): " << result2 << endl;

    // 3. invoke 调用成员函数（通过对象引用）
    Calculator calc;
    auto result3 = invoke(&Calculator::add, calc, 5, 3);
    cout << " = " << result3 << endl;

    // 4. invoke 调用成员函数（通过对象指针）
    Calculator* pCalc = &calc;
    auto result4 = invoke(&Calculator::add, pCalc, 10, 20);
    cout << " (指针) = " << result4 << endl;

    // 5. invoke 调用静态成员函数
    auto result5 = invoke(&Calculator::multiply, 6, 7);
    cout << "invoke(静态): " << result5 << endl;

    // 6. invoke 访问成员变量
    auto& factor = invoke(&Calculator::factor, calc);
    cout << "invoke(成员变量): " << factor << endl;

    // 7. 实际应用：通用回调调度
    vector<function<void()>> tasks;
    Calculator c;

    // 将各种调用统一包装
    tasks.push_back([&]() {
        cout << invoke(&Calculator::add, c, 1, 2) << endl;
    });
    tasks.push_back([&]() {
        cout << invoke(lambda, 9) << endl;
    });

    for (auto& task : tasks) task();

    return 0;
}
