案例 86：std::search 子序列查找

使用 search 在序列中查找子序列，理解几种搜索算法的区别。

核心知识点：search、find_end、search_n、Boyer-Moore

完整代码（含逐行注释）

// 案例 86：std::search 子序列查找
// 在大序列中搜索子序列
#include <iostream>
#include <algorithm>
#include <vector>
#include <string>
#include <functional>
using namespace std;

int main() {
    // 1. search：查找子序列首次出现的位置
    vector<int> haystack = {1, 2, 3, 4, 1, 2, 3, 4, 5, 6};
    vector<int> needle = {3, 4, 5};

    auto it = search(haystack.begin(), haystack.end(),
                     needle.begin(), needle.end());
    if (it != haystack.end()) {
        cout << "子序列 [3,4,5] 首次在索引 "
             << distance(haystack.begin(), it) << " 处找到" << endl;
    }

    // 2. find_end：查找最后一次出现
    vector<int> needle2 = {1, 2};
    auto last = find_end(haystack.begin(), haystack.end(),
                         needle2.begin(), needle2.end());
    cout << "子序列 [1,2] 最后出现在索引 "
         << distance(haystack.begin(), last) << endl;

    // 3. search_n：查找连续 N 个相同值
    vector<int> data = {1, 2, 2, 2, 3, 4, 4, 4, 4, 5};
    auto pos = search_n(data.begin(), data.end(),
                        3, 2); // 连续 3 个 2
    cout << "\n连续 3 个 2 在索引 "
         << distance(data.begin(), pos) << " 处" << endl;

    pos = search_n(data.begin(), data.end(),
                   4, 4); // 连续 4 个 4
    cout << "连续 4 个 4 在索引 "
         << distance(data.begin(), pos) << " 处" << endl;

    // 4. 字符串搜索
    string text = "hello world, welcome to C++ world!";
    string pattern = "world";

    auto found = search(text.begin(), text.end(),
                        pattern.begin(), pattern.end());
    if (found != text.end()) {
        cout << "\n\"world\" 首次位置: "
             << distance(text.begin(), found) << endl;
        // 再次搜索
        auto found2 = search(found + 1, text.end(),
                             pattern.begin(), pattern.end());
        if (found2 != text.end()) {
            cout << "\"world\" 第二次位置: "
                 << distance(text.begin(), found2) << endl;
        }
    }

    return 0;
}

语法规则：std::search

auto it = search(haystack.begin(), haystack.end(),
    needle.begin(), needle.end());
// 在 haystack 中查找 needle 子序列
// 返回首次匹配的起始迭代器，未找到返回 haystack.end()

举一反三：search 默认用平凡算法（O(N*M)）。C++17 可传入执行策略和搜索器（如 Boyer-Moore）。对短 needle 平凡算法可能更快，长序列考虑 BM。

语法规则：find_end：最后一次出现

auto last = find_end(v.begin(), v.end(),
    sub.begin(), sub.end());
// 查找子序列最后一次出现的位置
// 返回最后一次匹配的起始迭代器

举一反三：find_end 类似于反向 search。实现通常从末尾向前搜索或先 search 再用 while 循环找最后一个。当有多个匹配时，只关心最后一个。

语法规则：search_n：连续重复

auto it = search_n(v.begin(), v.end(), count, value);
// 查找 count 个连续 value 的起始位置
// search_n 也可传自定义二元谓词（如近似相等）

举一反三：search_n 是 search 的特化版本。适合找连续出现的模式（如连续 N 天涨停、连续 N 个相同字符）。自定义谓词支持模糊匹配（如误差范围内相等）。
