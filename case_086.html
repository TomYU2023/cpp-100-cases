案例 86：命令行 TODO 管理器

一个完整的命令行 TODO 工具，支持添加、查看、完成、删除和持久化存储，涵盖文件 IO、CRUD、交互式菜单。

核心知识点：文件流 ofstream/ifstream、find_if+lambda、cin.ignore 缓冲区处理、struct 数据建模

完整代码（含逐行注释）

// 案例 86：命令行 TODO 管理器
// 一个完整的交互式命令行 TODO 工具，支持增删改查、标记完成、持久化存储。
// 涵盖文件 IO、CRUD 操作、交互式菜单、JSON 序列化等实战技能。

#include <iostream>
#include <vector>
#include <fstream>
#include <string>
#include <algorithm>
#include <iomanip>

// ─── 数据结构 ────────────────────────────────────
// 每条 TODO 记录的核心字段
struct Todo {
    int id;              // 唯一标识
    std::string task;    // 任务内容
    bool completed;      // 是否完成

    // 格式化打印单条记录（宽度对齐）
    void print() const {
        std::cout << std::setw(3) << id << ". "
                  << (completed ? "[✓]" : "[ ]") << " "
                  << task << std::endl;
    }
};

// ─── 全局状态 ────────────────────────────────────
std::vector<Todo> todos;          // 内存中的 TODO 列表
int nextId = 1;                   // 自增 ID
const std::string FILENAME = "todos.txt";  // 持久化文件

// ─── 持久化：保存到文件 ─────────────────────────
// 使用简单的行格式：id|task|completed
void save() {
    std::ofstream out(FILENAME);   // 打开文件（覆盖模式）
    if (!out) { std::cerr << "无法保存文件！" << std::endl; return; }
    for (const auto& t : todos)
        out << t.id << "|" << t.task << "|" << t.completed << "\n";
    std::cout << "已保存 " << todos.size() << " 条记录。" << std::endl;
}

// ─── 持久化：从文件加载 ─────────────────────────
void load() {
    std::ifstream in(FILENAME);
    if (!in) return;  // 文件不存在则跳过（首次运行）
    todos.clear();
    std::string line;
    while (std::getline(in, line)) {
        auto p1 = line.find('|');
        auto p2 = line.find('|', p1 + 1);
        if (p1 == std::string::npos || p2 == std::string::npos) continue;
        Todo t;
        t.id = std::stoi(line.substr(0, p1));
        t.task = line.substr(p1 + 1, p2 - p1 - 1);
        t.completed = (line.substr(p2 + 1) == "1");
        todos.push_back(t);
    }
    // 计算下一个可用 ID
    if (!todos.empty())
        nextId = todos.back().id + 1;
}

// ─── 核心操作 ────────────────────────────────────
void addTask(const std::string& task) {
    todos.push_back({nextId++, task, false});
    std::cout << "已添加 #" << nextId - 1 << std::endl;
}

void listTasks() {
    if (todos.empty()) { std::cout << "暂无任务。" << std::endl; return; }
    for (const auto& t : todos) t.print();
}

void completeTask(int id) {
    auto it = std::find_if(todos.begin(), todos.end(),
        [id](const Todo& t) { return t.id == id; });
    if (it != todos.end()) {
        it->completed = true;
        std::cout << "已标记 #" << id << " 完成！" << std::endl;
    } else {
        std::cout << "找不到 #" << id << std::endl;
    }
}

void deleteTask(int id) {
    auto it = std::find_if(todos.begin(), todos.end(),
        [id](const Todo& t) { return t.id == id; });
    if (it != todos.end()) {
        std::cout << "已删除: " << it->task << std::endl;
        todos.erase(it);  // vector::erase 会移动后续元素
    } else {
        std::cout << "找不到 #" << id << std::endl;
    }
}

// ─── 主程序：交互式循环 ─────────────────────────
int main() {
    load();  // 启动时加载已有数据

    while (true) {
        std::cout << "\n===== TODO 管理器 =====\n";
        std::cout << "1. 查看  2. 添加  3. 完成  4. 删除  5. 保存  0. 退出\n> ";

        int choice;
        std::cin >> choice;
        std::cin.ignore();  // 清除换行符，防止干扰 getline

        if (choice == 0) {
            std::cout << "退出前是否保存？(y/n): ";
            char c; std::cin >> c;
            if (c == 'y' || c == 'Y') save();
            break;
        }

        switch (choice) {
            case 1: listTasks(); break;
            case 2: {
                std::cout << "输入任务: ";
                std::string task;
                std::getline(std::cin, task);  // 读取含空格的整行
                if (!task.empty()) addTask(task);
                break;
            }
            case 3: {
                std::cout << "输入任务 ID: ";
                int id; std::cin >> id;
                completeTask(id);
                break;
            }
            case 4: {
                std::cout << "输入任务 ID: ";
                int id; std::cin >> id;
                deleteTask(id);
                break;
            }
            case 5: save(); break;
            default: std::cout << "无效选项" << std::endl;
        }
    }
    return 0;
}

语法规则：std::ofstream / std::ifstream 文件流

#include <fstream>
std::ofstream out("file.txt");       // 写入模式（覆盖）
if (!out) { /* 打开失败处理 */ }
out << "Hello" << std::endl;
out.close();  // 析构时自动关闭，也可手动

std::ifstream in("file.txt");        // 读取模式
std::string line;
while (std::getline(in, line)) { }   // 逐行读取

💡 举一反三：所有文件流对象离开作用域时自动关闭（RAII）。ofstream 默认覆盖，加 std::ios::app 切换为追加模式。ifstream 读取到文件尾时，getline 返回空流（转为 false），自然退出循环。

语法规则：std::find_if + lambda 条件查找

#include <algorithm>
auto it = std::find_if(vec.begin(), vec.end(),
    [](const T& item) { return item.id == target; });
if (it != vec.end()) { /* 找到了 */ }

💡 举一反三：find_if 传入一个谓词（返回 bool 的函数），返回第一个满足条件的迭代器。lambda 的捕获列表 [id] 将外部变量按值捕获进闭包。这是 STL 和 lambda 结合的经典模式，替代手写 for 循环。

语法规则：std::cin.ignore() 清除输入缓冲区

int n; std::cin >> n;
std::cin.ignore();  // 丢弃换行符
std::string line; std::getline(std::cin, line);

💡 举一反三：cin >> 读取数字后会在缓冲区留下换行符，如果不 ignore，紧接着的 getline 会读到空行。这是一个常见的输入 Bug，务必记住这个组合。
