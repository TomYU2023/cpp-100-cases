案例 86：std::search 子序列查找

使用 search 在序列中查找子序列，理解几种搜索算法的区别。

核心知识点：search、find_end、search_n、Boyer-Moore

完整代码（含逐行注释）

// 案例 86：std::search 子序列查找
// 在大序列中搜索子序列
#include <iostream>
#include <algorithm>
#include <vector>
#include <string>
#include <functional>
using namespace std;

int main() {
    // 1. search：查找子序列首次出现的位置
    vector<int> haystack = {1, 2, 3, 4, 1, 2, 3, 4, 5, 6};
    vector<int> needle = {3, 4, 5};

    auto it = search(haystack.begin(), haystack.end(),
                     needle.begin(), needle.end());
    if (it != haystack.end()) {
        cout << "子序列 [3,4,5] 首次在索引 "
             << distance(haystack.begin(), it) << " 处找到" << endl;
    }

    // 2. find_end：查找最后一次出现
    vector<int> needle2 = {1, 2};
    auto last = find_end(haystack.begin(), haystack.end(),
                         needle2.begin(), needle2.end());
    cout << "子序列 [1,2] 最后出现在索引 "
         << distance(haystack.begin(), last) << endl;

    // 3. search_n：查找连续 N 个相同值
    vector<int> data = {1, 2, 2, 2, 3, 4, 4, 4, 4, 5};
    auto pos = search_n(data.begin(), data.end(),
                        3, 2); // 连续 3 个 2
    cout << "\n连续 3 个 2 在索引 "
         << distance(data.begin(), pos) << " 处" << endl;

    pos = search_n(data.begin(), data.end(),
                   4, 4); // 连续 4 个 4
    cout << "连续 4 个 4 在索引 "
         << distance(data.begin(), pos) << " 处" << endl;

    // 4. 字符串搜索
    string text = "hello world, welcome to C++ world!";
    string pattern = "world";

    auto found = search(text.begin(), text.end(),
                        pattern.begin(), pattern.end());
    if (found != text.end()) {
        cout << "\n\"world\" 首次位置: "
             << distance(text.begin(), found) << endl;
        // 再次搜索
        auto found2 = search(found + 1, text.end(),
                             pattern.begin(), pattern.end());
        if (found2 != text.end()) {
            cout << "\"world\" 第二次位置: "
                 << distance(text.begin(), found2) << endl;
        }
    }

    return 0;
}

子序列 [1,2] 最后出现在索引 4

连续 3 个 2 在索引 1 处
连续 4 个 4 在索引 5 处

"world" 首次位置: 6
"world" 第二次位置: 28
