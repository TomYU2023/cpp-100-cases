案例 46：INI配置文件读写

实现一个 INI 格式配置文件的读取和写入工具，练习文件 I/O 和字符串解析。

核心知识点：文件读写、行解析、map 嵌套

完整代码（含逐行注释）

// 案例 46：INI 配置文件读写
// 解析标准 INI 格式配置文件
#include <iostream>
#include <fstream>
#include <string>
#include <map>
#include <algorithm>
using namespace std;

// [section] -> key=value
using IniData = map<string, map<string, string>>;

class IniParser {
private:
    IniData data;

    // 去除字符串首尾空格
    static string trim(const string& s) {
        size_t start = s.find_first_not_of(" \t\r\n");
        if (start == string::npos) return "";
        size_t end = s.find_last_not_of(" \t\r\n");
        return s.substr(start, end - start + 1);
    }

public:
    bool load(const string& filename) {
        ifstream file(filename);
        if (!file.is_open()) return false;

        string line;
        string currentSection;
        int lineNum = 0;

        while (getline(file, line)) {
            lineNum++;
            line = trim(line);

            // 跳过空行和注释行
            if (line.empty() || line[0] == ';' || line[0] == '#')
                continue;

            // 解析节 [section]
            if (line[0] == '[' && line.back() == ']') {
                currentSection = line.substr(1, line.size() - 2);
                continue;
            }

            // 解析键值对 key=value
            size_t eqPos = line.find('=');
            if (eqPos != string::npos && !currentSection.empty()) {
                string key = trim(line.substr(0, eqPos));
                string val = trim(line.substr(eqPos + 1));
                data[currentSection][key] = val;
            }
        }

        file.close();
        return true;
    }

    // 读取值
    string get(const string& section, const string& key,
               const string& defaultVal = "") const {
        auto secIt = data.find(section);
        if (secIt != data.end()) {
            auto keyIt = secIt->second.find(key);
            if (keyIt != secIt->second.end())
                return keyIt->second;
        }
        return defaultVal;
    }

    // 打印全部配置
    void print() const {
        for (const auto& [section, keys] : data) {
            cout << "[" << section << "]" << endl;
            for (const auto& [k, v] : keys) {
                cout << "  " << k << " = " << v << endl;
            }
        }
    }
};

int main() {
    // 创建测试文件
    ofstream out("test.ini");
    out << "[Database]\n"
        << "host=localhost\n"
        << "port=3306\n"
        << "name=mydb\n\n"
        << "[Server]\n"
        << "host=0.0.0.0\n"
        << "port=8080\n"
        << "timeout=30\n";
    out.close();

    IniParser parser;
    if (parser.load("test.ini")) {
        parser.print();
        cout << "\nDatabase.host = "
             << parser.get("Database", "host") << endl;
        cout << "Server.port = "
             << parser.get("Server", "port") << endl;
    }

    return 0;
}

语法规则：文件读取：ifstream

#include <fstream>
ifstream file("config.ini");
if (!file.is_open()) { /* 错误处理 */ }
string line;
while (getline(file, line)) { /* 处理每一行 */ }
file.close();

举一反三：ifstream 用于读取文件。is_open() 检查是否成功打开。getline 逐行读取。RAII 可让文件在析构时自动关闭，但显式 close 更明确。

语法规则：嵌套 map 数据结构

// map<section_name, map<key, value>>
map<string, map<string, string>> ini;
ini["Database"]["host"] = "localhost";

举一反三：嵌套 map 适合表达层级结构。访问时逐层检查存在性。可用 emplace 减少中间临时对象。深度嵌套建议封装类提供安全访问接口。

语法规则：查找并忽略注释

if (line[0] == ';' || line[0] == '#') continue;
// 跳过以 ; 或 # 开头的注释行
if (line.empty()) continue;
// 跳过空行

举一反三：先 trim 再判断可以正确处理前导空格后的注释。注意检查 line 非空再访问 line[0]。结构化格式（如 INI）通常都有注释和空行处理。
