案例 46：INI配置文件读写

实现一个 INI 格式配置文件的读取和写入工具，练习文件 I/O 和字符串解析。

核心知识点：文件读写、行解析、map 嵌套

完整代码（含逐行注释）

// 案例 46：INI 配置文件读写
// 解析标准 INI 格式配置文件
#include <iostream>
#include <fstream>
#include <string>
#include <map>
#include <algorithm>
using namespace std;

// [section] -> key=value
using IniData = map<string, map<string, string>>;

class IniParser {
private:
    IniData data;

    // 去除字符串首尾空格
    static string trim(const string& s) {
        size_t start = s.find_first_not_of(" \t\r\n");
        if (start == string::npos) return "";
        size_t end = s.find_last_not_of(" \t\r\n");
        return s.substr(start, end - start + 1);
    }

public:
    bool load(const string& filename) {
        ifstream file(filename);
        if (!file.is_open()) return false;

        string line;
        string currentSection;
        int lineNum = 0;

        while (getline(file, line)) {
            lineNum++;
            line = trim(line);

            // 跳过空行和注释行
            if (line.empty() || line[0] == ';' || line[0] == '#')
                continue;

            // 解析节 [section]
            if (line[0] == '[' && line.back() == ']') {
                currentSection = line.substr(1, line.size() - 2);
                continue;
            }

            // 解析键值对 key=value
            size_t eqPos = line.find('=');
            if (eqPos != string::npos && !currentSection.empty()) {
                string key = trim(line.substr(0, eqPos));
                string val = trim(line.substr(eqPos + 1));
                data[currentSection][key] = val;
            }
        }

        file.close();
        return true;
    }

    // 读取值
    string get(const string& section, const string& key,
               const string& defaultVal = "") const {
        auto secIt = data.find(section);
        if (secIt != data.end()) {
            auto keyIt = secIt->second.find(key);
            if (keyIt != secIt->second.end())
                return keyIt->second;
        }
        return defaultVal;
    }

    // 打印全部配置
    void print() const {
        for (const auto& [section, keys] : data) {
            cout << "[" << section << "]" << endl;
            for (const auto& [k, v] : keys) {
                cout << "  " << k << " = " << v << endl;
            }
        }
    }
};

int main() {
    // 创建测试文件
    ofstream out("test.ini");
    out << "[Database]\n"
        << "host=localhost\n"
        << "port=3306\n"
        << "name=mydb\n\n"
        << "[Server]\n"
        << "host=0.0.0.0\n"
        << "port=8080\n"
        << "timeout=30\n";
    out.close();

    IniParser parser;
    if (parser.load("test.ini")) {
        parser.print();
        cout << "\nDatabase.host = "
             << parser.get("Database", "host") << endl;
        cout << "Server.port = "
             << parser.get("Server", "port") << endl;
    }

    return 0;
}
