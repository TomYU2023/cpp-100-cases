案例 22：map 电话号码簿

使用 std::map 实现一个电话号码簿，支持添加、查找、删除、列表功能，练习关联容器的使用。

核心知识点：map 增删查、迭代器、结构化绑定

完整代码（含逐行注释）

// 案例 22：map 电话号码簿
// 使用 std::map 管理姓名-电话映射
#include <iostream>
#include <map>
#include <string>
using namespace std;

class PhoneBook {
private:
    map<string, string> book; // 姓名 -> 电话号码

public:
    // 添加或更新联系人
    void add(const string& name, const string& phone) {
        auto result = book.insert({name, phone});
        if (!result.second) {
            // 已存在则更新
            book[name] = phone;
            cout << "已更新 " << name << endl;
        } else {
            cout << "已添加 " << name << endl;
        }
    }

    // 查找联系人
    void find(const string& name) const {
        auto it = book.find(name);
        if (it != book.end()) {
            cout << name << " -> " << it->second << endl;
        } else {
            cout << "未找到 " << name << endl;
        }
    }

    // 删除联系人
    void remove(const string& name) {
        if (book.erase(name)) {
            cout << "已删除 " << name << endl;
        } else {
            cout << name << " 不存在" << endl;
        }
    }

    // 列出所有联系人（map 自动按 key 排序）
    void listAll() const {
        if (book.empty()) {
            cout << "电话簿为空" << endl;
            return;
        }
        cout << "=== 电话簿（按姓名排序）===" << endl;
        for (const auto& [name, phone] : book) {
            cout << "  " << name << " : " << phone << endl;
        }
    }
};

int main() {
    PhoneBook pb;

    pb.add("张三", "13800138000");
    pb.add("李四", "13900139000");
    pb.add("王五", "13700137000");
    pb.add("张三", "13600136000"); // 重复姓名，更新号码

    pb.find("李四");
    pb.find("赵六");

    pb.listAll();

    pb.remove("王五");
    pb.listAll();

    return 0;
}
