6 // 反转字符串（返回新字符串，不修改原串）
7 string reverseStr(const string& s) { const string& — 常量引用，避免拷贝且保证不修改原串
8 string result = s; // string 支持直接用 = 复制 string 类型重载了 operator=，可以直接赋值来复制字符串

13 // 查找子串出现次数
14 int countOccurrences(const string& text, const string& pattern) {
15 if (pattern.empty()) return 0;
16 int count = 0; size_t 是无符号整型，用于表示大小和索引
17 size_t pos = 0;
18 // find 返回找到的位置，找不到返回 string::npos
19 while ((pos = text.find(pattern, pos)) != string::npos) { find(pattern, pos) 从 pos 位置开始查找，返回首次出现的位置
20 count++; string::npos 是 find 找不到时返回的特殊值（通常为 size_t 的最大值）
21 pos += pattern.length(); // 跳过已匹配的部分
22 } pattern.length() 返回字符串的长度
23 return count;
24 }

26 // 替换所有匹配子串
27 string replaceAll(string text, const string& from, const string& to) {
28 if (from.empty()) return text;
29 size_t pos = 0; replace(pos, len, str) 从 pos 开始删除 len 个字符，然后插入 str
30 while ((pos = text.find(from, pos)) != string::npos) {
31 text.replace(pos, from.length(), to);
32 pos += to.length(); // 跳过新插入的文本
33 }
34 return text;
35 }

新语法讲解

std::string 常用操作

C++ 的 string 类型远比 C 风格字符串安全方便。常用操作：`s.length()` / `s.size()` 获取长度；`s.empty()` 判空；`s.clear()` 清空；`s += t` 拼接；`s == t` 比较。
string a = "Hello";
a += " World";       // "Hello World"
if (a == "Hello") ... // 直接用 == 比较内容

引用 &

引用是变量的别名，`int& r = x` 之后对 r 的操作等价于对 x 的操作。引用必须在定义时初始化，且不能改变绑定。const 引用可以绑定到临时对象，延长其生命周期。
int x = 10;
int& r = x;   // r 是 x 的别名
r = 20;       // x 也变成 20
const int& cr = 42;  // 合法，const引用绑定临时值

size_t 类型

`size_t` 定义在 `` 中，是 `sizeof` 运算符的返回类型，一个无符号整数。用于表示大小、计数、索引。注意：`size_t` 不会为负数，循环减到负数会回绕。
for (size_t i = v.size(); i-- > 0; )  // 安全反向遍历
    cout << v[i];
