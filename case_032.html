案例 32：next_permutation

使用 next_permutation 生成序列的所有排列，学习排列枚举和字典序的原理。

核心知识点：next_permutation、字典序、do-while

完整代码（含逐行注释）

// 案例 32：next_permutation
// 生成序列的所有排列
#include <iostream>
#include <vector>
#include <algorithm>
#include <string>
using namespace std;

int main() {
    // 1. 字符串的全排列
    string s = "123";
    cout << "\"123\" 的全排列：" << endl;

    // 必须先从最小字典序开始！
    sort(s.begin(), s.end());

    int count = 0;
    do {
        cout << ++count << ": " << s << endl;
    } while (next_permutation(s.begin(), s.end()));

    cout << "共 " << count << " 种排列" << endl;

    // 2. 数组部分排列（前 3 个位置）
    cout << "\n--- 数组部分排列 ---" << endl;
    vector<int> nums = {1, 2, 3, 4, 5};
    // 只排列前 3 个元素
    count = 0;
    do {
        cout << ++count << ": ";
        for (int n : nums) cout << n << " ";
        cout << endl;
    } while (next_permutation(nums.begin(), nums.begin() + 3));
    // 注意：排列前 3 个时，后 2 个保持不变

    // 3. prev_permutation：上一个排列
    string t = "321";
    cout << "\n\"321\" 的逆序排列：" << endl;
    do {
        cout << t << " ";
    } while (prev_permutation(t.begin(), t.end()));
    cout << endl;

    return 0;
}
