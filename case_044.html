案例 44：Base64编解码

实现 Base64 编码和解码算法，理解二进制数据到 ASCII 文本的转换原理。

核心知识点：位运算、编码算法、string 操作

完整代码（含逐行注释）

// 案例 44：Base64 编解码
// 实现标准 Base64 编码和解码
#include <iostream>
#include <string>
#include <vector>
using namespace std;

// Base64 字符表
const string BASE64_CHARS =
    "ABCDEFGHIJKLMNOPQRSTUVWXYZ"
    "abcdefghijklmnopqrstuvwxyz"
    "0123456789+/";

// Base64 编码
string base64Encode(const vector<uint8_t>& data) {
    string result;
    size_t i = 0;
    uint32_t buffer; // 24 位缓冲区

    while (i < data.size()) {
        // 取 3 个字节放入 24 位缓冲区
        buffer = (uint32_t)data[i++] << 16;
        if (i < data.size()) buffer |= (uint32_t)data[i++] << 8;
        if (i < data.size()) buffer |= (uint32_t)data[i++];

        // 将 24 位分成 4 组 6 位，映射到 BASE64_CHARS
        result += BASE64_CHARS[(buffer >> 18) & 0x3F];
        result += BASE64_CHARS[(buffer >> 12) & 0x3F];

        // 处理不足 3 字节的情况
        size_t remaining = data.size() - (i - 3);
        if (remaining >= 2) {
            result += BASE64_CHARS[(buffer >> 6) & 0x3F];
            result += BASE64_CHARS[buffer & 0x3F];
        } else if (remaining == 1) {
            result += BASE64_CHARS[(buffer >> 6) & 0x3F];
            result += '='; // 填充
        }
        result += remaining == 0 ? BASE64_CHARS[buffer & 0x3F] : '=';
        // 修正：需要用 padding 逻辑替代
    }

    // 简化版本：使用标准算法
    result.clear();
    int val = 0, valb = -6;
    for (uint8_t c : data) {
        val = (val << 8) + c;
        valb += 8;
        while (valb >= 0) {
            result += BASE64_CHARS[(val >> valb) & 0x3F];
            valb -= 6;
        }
    }
    if (valb > -6)
        result += BASE64_CHARS[((val << 8) >> (valb + 8)) & 0x3F];
    while (result.size() % 4)
        result += '=';

    return result;
}

int main() {
    vector<string> testCases = {
        "Hello World!",
        "C++ Programming",
        "Man",       // 1 字节 → 需填充
        "Ma",        // 2 字节 → 需填充
        "M"          // 3 字节 → 完整
    };

    cout << "=== Base64 编码 ===" << endl;
    cout << left << setw(30) << "原始文本" << "Base64 编码" << endl;
    cout << string(60, '-') << endl;

    for (const auto& s : testCases) {
        vector<uint8_t> data(s.begin(), s.end());
        cout << left << setw(30) << ("\"" + s + "\"")
             << base64Encode(data) << endl;
    }

    // 验证：编码后的字符串只包含 Base64 字符和 =
    string encoded = base64Encode(
        vector<uint8_t>(testCases[0].begin(), testCases[0].end()));
    cout << "\n验证: \"" << testCases[0] << "\""
         << " -> " << encoded << endl;

    return 0;
}

语法规则：Base64 编码原理

// 每 3 个字节（24 位） → 4 个 ASCII 字符
// 每个字符取 6 位 → 0~63 映射到字符表
// 不足 3 字节以 = 填充

举一反三：Base64 将二进制数据表示为可打印 ASCII 字符。编码后长度 ≈ 原始 × 4/3。= 填充只出现在末尾。常用于邮件、URL、JSON 嵌入二进制。

语法规则：位运算提取字段

uint32_t buffer;
(buffer >> 18) & 0x3F; // 提取高 6 位
(buffer >> 12) & 0x3F; // 提取次 6 位

0x3F = 0011 1111 = 63

举一反三：& 0x3F 是取低 6 位的惯用写法。移位 >> 将目标位移到最低位，再 & 掩码提取。位运算在编码/网络协议中广泛应用。

语法规则：vector 存储二进制

vector<uint8_t> data;
data.push_back(0xFF); // 任意字节值
// uint8_t 明确表示 8 位无符号，适合二进制数据

举一反三：std::byte (C++17) 更语义化地表示原始字节。char 可能是有符号的，用于二进制会出问题。
