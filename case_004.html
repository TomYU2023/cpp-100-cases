案例 4：素数/回文/完数

实现三个数学判断函数：判断素数、判断回文数、判断完数，综合练习函数封装和算法逻辑。

核心知识点：函数定义与调用、sqrt 优化、字符串反转

完整代码（含逐行注释）

// 案例 4：素数 / 回文 / 完数判断
// 综合练习数学与算法函数
#include <iostream>
#include <cmath>     // sqrt()
#include <string>
#include <algorithm> // reverse()
using namespace std;

// 判断素数：只能被 1 和自己整除
bool isPrime(int n) {
    if (n < 2) return false;
    // 只需检查到 sqrt(n)，大幅减少循环次数
    for (int i = 2; i <= sqrt(n); i++) {
        if (n % i == 0) return false;
    }
    return true;
}

// 判断回文数：正读反读都一样，如 121
bool isPalindrome(int n) {
    string s = to_string(n);          // 转为字符串
    string rev = s;
    reverse(rev.begin(), rev.end());  // 反转字符串
    return s == rev;
}

// 判断完数：所有真因子之和等于自身，如 6 = 1+2+3
bool isPerfect(int n) {
    if (n < 2) return false;
    int sum = 1; // 1 是所有数的因子
    for (int i = 2; i <= sqrt(n); i++) {
        if (n % i == 0) {
            sum += i;
            if (i != n / i) sum += n / i; // 成对加入
        }
    }
    return sum == n;
}

int main() {
    int num;
    cout << "请输入一个正整数：";
    cin >> num;

    if (isPrime(num))
        cout << num << " 是素数" << endl;
    else
        cout << num << " 不是素数" << endl;

    if (isPalindrome(num))
        cout << num << " 是回文数" << endl;
    else
        cout << num << " 不是回文数" << endl;

    if (isPerfect(num))
        cout << num << " 是完数" << endl;
    else
        cout << num << " 不是完数" << endl;

    return 0;
}

bool isPrime(int n) {
    // 函数体
    return true;
}
// 调用
if (isPrime(17)) { /* ... */ }

举一反三：将功能封装为函数提高复用性。函数声明放在调用之前，或用头文件 + 前向声明。

语法规则：sqrt() 优化循环

#include <cmath>
for (int i = 2; i <= sqrt(n); i++) { /* ... */ }

举一反三：找 n 的所有因子只需遍历到 sqrt(n)，另一半因子自动获得。时间复杂度从 O(n) 降到 O(√n)。

语法规则：字符串与数字互转

string s = to_string(123);
int n = stoi("456");

举一反三：C++11 提供 to_string() 和 stoi/stod 系列函数。还可使用 stringstream 进行复杂格式化转换。
