案例 65：std::variant 类型安全联合

使用 variant 替代 union，实现类型安全的联合体。

核心知识点：variant、visit、get、holds_alternative

完整代码（含逐行注释）

// 案例 65：std::variant 类型安全联合
// 安全的多类型联合体
#include <iostream>
#include <variant>
#include <string>
#include <vector>
#include <cmath>
using namespace std;

// 可表示整数、浮点数或字符串的值
using Value = variant<int, double, string>;

// 使用 visit 处理 variant
void printValue(const Value& v) {
    visit([](const auto& val) {
        using T = decay_t<decltype(val)>;
        if constexpr (is_same_v<T, int>) {
            cout << "整数: " << val << endl;
        } else if constexpr (is_same_v<T, double>) {
            cout << "浮点数: " << fixed << setprecision(2) << val << endl;
        } else if constexpr (is_same_v<T, string>) {
            cout << "字符串: \"" << val << "\"" << endl;
        }
    }, v);
}

// 对 variant 执行运算
Value compute(Value a, Value b, char op) {
    return visit([op](auto&& x, auto&& y) -> Value {
        using T = common_type_t<decay_t<decltype(x)>,
                                 decay_t<decltype(y)>>;
        if constexpr (is_arithmetic_v<T>) {
            switch (op) {
                case '+': return T(x + y);
                case '-': return T(x - y);
                case '*': return T(x * y);
                case '/': return (y != 0) ? Value(T(x / y))
                                          : Value(string("除零错误"));
            }
        }
        return Value(string("不支持的操作"));
    }, a, b);
}

int main() {
    // 创建不同类型的值
    Value v1 = 42;
    Value v2 = 3.14;
    Value v3 = string("Hello");

    printValue(v1);
    printValue(v2);
    printValue(v3);

    // 检查当前存储的类型
    cout << "\nv1 是 int? " << holds_alternative<int>(v1) << endl;
    cout << "v1 是 double? " << holds_alternative<double>(v1) << endl;

    // 获取值
    cout << "get<int>(v1) = " << get<int>(v1) << endl;

    // 计算
    auto result = compute(10, 3.5, '+');
    printValue(result);

    result = compute(string("hi"), string("!"), '+');
    printValue(result);

    return 0;
}

Variant vs Union vs Any 对比：
  Variant:  类型安全，编译期检查，无堆分配，访问必须处理所有可能类型
  Union:    无类型标记，需要手动跟踪，无运行时安全检查
  Any:      可存任意类型，有堆分配，有运行时开销
