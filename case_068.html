案例 68：结构化绑定 structured binding

使用 C++17 结构化绑定简化 tuple、pair、struct 的元素访问。

核心知识点：结构化绑定、auto [a,b]、tuple、map 遍历

完整代码（含逐行注释）

// 案例 68：结构化绑定
// C++17 结构化绑定简化多值返回和 map 遍历
#include <iostream>
#include <tuple>
#include <map>
#include <string>
#include <vector>
using namespace std;

// 返回多个值
tuple<string, int, double> getStudentInfo() {
    return {"Alice", 25, 95.5};
}

struct Point {
    double x, y, z;
};

int main() {
    // 1. 绑定元组
    auto [name, age, score] = getStudentInfo();
    cout << "姓名: " << name << ", 年龄: " << age
         << ", 分数: " << score << endl;

    // 2. 绑定 pair（map 遍历）
    map<string, int> scores = {
        {"Alice", 95}, {"Bob", 88}, {"Charlie", 92}
    };
    cout << "\n成绩表:" << endl;
    for (const auto& [student, grade] : scores) {
        cout << "  " << student << ": " << grade << endl;
    }

    // 3. 绑定结构体成员
    Point p{1.0, 2.0, 3.0};
    auto [x, y, z] = p;
    cout << "\n坐标: (" << x << ", " << y << ", " << z << ")" << endl;

    // 4. 绑定引用以修改原始值
    Point origin{0, 0, 0};
    auto& [ox, oy, oz] = origin;
    ox = 5.0; // 修改 origin.x
    cout << "修改后: (" << origin.x << ", " << origin.y << ", "
         << origin.z << ")" << endl;

    // 5. 在范围 for 中的结构化绑定
    vector<pair<string, int>> items = {
        {"apple", 3}, {"banana", 5}, {"orange", 2}
    };
    cout << "\n物品列表:" << endl;
    for (const auto& [item, quantity] : items) {
        cout << "  " << item << " x " << quantity << endl;
    }

    return 0;
}

运行结果

姓名: Alice, 年龄: 25, 分数: 95.5

成绩表:
  Alice: 95
  Bob: 88
  Charlie: 92

坐标: (1, 2, 3)
修改后: (5, 0, 0)

物品列表:
  apple x 3
  banana x 5
  orange x 2
