案例 81：插件系统（动态加载 DLL）

实现插件管理器，支持运行时加载 DLL 并调用其中的插件，理解动态库加载与面向接口编程。

核心知识点：抽象基类、extern "C"、LoadLibrary/GetProcAddress、unique_ptr 自定义删除器、工厂模式

完整代码（含逐行注释）

#include <iostream>
#include <string>
#include <functional>
#include <memory>

// Windows 下动态加载 DLL 的 API
#ifdef _WIN32
#include <windows.h>
#define LOAD_LIB(path)    LoadLibraryA(path)
#define GET_FUNC(lib, name) GetProcAddress(lib, name)
#define FREE_LIB(lib)     FreeLibrary(lib)
#else
#include <dlfcn.h>
#define LOAD_LIB(path)    dlopen(path, RTLD_LAZY)
#define GET_FUNC(lib, name) dlsym(lib, name)
#define FREE_LIB(lib)     dlclose(lib)
#endif

// ========== 1. 插件接口定义 ==========
// 语法规则：纯虚函数 + = 0 — 声明接口
// virtual void func() = 0; 表示该函数是纯虚函数，子类必须实现。
// 含有纯虚函数的类叫"抽象类"，不能实例化。
struct IPlugin {
    virtual ~IPlugin() = default; // 虚析构保证正确释放子类资源
    virtual const char* name() const = 0;   // 插件名称
    virtual double execute(double x) = 0;   // 执行插件逻辑
};

// ========== 2. 插件管理器（RAII 封装） ==========
// 语法规则：std::unique_ptr 自定义删除器
// std::unique_ptr<T, Deleter> 第二个模板参数指定删除器类型。
// 删除器是一个可调用对象，在 unique_ptr 销毁时调用。
// 默认删除器是 std::default_delete<T>，即调用 delete。
class PluginManager {
    using PluginHandle = std::unique_ptr<std::remove_pointer_t<HMODULE>,
          std::function<void(HMODULE)>>;
    // std::remove_pointer_t<HMODULE> 移除指针得到 HMODULE__ 结构体类型
    // 自定义删除器：std::function<void(HMODULE)>，内部调用 FreeLibrary

    std::vector<PluginHandle> handles_;    // 动态库句柄
    std::vector<std::unique_ptr<IPlugin>> plugins_; // 插件实例

public:
    // 加载插件：加载 DLL → 获取工厂函数 → 创建插件实例
    bool load(const std::string& dllPath) {
        // 1. 加载动态库
        // 语法规则：LoadLibraryA(path) — Windows API，加载 DLL 到进程地址空间
        // 返回 HMODULE 句柄，失败返回 NULL。
        HMODULE h = LOAD_LIB(dllPath.c_str());
        if (!h) {
            std::cerr << "加载 DLL 失败: " << dllPath << "\n";
            return false;
        }

        // 2. 用 unique_ptr + 自定义删除器管理句柄（RAII）
        // lambda 删除器：在句柄离开作用域时自动调用 FreeLibrary
        handles_.emplace_back(h, [](HMODULE m) { FREE_LIB(m); });
        // emplace_back 直接构造对象，避免临时对象的拷贝

        // 3. 获取工厂函数
        // 语法规则：GetProcAddress(hMod, funcName) — 获取 DLL 中导出函数的地址
        // 返回 FARPROC（函数指针），需要用 reinterpret_cast 转成目标类型
        using FactoryFunc = IPlugin* (*)(); // 函数指针类型定义
        auto factory = reinterpret_cast<FactoryFunc>(
            GET_FUNC(h, "createPlugin"));

        if (!factory) {
            std::cerr << "未找到 createPlugin 函数\n";
            return false;
        }

        // 4. 调用工厂函数创建插件
        plugins_.push_back(std::unique_ptr<IPlugin>(factory()));
        std::cout << "插件加载成功: " << plugins_.back()->name() << "\n";
        return true;
    }

    void runAll(double input) {
        for (auto& p : plugins_) {
            std::cout << p->name() << "(" << input << ") = "
                      << p->execute(input) << "\n";
        }
    }
};

// ========== 3. 模拟一个插件的代码（正常应该在单独的 DLL 项目中） ==========
// 为了演示，我们将插件逻辑内联；真实场景中下面代码在另一个 DLL 项目里
class DoublePlugin : public IPlugin {
public:
    const char* name() const override { return "Double"; }
    double execute(double x) override { return x * 2; }
};

// 导出函数：DLL 必须提供 createPlugin 和 destroyPlugin
// 语法规则：extern "C" — 告诉编译器使用 C 链接方式（不进行 C++ 名称修饰）
// 没有 extern "C"，C++ 编译器会对函数名进行 name mangling（如 createPlugin → ?createPlugin@@YA...）
// GetProcAddress 按原始名称查找，所以必须 extern "C" 禁用修饰。
extern "C" {
    __declspec(dllexport) IPlugin* createPlugin() { return new DoublePlugin(); }
    __declspec(dllexport) void destroyPlugin(IPlugin* p) { delete p; }
}

int main() {
    std::cout << "=== 插件系统演示 ===\n\n";

    // 核心概念：主程序只依赖接口（IPlugin），不依赖具体实现
    // 新功能通过加载 DLL 来扩展，无需重新编译主程序

    PluginManager pm;

    // 模拟：假设 DoublePlugin 编译成了 double_plugin.dll
    // pm.load("double_plugin.dll");

    // 直接演示：手动创建插件并调用
    DoublePlugin dp;
    std::cout << "直接演示: " << dp.name() << "(" << 5.0 << ") = "
              << dp.execute(5.0) << "\n";

    std::cout << "\n设计要点：\n"
              << "1. 接口与实现分离（面向接口编程）\n"
              << "2. 工厂函数模式创建实例\n"
              << "3. RAII 管理动态库生命周期\n"
              << "4. extern \"C\" 确保函数名可查找\n";
    return 0;
}

语法规则：纯虚函数与抽象基类

struct IShape {
    virtual double area() const = 0; // 纯虚函数
    virtual ~IShape() = default;
};
struct Circle : IShape {
    double r;
    double area() const override { return 3.14 * r * r; }
};

💡 举一反三：= 0 标记纯虚函数，子类必须 override。写接口（API 设计）时大量使用。类似于 Java 的 interface 或 Rust 的 trait。

语法规则：extern "C" — C 链接约定

extern "C" {
    void my_c_function(int x);  // C 链接，不进行名称修饰
    int add(int a, int b);       // 能被 C 代码或 dlsym 找到
}

💡 举一反三：C++ 编译器会对函数名进行 name mangling（编码参数类型到函数名中）。extern "C" 告诉编译器用 C 风格的纯函数名，方便动态加载和跨语言调用。

语法规则：LoadLibrary / GetProcAddress / FreeLibrary

HMODULE h = LoadLibraryA("mylib.dll");        // 加载 DLL
using FuncType = int(*)(int, int);                // 函数指针类型
FuncType add = (FuncType)GetProcAddress(h, "add"); // 获取函数
int result = add(3, 4);                           // 调用
FreeLibrary(h);                                   // 释放

💡 举一反三：Windows 动态加载三部曲。Linux 对应 dlopen/dlsym/dlclose。注意函数指针类型必须与实际签名一致，否则行为未定义。

语法规则：std::unique_ptr 自定义删除器

// 自定义删除器示例：关闭文件句柄
auto fileDeleter = [](FILE* f) { if(f) fclose(f); };
std::unique_ptr<FILE, decltype(fileDeleter)> fp(fopen("t.txt","r"), fileDeleter);
// fp 离开作用域时自动调用 fclose

💡 举一反三：当资源释放不是简单的 delete 时（如句柄、socket、文件），使用自定义删除器。删除器类型作为第二个模板参数。
