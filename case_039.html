condition_variable 生产者消费者

完整代码（逐行注释）

1 #include <iostream>
2 #include <thread>
3 #include <mutex>
4 #include <condition_variable> 条件变量
5 #include <queue>
6 #include <string>
7 #include <chrono>
8 using namespace std;
9
10 // 线程安全的生产者-消费者队列
11 template<typename T>
12 class ProducerConsumerQueue {
13 queue<T> q_;
14 mutable mutex mtx_;
15 condition_variable cv_;
16 size_t maxSize_; 队列容量上限
17 public:
18 ProducerConsumerQueue(size_t maxSize = 10) : maxSize_(maxSize) {}
19
20 void produce(T item) { 生产者
21 unique_lock<mutex> lock(mtx_);
22 cv_.wait(lock, [this]() { return q_.size() < maxSize_; }); 队列未满才生产
23 q_.push(item);
24 cout << "生产: " << item << " (队列大小: " << q_.size() << ")\n";
25 cv_.notify_one(); 通知一个消费者
26 }
27
28 T consume() { 消费者
29 unique_lock<mutex> lock(mtx_);
30 cv_.wait(lock, [this]() { return !q_.empty(); }); 队列非空才消费
31 T item = q_.front();
32 q_.pop();
33 cout << "消费: " << item << " (队列大小: " << q_.size() << ")\n";
34 cv_.notify_one(); 通知一个生产者（有空位了）
35 return item;
36 }
37 };
38
39 int main() {
40 ProducerConsumerQueue<int> pcq(5); 容量为 5
41
42 // 生产者线程
43 thread producer([&]() {
44 for (int i = 1; i <= 20; i++) {
45 pcq.produce(i);
46 this_thread::sleep_for(100ms); 生产速度
47 }
48 });
49
50 // 消费者线程
51 thread consumer([&]() {
52 for (int i = 0; i < 20; i++) {
53 int val = pcq.consume();
54 this_thread::sleep_for(150ms); 消费速度慢于生产，队列会填满
55 }
56 });
57
58 producer.join();
59 consumer.join();
60 cout << "处理完毕!" << endl;
61 return 0;
62 }

新语法讲解

condition_variable 条件变量

解决线程间的等待/通知问题。消费者在队列空时等待，生产者在放入数据后通知。关键模式：cv.wait(lock, predicate)——等价于 while(!predicate()) cv.wait(lock)，自动处理虚假唤醒。

举一反三

// 标准模式
unique_lock lock(mtx);
cv.wait(lock, []{ return condition; });
// 等价于:
// while(!condition) cv.wait(lock);
// 带谓词的 wait 更安全，自动处理虚假唤醒

notify_one vs notify_all

notify_one()：唤醒一个等待线程（随机），适合只生产一个数据时用。notify_all()：唤醒所有等待线程，适合状态变化可能满足多个线程时用（如析构时通知所有线程退出）。

举一反三

// 修改共享状态后通知
{
    lock_guard lock(mtx_);
    ready_ = true;
}
cv_.notify_all(); // 通知所有等待 ready_==true 的线程

举一反三

带超时的 wait (避免永久阻塞)

if (cv_.wait_for(lock, 5s, []{ return !q_.empty(); })) {
  // 有数据
} else {
  // 超时，做清理或退出
}

wait_for / wait_until 设置最大等待时间

用原子变量替代条件变量（简单场景）

atomic ready{false};
// 消费者: while(!ready.load()) this_thread::yield();
// 生产者: ready.store(true);
// 简单但浪费 CPU（忙等待）

简单标志用 atomic，复杂条件用 condition_variable
