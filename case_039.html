案例 39：condition_variable 生产消费

使用 condition_variable 实现经典的生产者-消费者模式，练习线程间通信和同步。

核心知识点：condition_variable、unique_lock、wait/notify

完整代码（含逐行注释）

// 案例 39：condition_variable 生产者-消费者
// 经典多线程同步模式
#include <iostream>
#include <thread>
#include <mutex>
#include <condition_variable>
#include <queue>
#include <chrono>
using namespace std;

class ProducerConsumer {
private:
    queue<int> buffer;              // 共享缓冲区
    const size_t maxSize;           // 缓冲区最大容量
    mutex mtx;                      // 互斥锁
    condition_variable cvProducer;  // 生产者条件变量
    condition_variable cvConsumer;  // 消费者条件变量
    bool done;                      // 生产完成标志

public:
    explicit ProducerConsumer(size_t max) : maxSize(max), done(false) {}

    // 生产者：往缓冲区放数据
    void produce(int id, int count) {
        for (int i = 0; i < count; i++) {
            unique_lock<mutex> lock(mtx);

            // 缓冲区满则等待消费者消费
            cvProducer.wait(lock, [this]() {
                return buffer.size() < maxSize;
            });

            int item = id * 1000 + i; // 生成产品编号
            buffer.push(item);

            cout << "生产者" << id << " 生产: " << item
                 << " (缓冲区: " << buffer.size() << ")" << endl;

            lock.unlock();
            cvConsumer.notify_one(); // 通知消费者
        }
    }

    // 消费者：从缓冲区取数据
    void consume(int id) {
        while (true) {
            unique_lock<mutex> lock(mtx);

            // 缓冲区为空时等待（但生产完成且有数据时需退出）
            cvConsumer.wait(lock, [this]() {
                return !buffer.empty() || done;
            });

            if (buffer.empty() && done) break;

            int item = buffer.front();
            buffer.pop();

            cout << "消费者" << id << " 消费: " << item
                 << " (缓冲区: " << buffer.size() << ")" << endl;

            lock.unlock();
            cvProducer.notify_one(); // 通知生产者有空位
        }
    }

    void setDone() {
        lock_guard<mutex> lock(mtx);
        done = true;
        cvConsumer.notify_all();
    }
};

int main() {
    ProducerConsumer pc(5); // 缓冲区容量为 5

    // 2 个生产者，各生产 8 个产品
    thread p1(&ProducerConsumer::produce, &pc, 1, 8);
    thread p2(&ProducerConsumer::produce, &pc, 2, 8);

    // 2 个消费者
    thread c1(&ProducerConsumer::consume, &pc, 1);
    thread c2(&ProducerConsumer::consume, &pc, 2);

    p1.join();
    p2.join();

    // 生产者完成后通知消费者退出
    pc.setDone();

    c1.join();
    c2.join();

    cout << "生产-消费完成" << endl;
    return 0;
}

语法规则：condition_variable::wait

unique_lock<mutex> lock(mtx);
cv.wait(lock, []() { return condition; });
// 等价于: while (!condition) cv.wait(lock);

举一反三：wait 在条件不满足时释放锁并阻塞，被唤醒后重新获取锁并检查条件。lambda 版本可防止虚假唤醒。注意：wait 必须配合 unique_lock。

语法规则：notify_one vs notify_all

cv.notify_one(); // 唤醒一个等待线程
cv.notify_all(); // 唤醒所有等待线程

举一反三：notify_one 适用于一个产品唤醒一个消费者的场景。notify_all 适用于状态改变影响多个线程的场景（如 setDone）。通知前通常先释放锁。

语法规则：生产者-消费者模式要点

// 1. 使用两个条件变量：一个通知「缓冲区有空间」，另一个通知「缓冲区有数据」
// 2. 消费者退出条件：缓冲区为空 且 生产已完成
// 3. 尽量在锁外处理业务逻辑

举一反三：单条件变量可行但性能差（每次 notify 会唤醒错误的消费者/生产者）。用两个条件变量可精准通知，减少不必要的线程切换。
