案例 66：std::span 视图

使用 C++20 span 实现非拥有型数组视图，零拷贝引用连续数据。

核心知识点：span、零拷贝引用、subspan、连续内存

完整代码（含逐行注释）

// 案例 66：std::span 视图
// 非拥有型的数组/容器视图
#include <iostream>
#include <span>
#include <vector>
#include <array>
#include <algorithm>
using namespace std;

// 接受 span 的函数：不关心底层容器类型
void printSpan(span<const int> s, const string& label) {
    cout << label << " (size=" << s.size() << "): ";
    for (int x : s) cout << x << " ";
    cout << endl;
}

// 计算 span 中元素的和
int sumSpan(span<const int> s) {
    int total = 0;
    for (int x : s) total += x;
    return total;
}

// 查找 span 中的最大值
optional<int> maxSpan(span<const int> s) {
    if (s.empty()) return nullopt;
    return *max_element(s.begin(), s.end());
}

int main() {
    // 1. 从不同容器构造 span
    int cArray[] = {1, 2, 3, 4, 5};
    vector<int> vec = {10, 20, 30, 40, 50};
    array<int, 4> arr = {100, 200, 300, 400};

    printSpan(cArray, "C 数组");
    printSpan(vec, "vector");
    printSpan(arr, "array");

    // 2. 子视图 (subspan)
    span<int> fullSpan(vec);
    // 从索引 1 开始，取 3 个元素
    auto sub = fullSpan.subspan(1, 3);
    printSpan(sub, "subspan(1,3)");

    // 前 3 个元素
    auto first3 = fullSpan.first(3);
    printSpan(first3, "first(3)");

    // 后 2 个元素
    auto last2 = fullSpan.last(2);
    printSpan(last2, "last(2)");

    // 3. 检查 span 的大小
    cout << "\nspan 是否为空: " << boolalpha
         << span<const int>{}.empty() << endl;

    cout << "vector span 大小: " << fullSpan.size() << endl;
    cout << "vector span 大小(字节): " << fullSpan.size_bytes() << endl;

    // 4. 统一接口：同一函数处理不同容器
    cout << "\nsum(cArray) = " << sumSpan(cArray) << endl;
    cout << "sum(vec) = " << sumSpan(vec) << endl;
    cout << "max(vec) = " << maxSpan(vec).value_or(-1) << endl;

    return 0;
}

语法规则：span 构造

#include <span>
span<int> s(vec);          // 从 vector
span<int> s(arr, 5);       // 从指针+长度
span<int> s(cArray);       // 从 C 数组
span<const int> s = vec;   // 只读视图

举一反三：span 不拥有数据，只是指针+长度的轻量视图。拷贝 span 不拷贝数据。动态 extent 运行时确定大小，静态 extent 编译期确定（span）。

语法规则：subspan / first / last

span<int> s(vec);
s.subspan(2, 3);    // 索引 2 开始，取 3 个
s.first(3);          // 前 3 个
s.last(2);           // 后 2 个
// 均为 O(1)，不分配内存

举一反三：所有子视图操作都是 O(1)（只调整指针和长度）。subspan 参数 offset+count，count 可选（默认到末尾）。first 和 last 是 subspan 的语法糖。

语法规则：span 作为函数参数

void process(span<const int> data);
// 可接受：C数组、vector、array、其他 span
// 不需要模板、不需要重载
// 比 (int* ptr, size_t len) 更安全

举一反三：用 span 替代 (T* ptr, size_t len) 参数对。span 带边界信息，减少越界风险。const span 表示只读（元素不可修改），非 const span 可修改元素。
