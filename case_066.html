案例 66：HTTP 静态文件服务器

基于原生 socket API 实现一个能够处理 GET 请求的 HTTP/1.1 服务器，返回本地文件并附带正确的 MIME 类型。

核心知识点：socket 编程、HTTP 协议、MIME 类型、多线程并发处理

完整代码（含逐行注释）

// 案例 66：HTTP 静态文件服务器（基于 TCP socket）
// 实现一个极简的 HTTP/1.1 静态文件服务器，支持 GET 请求
// 访问 http://localhost:8080/index.html 会返回对应文件内容

#include <iostream>      // std::cout, std::cerr
#include <string>        // std::string
#include <fstream>       // 文件读取：std::ifstream
#include <sstream>       // 字符串流：std::ostringstream
#include <thread>        // 多线程：std::thread
#include <vector>        // std::vector
#include <cstring>       // memset, strlen

/* ── 语法规则：POSIX socket API 入门 ──
   Windows 使用 Winsock2，需链接 ws2_32.lib；Linux 直接用 <sys/socket.h>

   socket(domain, type, protocol)
     domain: AF_INET(IPv4) / AF_INET6(IPv6)
     type: SOCK_STREAM(TCP) / SOCK_DGRAM(UDP)
     protocol: 0(自动) / IPPROTO_TCP
     返回：失败 -1，成功返回文件描述符(fd)

   bind(fd, addr, addrlen)  → 将 socket 绑定到地址端口
   listen(fd, backlog)      → 开始监听，backlog 为等待队列长度
   accept(fd, addr, addrlen)→ 阻塞等待客户端连接，返回新 fd

   recv(fd, buf, len, flags)→ 接收数据
   send(fd, buf, len, flags)→ 发送数据
   close(fd)                → 关闭连接

   示例：创建一个简单 TCP echo 服务器
     int fd = socket(AF_INET, SOCK_STREAM, 0);
     sockaddr_in addr{AF_INET, htons(8080), INADDR_ANY};
     bind(fd, (sockaddr*)&addr, sizeof(addr));
     listen(fd, 5);
     int client = accept(fd, nullptr, nullptr);
     char buf[1024]; recv(client, buf, 1024, 0);
     send(client, buf, strlen(buf), 0);
     close(client); close(fd);

   举一反三：socket 是所有网络编程的基石——Web服务器、数据库驱动、
   即时通讯、游戏服务器全部建立在这组 API 之上。
*/

#ifdef _WIN32
    #include <winsock2.h>   // Windows socket API
    #include <ws2tcpip.h>   // inet_ntop 等
    #pragma comment(lib, "ws2_32.lib")  // 链接 Winsock 库
    using socklen_t = int;
    #define CLOSE_SOCK closesocket  // Windows 用 closesocket 关闭
#else
    #include <sys/socket.h>  // socket/bind/listen/accept
    #include <netinet/in.h>  // sockaddr_in
    #include <arpa/inet.h>   // inet_ntop
    #include <unistd.h>      // close
    #define CLOSE_SOCK close
#endif

// 读取文件内容，返回字符串
std::string readFile(const std::string& path) {
    // 打开文件，std::ios::binary 防止文本模式自动转换换行符
    std::ifstream file(path, std::ios::binary);
    if (!file) return "";  // 文件不存在返回空串

    // ostringstream 把文件内容全部读入内存
    std::ostringstream oss;
    oss << file.rdbuf();
    return oss.str();
}

// 根据文件扩展名返回 MIME 类型
std::string getMimeType(const std::string& path) {
    // 找到最后一个 '.' 的位置
    size_t dot = path.find_last_of('.');
    if (dot == std::string::npos) return "application/octet-stream";

    std::string ext = path.substr(dot);  // 提取扩展名
    if (ext == ".html") return "text/html; charset=utf-8";
    if (ext == ".css")  return "text/css";
    if (ext == ".js")   return "application/javascript";
    if (ext == ".json") return "application/json";
    if (ext == ".png")  return "image/png";
    if (ext == ".jpg" || ext == ".jpeg") return "image/jpeg";
    if (ext == ".svg")  return "image/svg+xml";
    return "text/plain; charset=utf-8";  // 默认纯文本
}

// 处理单个 HTTP 请求
void handleClient(int clientFd) {
    char buffer[4096] = {0};                       // 接收缓冲区，4KB 足够解析请求头
    int bytesRead = recv(clientFd, buffer, 4096, 0); // 从客户端读取数据

    if (bytesRead <= 0) {                          // 连接断开或读取失败
        CLOSE_SOCK(clientFd);
        return;
    }

    // 解析 HTTP 请求行：GET /path HTTP/1.1
    std::string request(buffer, bytesRead);
    std::istringstream iss(request);
    std::string method, path, version;
    iss >> method >> path >> version;              // 提取三个字段

    std::cout << "[请求] " << method << " " << path << std::endl;

    // 只处理 GET 方法
    if (method != "GET") {
        std::string response = "HTTP/1.1 405 Method Not Allowed\r\n\r\n";
        send(clientFd, response.c_str(), response.size(), 0);
        CLOSE_SOCK(clientFd);
        return;
    }

    // 默认首页 index.html
    if (path == "/") path = "/index.html";

    // 路径安全：防止目录穿越攻击（如 GET /../../../etc/passwd）
    if (path.find("..") != std::string::npos) {
        std::string response = "HTTP/1.1 403 Forbidden\r\n\r\n";
        send(clientFd, response.c_str(), response.size(), 0);
        CLOSE_SOCK(clientFd);
        return;
    }

    // 拼接文件路径：当前目录 + 请求路径
    std::string filePath = "." + path;
    std::string content = readFile(filePath);

    std::string response;
    if (content.empty()) {
        // 404 Not Found
        response = "HTTP/1.1 404 Not Found\r\nContent-Type: text/plain\r\n\r\n404 - File Not Found";
    } else {
        // 200 OK + 响应头 + 空行 + 内容
        std::string mime = getMimeType(path);
        response = "HTTP/1.1 200 OK\r\n"
                   "Content-Type: " + mime + "\r\n"
                   "Content-Length: " + std::to_string(content.size()) + "\r\n"
                   "Connection: close\r\n"           // 短连接：响应后关闭
                   "\r\n"                            // 请求头结束空行
                   + content;
    }

    send(clientFd, response.c_str(), response.size(), 0); // 发送响应
    CLOSE_SOCK(clientFd);                                   // 关闭连接
    std::cout << "[完成] " << path << std::endl;
}

int main() {
#ifdef _WIN32
    // Windows 需要先初始化 Winsock
    WSADATA wsa;
    WSAStartup(MAKEWORD(2,2), &wsa);
#endif

    // 1. 创建 TCP socket
    int serverFd = socket(AF_INET, SOCK_STREAM, 0);
    if (serverFd < 0) {
        std::cerr << "socket 创建失败" << std::endl;
        return 1;
    }

    // 2. 设置地址重用（避免重启时"Address already in use"）
    int opt = 1;
    setsockopt(serverFd, SOL_SOCKET, SO_REUSEADDR,
#ifdef _WIN32
               (const char*)&opt, sizeof(opt));
#else
               &opt, sizeof(opt));
#endif

    // 3. 绑定到 8080 端口
    sockaddr_in addr{};
    addr.sin_family = AF_INET;              // IPv4
    addr.sin_port = htons(8080);            // 端口 8080，htons 将主机字节序转为网络字节序
    addr.sin_addr.s_addr = INADDR_ANY;      // 监听所有网络接口

    if (bind(serverFd, (sockaddr*)&addr, sizeof(addr)) < 0) {
        std::cerr << "bind 失败" << std::endl;
        return 1;
    }

    // 4. 开始监听，backlog=10
    if (listen(serverFd, 10) < 0) {
        std::cerr << "listen 失败" << std::endl;
        return 1;
    }

    std::cout << "HTTP 服务器运行在 http://localhost:8080/" << std::endl;

    // 5. 主循环：接受连接，每个连接分配一个线程
    while (true) {
        sockaddr_in clientAddr{};
        socklen_t addrLen = sizeof(clientAddr);
        int clientFd = accept(serverFd, (sockaddr*)&clientAddr, &addrLen);

        if (clientFd < 0) continue;

        // detach 让线程独立运行，主线程继续 accept
        std::thread(handleClient, clientFd).detach();
    }

    CLOSE_SOCK(serverFd);
    return 0;
}

语法规则：HTTP 协议基础

请求格式：  METHOD /path HTTP/1.1\r\nHeader: value\r\n\r\nBODY
响应格式：  HTTP/1.1 200 OK\r\nContent-Type: text/html\r\n\r\nBODY

💡 举一反三：HTTP 是文本协议，\r\n 换行，请求头与 Body 之间空一行。所有 Web 框架（Express/Spring/Flask）底层都是这样构建响应的。

语法规则：htons() / htonl() 网络字节序

uint16_t port = htons(8080);  // 主机→网络(大端)
uint16_t local = ntohs(port);   // 网络→主机
// x86 是小端，网络是大端，必须转换！

💡 举一反三：所有多字节字段（端口、IP、数据长度）在 socket 编程中都必须考虑字节序。htons=Host TO Network Short，htonl=Host TO Network Long。

语法规则：setsockopt 设置 socket 选项

int opt = 1;
setsockopt(fd, SOL_SOCKET, SO_REUSEADDR, &opt, sizeof(opt));
// SO_REUSEADDR 允许重用 TIME_WAIT 状态的地址

💡 举一反三：开发调试时必须加 SO_REUSEADDR，否则按 Ctrl+C 重启服务会报 'Address already in use'。其他常用选项：SO_KEEPALIVE（心跳）、SO_RCVBUF（接收缓冲区大小）。
