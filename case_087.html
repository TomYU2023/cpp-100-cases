案例 87：std::is_sorted/sorted_until 顺序检查

使用 is_sorted 检查序列是否有序，找到第一个无序位置。

核心知识点：is_sorted、is_sorted_until、有序性检查

完整代码（含逐行注释）

// 案例 87：std::is_sorted / sorted_until 顺序检查
// 检测序列是否有序
#include <iostream>
#include <algorithm>
#include <vector>
#include <string>
using namespace std;

int main() {
    // 1. is_sorted：检查是否有序
    vector<int> v1 = {1, 2, 3, 4, 5};
    vector<int> v2 = {1, 3, 2, 4, 5};
    vector<int> v3 = {5, 4, 3, 2, 1};

    cout << "检查有序性:" << endl;
    cout << "v1 [1-5] 升序: " << boolalpha
         << is_sorted(v1.begin(), v1.end()) << endl;
    cout << "v2 [1,3,2,4,5] 升序: "
         << is_sorted(v2.begin(), v2.end()) << endl;
    cout << "v3 [5 → 1] 降序: "
         << is_sorted(v3.begin(), v3.end(), greater<int>()) << endl;

    // 2. is_sorted_until：找到第一个无序的位置
    auto it = is_sorted_until(v2.begin(), v2.end());
    cout << "\nv2 的第一个无序位置: "
         << distance(v2.begin(), it) << " (值=" << *it << ")" << endl;

    // 3. 实际应用：数据验证
    // 检查成绩是否按从高到低排列
    vector<int> scores = {95, 92, 88, 85, 80, 82, 75};
    cout << "\n成绩: ";
    for (int s : scores) cout << s << " ";
    cout << endl;

    if (!is_sorted(scores.begin(), scores.end(), greater<int>())) {
        auto unorderedPos = is_sorted_until(
            scores.begin(), scores.end(), greater<int>());
        cout << "排序错误在位置 "
             << distance(scores.begin(), unorderedPos)
             << " (值=" << *unorderedPos << ")" << endl;
    }

    // 4. 检查部分有序
    vector<int> data = {10, 20, 30, 5, 15, 25, 35};
    // 检查前 3 个是否有序
    cout << "\n前 3 个有序? "
         << is_sorted(data.begin(), data.begin() + 3) << endl;
    // 检查后 4 个是否有序
    cout << "后 4 个有序? "
         << is_sorted(data.begin() + 3, data.end()) << endl;

    return 0;
}
