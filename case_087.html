案例 87：终端俄罗斯方块

在控制台窗口内实现俄罗斯方块游戏，包含 7 种方块、旋转碰撞检测、消行计分、非阻塞输入和定时刷新。

核心知识点：_kbhit 非阻塞输入、SetConsoleCursorPosition 光标控制、std::array、游戏循环架构

完整代码（含逐行注释）

// 案例 87：终端俄罗斯方块
// 在控制台窗口内实现经典俄罗斯方块，包含 7 种方块形状、旋转、消行、计分。
// 使用 Windows API 实现非阻塞键盘输入和定时刷新。

#include <iostream>
#include <vector>
#include <array>
#include <conio.h>     // _kbhit(), _getch() — Windows 控制台输入
#include <windows.h>   // Sleep(), 光标控制
#include <ctime>

// ─── 常量 ────────────────────────────────────────
const int W = 10, H = 20;  // 面板宽高

// ─── 7 种方块定义（4x4 矩阵）─────────────────────
// 每种方块有 4 种旋转状态，这里仅展示初始状态
const std::array<std::array<std::string, 4>, 7> PIECES = {{
    // I 型
    {"....","XXXX","....","...."},
    // O 型
    {"....",".XX.",".XX.","...."},
    // T 型
    {"....",".X..","XXX.","...."},
    // S 型
    {"....","..XX",".XX.","...."},
    // Z 型
    {"....","XX..",".XX.","...."},
    // L 型
    {"....","..X.","XXX.","...."},
    // J 型
    {"....","X...","XXX.","...."},
}};

// ─── 全局状态 ────────────────────────────────────
int board[H][W] = {0};       // 固定面板 0=空 1~7=颜色
int curPiece, curX=3, curY=0, curRot=0;
int score = 0;

// ─── 辅助函数 ────────────────────────────────────
void gotoXY(int x, int y) {
    COORD c = {(SHORT)x, (SHORT)y};
    SetConsoleCursorPosition(GetStdHandle(STD_OUTPUT_HANDLE), c);
}

void hideCursor() {
    CONSOLE_CURSOR_INFO info = {100, FALSE};
    SetConsoleCursorInfo(GetStdHandle(STD_OUTPUT_HANDLE), &info);
}

// ─── 碰撞检测 ────────────────────────────────────
bool collide(int px, int py, int rot) {
    for (int y = 0; y < 4; y++)
        for (int x = 0; x < 4; x++)
            if (PIECES[curPiece][rot][y] == 'X') {
                int bx = px + x, by = py + y;
                if (bx < 0 || bx >= W || by >= H) return true;
                if (by >= 0 && board[by][bx]) return true;
            }
    return false;
}

// ─── 固定方块到面板 ──────────────────────────────
void lock() {
    for (int y = 0; y < 4; y++)
        for (int x = 0; x < 4; x++)
            if (PIECES[curPiece][curRot][y] == 'X') {
                int bx = curX + x, by = curY + y;
                if (by >= 0 && by < H)
                    board[by][bx] = curPiece + 1;
            }
}

// ─── 消行 ────────────────────────────────────────
void clearLines() {
    for (int y = H - 1; y >= 0; y--) {
        bool full = true;
        for (int x = 0; x < W; x++)
            if (!board[y][x]) { full = false; break; }
        if (full) {
            // 下移上方所有行
            for (int yy = y; yy > 0; yy--)
                for (int x = 0; x < W; x++)
                    board[yy][x] = board[yy-1][x];
            for (int x = 0; x < W; x++) board[0][x] = 0;
            y++;  // 重新检查当前行
            score += 100;
        }
    }
}

// ─── 渲染 ────────────────────────────────────────
void draw() {
    gotoXY(0, 0);
    // 绘制面板
    for (int y = 0; y < H; y++) {
        std::cout << "|";
        for (int x = 0; x < W; x++) {
            bool isPiece = false;
            for (int py = 0; py < 4; py++)
                for (int px = 0; px < 4; px++)
                    if (PIECES[curPiece][curRot][py] == 'X'
                        && curX+px == x && curY+py == y)
                        isPiece = true;
            if (isPiece) std::cout << "#";
            else if (board[y][x]) {
                std::cout << char('0' + board[y][x]);  // 数字表示颜色
            }
            else std::cout << ".";
        }
        std::cout << "|" << std::endl;
    }
    std::cout << "+----------+" << std::endl;
    std::cout << "得分: " << score << "   操作: ←→ 移动  ↑ 旋转  ↓ 加速" << std::endl;
}

// ─── 旋转方块（顺时针）───────────────────────────
void rotate() {
    int newRot = (curRot + 1) % 4;
    if (!collide(curX, curY, newRot)) curRot = newRot;
}

// ─── 主循环 ──────────────────────────────────────
int main() {
    hideCursor();
    srand((unsigned)time(0));
    curPiece = rand() % 7;

    // 游戏主循环
    while (true) {
        draw();

        // 处理键盘输入（非阻塞）
        if (_kbhit()) {
            int key = _getch();
            if (key == 224) {  // 方向键前缀
                key = _getch();
                if (key == 75 && !collide(curX-1, curY, curRot)) curX--;  // 左
                if (key == 77 && !collide(curX+1, curY, curRot)) curX++;  // 右
                if (key == 80 && !collide(curX, curY+1, curRot)) curY++;  // 下
                if (key == 72) rotate();  // 上
            }
            if (key == 27) break;  // ESC 退出
        }

        // 自然下落
        if (!collide(curX, curY + 1, curRot)) {
            curY++;
        } else {
            lock();
            clearLines();
            // 生成新方块
            curPiece = rand() % 7;
            curX = 3; curY = 0; curRot = 0;
            if (collide(curX, curY, curRot)) {
                std::cout << "游戏结束！最终得分: " << score << std::endl;
                break;
            }
        }
        Sleep(200);  // 控制下落速度（毫秒）
    }
    return 0;
}

语法规则：_kbhit() / _getch() 非阻塞键盘检测

#include <conio.h>
if (_kbhit()) {        // 有按键被按下？不等待返回
    int key = _getch();  // 读取按键（阻塞但仅在 _kbhit 返回 true 时调）
    if (key == 224) key = _getch();  // 方向键=224+方向码
}

💡 举一反三：_kbhit 是 Windows 特有的非阻塞键盘检测函数，返回 0 表示没有按键。游戏循环的核心是：每帧检查按键 → 更新状态 → 渲染 → Sleep。Linux 下用 ncurses 库替代，原理相同。

语法规则：SetConsoleCursorPosition 控制台光标定位

#include <windows.h>
void gotoXY(int x, int y) {
    COORD c = {(SHORT)x, (SHORT)y};
    SetConsoleCursorPosition(
        GetStdHandle(STD_OUTPUT_HANDLE), c);
}

💡 举一反三：通过 Windows Console API 直接移动光标，实现原地刷新（而非滚屏清屏）。这比 system("cls") 效率高得多，不会有闪烁感。在 Linux 上用 ANSI 转义码 \033[y;xH 实现同样效果。

语法规则：std::array 定长数组

#include <array>
std::array<int, 100> arr;  // 编译期确定大小的数组
arr.fill(0);                // 全部填 0
int s = arr.size();         // 返回 100（编译期常量）

💡 举一反三：array 是 C 数组的现代化包装，有 .size()、.at()（边界检查）、迭代器支持，且不额外占用内存。相比 vector，array 无堆分配，适合小块固定数据。
