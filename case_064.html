案例 64：对象池模式

阶段：工程能力锻造难度：⭐⭐⭐练什么：资源复用、工厂模式
对象池维护一组预创建的对象，借出/归还，避免频繁创建销毁。数据库连接池、线程池都是这个模式。

完整代码与逐行注释

case_064.cpp
1#include <iostream>


2#include <memory>


💬 第2行：unique_ptr / shared_ptr


3#include <vector>


4#include <mutex>


5#include <functional>


6 


7template<typename T>


💬 第7行：泛型对象池


8class ObjectPool {


9    std::vector<std::unique_ptr<T>> pool;


💬 第9行：存储空闲对象的池


10    std::mutex mtx;


💬 第10行：线程安全


11 


12public:


13    // 预创建对象


14    void preCreate(size_t count) {


15        std::lock_guard<std::mutex> lock(mtx);


16        for (size_t i = 0; i < count; ++i) {


17            pool.push_back(std::make_unique<T>());


💬 第17行：用 make_unique 创建默认对象


18        }


19    }


20 


21    // 获取一个对象（unique_ptr + 自定义删除器自动归还）


22    std::unique_ptr<T, std::function<void(T*)>> acquire() {


💬 第22行：返回类型含删除器签名


23        std::lock_guard<std::mutex> lock(mtx);


24        if (pool.empty()) {


💬 第24行：池空则动态创建


25            auto* raw = new T();


26            return {raw, [this](T* p) { release(p); }};


💬 第26行：即使动态创建的，归还时也走 release


27        }


28        auto obj = std::move(pool.back());


💬 第28行：从池末尾取（O(1)）


29        pool.pop_back();


30        T* raw = obj.release();


💬 第30行：释放 unique_ptr 的所有权，获取裸指针


31        return {raw, [this](T* p) { release(p); }};


💬 第31行：用自定义删除器包装：归还时调 release


32    }


33 


34    // 归还对象（通常由自定义删除器调用）


35    void release(T* obj) {


36        std::lock_guard<std::mutex> lock(mtx);


37        pool.push_back(std::unique_ptr<T>(obj));


💬 第37行：归还到池中


38    }


39 


40    size_t available() const { return pool.size(); }


41};


42 


43// 测试用的资源类


44class Connection {


45    int id;


46    static int nextId;


47public:


48    Connection() : id(nextId++) {


49        std::cout << "Connection " << id << " created\n";


50    }


51    ~Connection() { std::cout << "Connection " << id << " destroyed\n"; }


52    void query(const std::string& sql) {


53        std::cout << "[" << id << "] Executing: " << sql << std::endl;


54    }


55};


56int Connection::nextId = 1;


💬 第56行：静态成员类外定义


57 


58int main() {


59    ObjectPool<Connection> pool;


60    pool.preCreate(3);


💬 第60行：预创建 3 个连接


61    std::cout << "Pool available: " << pool.available() << std::endl;


62 


63    {


64        auto conn = pool.acquire();


💬 第64行：借一个连接


65        conn->query("SELECT * FROM users");


66        std::cout << "Pool available: " << pool.available() << std::endl;


💬 第66行：变为 2


67    }  // conn 离开作用域，自定义删除器自动归还！


68 


69    std::cout << "After release, available: " << pool.available() << std::endl;


💬 第69行：恢复为 3


70    return 0;


71}
📖 语法规则：std::unique_ptr 管理所有权
std::unique_ptr 独占所有权，离开作用域自动释放

示例：

auto p = std::make_unique(42);
p->bar();           // 像普通指针使用
auto p2 = std::move(p);  // 转移所有权，p 变为 nullptr

💡 举一反三：

  • unique_ptr 不支持拷贝，只支持移动，保证唯一所有权
  • 搭配自定义删除器：unique_ptr 可实现归还到池的逻辑
  • make_unique 是 C++14 引入的，比 new 更安全
📖 语法规则：自定义删除器
std::unique_ptr> ptr(obj, deleter);

示例：

auto deleter = [&pool](Widget* w) { pool.release(w); };
std::unique_ptr ptr(w, deleter);

💡 举一反三：

  • 删除器类型是 unique_ptr 签名的一部分
  • lambda 删除器需要 decltype 推导类型
  • 也可以用 std::function，但稍重
