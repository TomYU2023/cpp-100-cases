案例 64：std::optional 可选值

使用 optional 替代空指针和特殊值，优雅处理「可能不存在」的值。

核心知识点：optional、value_or、has_value、空值语义

完整代码（含逐行注释）

// 案例 64：std::optional 可选值
// 用 optional 替代空指针和哨兵值
#include <iostream>
#include <optional>
#include <string>
#include <map>
#include <vector>
using namespace std;

// 查找函数：可能找不到结果
optional<int> findIndex(const vector<int>& v, int target) {
    for (size_t i = 0; i < v.size(); i++) {
        if (v[i] == target) {
            return static_cast<int>(i); // 找到了
        }
    }
    return nullopt; // 没找到
}

// 解析整数：可能失败
optional<int> parseInteger(const string& s) {
    try {
        return stoi(s);
    } catch (...) {
        return nullopt;
    }
}

// 安全除法：除数为 0 返回空
optional<double> safeDivide(double a, double b) {
    if (b == 0) return nullopt;
    return a / b;
}

int main() {
    vector<int> data = {10, 20, 30, 40, 50};

    // 1. 查找元素
    auto idx = findIndex(data, 30);
    if (idx.has_value()) {
        cout << "找到 30，索引: " << idx.value() << endl;
    }

    idx = findIndex(data, 99);
    // value_or 提供默认值
    cout << "查找 99 的索引: " << idx.value_or(-1) << endl;

    // 2. 解析整数
    auto num = parseInteger("12345");
    cout << "解析 \"12345\": " << num.value_or(0) << endl;

    auto fail = parseInteger("abc");
    cout << "解析 \"abc\": " << (fail ? "成功「 : 」失败") << endl;

    // 3. 安全除法
    auto result = safeDivide(10.0, 3.0);
    cout << "10/3 = " << *result << endl; // 直接解引用

    auto divByZero = safeDivide(5.0, 0);
    if (!divByZero) {
        cout << "除数为零，无法计算" << endl;
    }

    return 0;
}

语法规则：optional 构造与判断

#include <optional>
optional<int> opt;
opt = 42;               // 有值
opt = nullopt;           // 无值
if (opt.has_value()) {}  // 判断
if (opt) {}              // 隐式转 bool

举一反三：optional 是值语义的「可能不存在」类型。比指针更安全（不涉及内存管理），比哨兵值（如 -1）更类型安全。nullopt 是空状态的字面量。

语法规则：value() / value_or() / *

opt.value();            // 获取值（空时抛 bad_optional_access）
opt.value_or(0);        // 获取值或默认值
*opt;                   // 解引用（空时 UB，需先检查）

举一反三：value() 带运行时检查但抛异常。value_or() 是最安全的方式（总有返回值）。* 解引用最高效但需要外部保证非空。推荐：先检查 has_value() 再用 *。

语法规则：optional 作为返回值

optional<int> find(const vector<int>& v, int target) {
    // 遍历查找...
    if (found) return index;
    return nullopt; // 明确表达「没找到」
}

举一反三：optional 返回值代替了「哨兵值 + 注释」模式（如返回 -1 表示没找到）。调用的代码更清晰：if (auto result = search()) ... 而不是检查特殊值。
