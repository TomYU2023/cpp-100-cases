案例 47：CSV解析器

实现一个 CSV 格式解析器，正确处理引号包裹字段和转义字符。

核心知识点：CSV 解析、字符串状态机、引号处理

完整代码（含逐行注释）

// 案例 47：CSV 解析器
// 正确处理引号包裹字段和逗号转义
#include <iostream>
#include <fstream>
#include <string>
#include <vector>
#include <sstream>
using namespace std;

// 解析一行 CSV
vector<string> parseCSVLine(const string& line) {
    vector<string> fields;
    string field;
    bool inQuotes = false; // 是否在引号内

    for (size_t i = 0; i < line.size(); i++) {
        char c = line[i];

        if (inQuotes) {
            if (c == '"') {
                // 检查是否是双引号转义 "" 
                if (i + 1 < line.size() && line[i + 1] == '"') {
                    field += '"'; // 转义为一个引号
                    i++;          // 跳过下一个引号
                } else {
                    inQuotes = false; // 引号结束
                }
            } else {
                field += c;
            }
        } else {
            if (c == '"') {
                inQuotes = true; // 开始引号
            } else if (c == ',') {
                fields.push_back(field); // 字段结束
                field.clear();
            } else {
                field += c;
            }
        }
    }
    fields.push_back(field); // 最后一个字段
    return fields;
}

int main() {
    // 测试 CSV 数据
    vector<string> csvLines = {
        "Name,Age,City",
        "Alice,25,\"New York\"",
        "\"Bob, Jr.\",30,\"Los Angeles\"",
        "\"Zhang \"\"San\"\"\",28,Beijing",
        "\"Hello, World\",42,\"A,B,C\""
    };

    cout << "=== CSV 解析结果 ===" << endl;

    for (const auto& line : csvLines) {
        auto fields = parseCSVLine(line);
        for (size_t i = 0; i < fields.size(); i++) {
            cout << "[" << fields[i] << "]";
            if (i < fields.size() - 1) cout << " | ";
        }
        cout << endl;
    }

    return 0;
}

语法规则：CSV 状态机解析

bool inQuotes = false;
for (char c : line) {
    if (c == '"') inQuotes = !inQuotes;
    else if (c == ',' && !inQuotes) { /* 新字段 */ }
    else field += c;
}

举一反三：状态机是解析结构化文本的通用方法。状态可以是：引号内/外、转义中、注释中等。状态机清晰表达「在什么状态下如何处理输入字符」。

语法规则：CSV 双引号转义

// CSV 标准：引号内的引号用两个引号表示 "" 
if (c == '"' && next == '"') {
    field += '"';
    i++;
}

举一反三：这是 RFC 4180 CSV 标准的关键规则。其他常见转义： → 、 → 。不同格式有不同的转义规则，需查阅对应规范。

语法规则：push_back + clear 模式

fields.push_back(field);
field.clear(); // 清空复用
// field 被 move 到 vector 中，clear 后重新使用

举一反三：复用 string 比反复创建临时 string 更高效。clear() 不释放内存，只重置 size。C++11 后 push_back(临时变量) 自动使用移动语义。
