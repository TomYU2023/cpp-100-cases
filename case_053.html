案例 53：pImpl 编译防火墙

使用 pImpl 惯用法隐藏实现细节，减少编译依赖，加速编译。

核心知识点：pImpl、前向声明、unique_ptr、编译隔离

完整代码（含逐行注释）

// 案例 53：pImpl 编译防火墙
// 隐藏实现细节，减少头文件依赖
#include <iostream>
#include <memory>
#include <string>
#include <vector>
using namespace std;

// === 头文件（public header）===
class Widget {
public:
    Widget();                           // 构造
    ~Widget();                          // 析构（需在 .cpp 中定义）
    Widget(Widget&&) noexcept;          // 移动
    Widget& operator=(Widget&&) noexcept;

    void setName(const string& name);
    string getName() const;
    void addData(int value);
    void print() const;

private:
    // 前向声明实现类（在 .cpp 中定义）
    struct Impl;
    unique_ptr<Impl> pImpl;             // 指向实现的指针
};

// === 实现文件（.cpp）===
struct Widget::Impl {
    string name;
    vector<int> data;

    void print() const {
        cout << "Widget: " << name << endl;
        cout << "Data: ";
        for (int v : data) cout << v << " ";
        cout << endl;
    }
};

// Widget 的成员函数都委托给 pImpl
Widget::Widget() : pImpl(make_unique<Impl>()) {
    cout << "Widget 构造" << endl;
}

// 析构函数必须在 Impl 完整定义之后
Widget::~Widget() = default;

Widget::Widget(Widget&&) noexcept = default;
Widget& Widget::operator=(Widget&&) noexcept = default;

void Widget::setName(const string& name) {
    pImpl->name = name;
}

string Widget::getName() const {
    return pImpl->name;
}

void Widget::addData(int value) {
    pImpl->data.push_back(value);
}

void Widget::print() const {
    pImpl->print();
}

int main() {
    Widget w;
    w.setName("pImpl Widget");
    w.addData(10);
    w.addData(20);
    w.addData(30);
    w.print();

    // 移动
    Widget w2 = move(w);
    w2.print();

    return 0;
}
