案例 70：策略模式

实现可插拔的排序策略系统——冒泡排序、快速排序、std::sort 可以随时切换，展示策略模式消除 if-else 分支的能力。

核心知识点：策略模式、面向接口编程、std::make_unique、算法切换、开闭原则

完整代码（含逐行注释）

// 案例 70：策略模式（Strategy Pattern）
// 策略模式：定义一系列算法，把它们封装起来，并使它们可以互相替换。
// 策略模式让算法独立于使用它的客户端而变化。
// 典型场景：排序算法可切换、支付方式可选、压缩算法可选。

#include <iostream>
#include <vector>
#include <memory>
#include <algorithm>

/* ── 语法规则：策略模式结构 ──
   Strategy（策略接口）：定义算法家族的公共接口
   ConcreteStrategy（具体策略）：实现具体的算法
   Context（上下文）：持有策略引用，调用策略执行算法

   示例（支付方式）：
     PaymentStrategy 接口         ← 定义 pay(amount) 方法
     ├── CreditCardPayment        ← 信用卡支付
     ├── PayPalPayment            ← PayPal 支付
     └── WeChatPayment            ← 微信支付
     ShoppingCart context          ← 购物车，setStrategy() 切换支付方式

   举一反三：策略模式是 if-else 地狱的解药。任何需要"根据不同条件选不同算法"
   的场景都适用。Java 的 Comparator、C++ 的 std::sort 第三个参数都是策略模式。
*/

// 步骤 1：定义排序策略接口
class SortStrategy {
public:
    virtual ~SortStrategy() = default;
    virtual void sort(std::vector<int>& data) = 0;
    virtual std::string name() const = 0;  // 返回策略名称
};

// 步骤 2：具体策略 1 — 冒泡排序 (O(n²)，稳定)
class BubbleSort : public SortStrategy {
public:
    void sort(std::vector<int>& data) override {
        size_t n = data.size();
        for (size_t i = 0; i < n - 1; ++i) {
            bool swapped = false;  // 优化：如果一轮没有交换说明已有序
            for (size_t j = 0; j < n - 1 - i; ++j) {
                if (data[j] > data[j + 1]) {
                    std::swap(data[j], data[j + 1]);  // 相邻元素比较并交换
                    swapped = true;
                }
            }
            if (!swapped) break;  // 提前结束
        }
    }
    std::string name() const override { return "冒泡排序 (O(n²))"; }
};

// 步骤 2：具体策略 2 — 快速排序 (O(n log n)，最快)
class QuickSort : public SortStrategy {
    void quickSort(std::vector<int>& data, int low, int high) {
        if (low >= high) return;               // 递归终止条件

        int pivot = data[high];                // 选择最后一个元素为基准
        int i = low - 1;                       // i 指向小于 pivot 的区域的末尾

        for (int j = low; j < high; ++j) {     // 遍历 [low, high-1]
            if (data[j] < pivot) {
                std::swap(data[++i], data[j]); // 小于 pivot 的元素移到左边
            }
        }
        std::swap(data[i + 1], data[high]);    // pivot 归位
        int pi = i + 1;

        quickSort(data, low, pi - 1);          // 递归排序左半部分
        quickSort(data, pi + 1, high);         // 递归排序右半部分
    }

public:
    void sort(std::vector<int>& data) override {
        if (!data.empty()) quickSort(data, 0, data.size() - 1);
    }
    std::string name() const override { return "快速排序 (O(n log n))"; }
};

// 步骤 2：具体策略 3 — 标准库排序 (std::sort，最省事)
class StdSort : public SortStrategy {
public:
    void sort(std::vector<int>& data) override {
        std::sort(data.begin(), data.end());   // 直接调用标准库
    }
    std::string name() const override { return "std::sort (内省排序)"; }
};

// 步骤 3：上下文 — 数据排序器，持有策略
class DataSorter {
    std::unique_ptr<SortStrategy> strategy;  // 当前策略（独占所有权）

public:
    // 设置排序策略：通过 unique_ptr 转移所有权
    void setStrategy(std::unique_ptr<SortStrategy> s) {
        strategy = std::move(s);
    }

    // 执行排序
    void execute(std::vector<int>& data) {
        if (!strategy) {
            std::cerr << "错误：未设置排序策略!" << std::endl;
            return;
        }
        std::cout << "使用策略: " << strategy->name() << std::endl;
        strategy->sort(data);
    }
};

// 辅助函数：打印 vector
void print(const std::vector<int>& v) {
    for (int x : v) std::cout << x << " ";
    std::cout << std::endl;
}

int main() {
    DataSorter sorter;

    // 准备两份相同数据用于对比
    std::vector<int> data1 = {64, 34, 25, 12, 22, 11, 90};
    std::vector<int> data2 = data1;  // 拷贝一份
    std::vector<int> data3 = data1;

    std::cout << "原始数据: "; print(data1);

    // 切换策略 1：冒泡排序
    sorter.setStrategy(std::make_unique<BubbleSort>());
    sorter.execute(data1);
    std::cout << "排序结果: "; print(data1);

    // 切换策略 2：快速排序
    sorter.setStrategy(std::make_unique<QuickSort>());
    sorter.execute(data2);
    std::cout << "排序结果: "; print(data2);

    // 切换策略 3：标准库
    sorter.setStrategy(std::make_unique<StdSort>());
    sorter.execute(data3);
    std::cout << "排序结果: "; print(data3);

    return 0;
}

语法规则：std::make_unique 工厂函数

auto p = std::make_unique<MyClass>(arg1, arg2);  // C++14
// 等价于 std::unique_ptr<MyClass> p(new MyClass(arg1, arg2));
// make_unique 的好处：异常安全 + 一次分配

💡 举一反三：永远优先使用 make_unique / make_shared 而不是手动 new。make_shared 还会把对象和控制块分配在一起，性能更好。

语法规则：std::swap 交换两个值

int a = 1, b = 2;
std::swap(a, b);  // a=2, b=1
// 原理：T temp = std::move(a); a = std::move(b); b = std::move(temp);

💡 举一反三：std::swap 利用移动语义避免不必要的拷贝。对于 vector，swap 只交换三个指针，O(1) 复杂度。
