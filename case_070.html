案例 70：bitset 位图操作

使用 bitset 进行位操作，实现集合运算、权限管理和布隆过滤器基础。

核心知识点：bitset、位运算、集合操作、权限管理

完整代码（含逐行注释）

// 案例 70：bitset 位图操作
// 位图实现权限管理和集合运算
#include <iostream>
#include <bitset>
#include <string>
using namespace std;

// 权限系统
enum Permission {
    READ    = 0,  // 读
    WRITE   = 1,  // 写
    EXECUTE = 2,  // 执行
    DELETE  = 3,  // 删除
    ADMIN   = 4   // 管理
};

int main() {
    // 1. 基本操作
    bitset<8> b1;             // 全 0
    bitset<8> b2(42);         // 00101010
    bitset<8> b3("1100");     // 00001100

    cout << "b1: " << b1 << endl;
    cout << "b2: " << b2 << " (=" << b2.to_ulong() << ")" << endl;
    cout << "b3: " << b3 << endl;

    // 2. 位操作
    cout << "\n=== 位运算 ===" << endl;
    cout << "b2 | b3: " << (b2 | b3) << endl; // 或
    cout << "b2 & b3: " << (b2 & b3) << endl; // 与
    cout << "b2 ^ b3: " << (b2 ^ b3) << endl; // 异或
    cout << "~b2: " << (~b2) << endl;         // 取反

    // 3. 单比特操作
    cout << "\n=== 单比特操作 ===" << endl;
    bitset<8> bs;
    bs.set(3);        // 设置第 3 位
    bs.set(5, true);  // 显式设置
    cout << "set(3,5): " << bs << endl;

    bs.reset(3);      // 清除第 3 位
    cout << "reset(3): " << bs << endl;

    bs.flip(2);       // 翻转第 2 位
    cout << "flip(2): " << bs << endl;

    cout << "位 5 是否设置: " << bs.test(5) << endl;
    cout << "设置位数: " << bs.count() << endl;
    cout << "是否有任何位设置: " << bs.any() << endl;

    // 4. 权限系统示例
    cout << "\n=== 权限系统 ===" << endl;
    bitset<8> userPerms; // 用户权限

    // 授予读写权限
    userPerms.set(READ);
    userPerms.set(WRITE);

    cout << "用户权限: " << userPerms << endl;
    cout << "可读: " << userPerms.test(READ) << endl;
    cout << "可写: " << userPerms.test(WRITE) << endl;
    cout << "可执行: " << userPerms.test(EXECUTE) << endl;

    // 检查是否有某组权限
    bitset<8> requiredPerms;
    requiredPerms.set(READ);
    requiredPerms.set(WRITE);
    bool hasPerms = ((userPerms & requiredPerms) == requiredPerms);
    cout << "同时拥有读写权限: " << boolalpha << hasPerms << endl;

    return 0;
}
