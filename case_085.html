案例 85：简易 Redis 客户端（RESP 协议）

实现 Redis RESP 协议的客户端，包括命令构建和响应解析，理解应用层协议的设计。

核心知识点：RESP 协议、std::optional、字符串查找/substr、static 函数、to_string

完整代码（含逐行注释）

#include <iostream>
#include <string>
#include <vector>
#include <optional>

// 简易 Redis RESP 协议客户端
// RESP (REdis Serialization Protocol) 是 Redis 的通信协议
// 格式非常简单：类型符号 + 数据 + \r\n

class SimpleRedisClient {
    // 注意：这里只演示 RESP 协议的构建和解析
    // 真实场景需要 socket 连接（参考案例 65/66）

    // ========== RESP 协议命令构建 ==========
    // 语法规则：static 成员函数 — 不需要对象实例即可调用
    // static 函数只能访问 static 成员，不能访问 this
public:
    static std::string buildCommand(const std::string& cmd) {
        // 将命令按空格分割
        std::vector<std::string> parts;
        size_t start = 0;
        while (start < cmd.size()) {
            size_t end = cmd.find(' ', start);
            if (end == std::string::npos) end = cmd.size();
            parts.push_back(cmd.substr(start, end - start));
            start = end + 1;
        }
        return buildArray(parts);
    }

    // 构建 SET key value 命令
    // 语法规则：std::to_string — 数字转字符串（C++11）
    // int x = 42; std::string s = std::to_string(x); // "42"
    // 也支持 double, long, long long 等类型
    static std::string buildSET(const std::string& key, const std::string& value) {
        return buildArray({"SET", key, value});
    }

    static std::string buildGET(const std::string& key) {
        return buildArray({"GET", key});
    }

    static std::string buildDEL(const std::string& key) {
        return buildArray({"DEL", key});
    }

private:
    // 构建 RESP 数组：*数量\r\n$长度\r\n数据\r\n...
    static std::string buildArray(const std::vector<std::string>& args) {
        std::string result;
        result += "*" + std::to_string(args.size()) + "\r\n"; // *3
        for (const auto& arg : args) {
            result += "$" + std::to_string(arg.size()) + "\r\n"; // $3
            result += arg + "\r\n";                               // SET
        }
        return result;
    }

    // ========== RESP 协议响应解析 ==========
public:
    // 解析 RESP 响应，返回可选值
    // 语法规则：std::optional<T> (C++17) — 可能不存在的值
    // std::optional<int> opt;           // 空 optional
    // opt = 42;                         // 有值
    // if (opt) { int v = *opt; }        // 检查并取值
    // int v = opt.value_or(0);          // 有值取值，无值返回默认值
    static std::optional<std::string> parseResponse(const std::string& resp) {
        if (resp.empty()) return std::nullopt;

        char type = resp[0];
        switch (type) {
            case '+': // 简单字符串 "+OK\r\n"
                return resp.substr(1, resp.size() - 3); // 去掉 \r\n

            case '-': // 错误 "-ERR unknown command\r\n"
                std::cerr << "Redis 错误: " << resp.substr(1, resp.size() - 3) << "\n";
                return std::nullopt;

            case ':': // 整数 ":42\r\n"
                return resp.substr(1, resp.size() - 3);

            case '$': { // 批量字符串 "$5\r\nhello\r\n"
                // 语法规则：std::string::find(substr) — 查找子串位置
                // 返回 size_t，找不到返回 std::string::npos
                size_t pos = resp.find("\r\n");
                if (pos == std::string::npos) return std::nullopt;

                int len = std::stoi(resp.substr(1, pos - 1));
                if (len == -1) return std::nullopt; // NULL 回复 $-1\r\n

                size_t dataStart = pos + 2;
                return resp.substr(dataStart, len);
            }

            case '*': // 数组 "*2\r\n$3\r\nfoo\r\n$3\r\nbar\r\n"
                return "数组: " + resp; // 简化处理

            default:
                return std::nullopt;
        }
    }

    // 使用示例：模拟交互
    static void demo() {
        std::cout << "=== RESP 协议演示 ===\n\n";

        // 构建命令
        auto setCmd = buildSET("username", "Alice");
        std::cout << "发送 SET username Alice:\n" << escapeResp(setCmd) << "\n";

        auto getCmd = buildGET("username");
        std::cout << "\n发送 GET username:\n" << escapeResp(getCmd) << "\n";

        // 模拟响应
        std::string setResp = "+OK\r\n";
        auto result1 = parseResponse(setResp);
        std::cout << "\nSET 响应: " << (result1 ? *result1 : "(null)") << "\n";

        std::string getResp = "$5\r\nAlice\r\n";
        auto result2 = parseResponse(getResp);
        std::cout << "GET 响应: " << (result2 ? *result2 : "(null)") << "\n";
    }

private:
    // 将 \r\n 可视化
    static std::string escapeResp(const std::string& s) {
        std::string r;
        for (char c : s) {
            if (c == '\r') r += "\\r";
            else if (c == '\n') r += "\\n\n";
            else r += c;
        }
        return r;
    }
};

int main() {
    SimpleRedisClient::demo();

    std::cout << "\n实际 Redis 客户端还需要：\n"
              << "1. TCP Socket 连接（参考案例 65）\n"
              << "2. 连接池管理（参考案例 84）\n"
              << "3. 管道（Pipeline）批量命令\n"
              << "4. 发布/订阅模式\n";
    return 0;
}

语法规则：std::optional (C++17)

#include <optional>
std::optional<int> findUser(const std::string& name) {
    if (name == "Alice") return 25;
    return std::nullopt;  // 表示"无值"
}

auto age = findUser("Bob");
std::cout << age.value_or(-1);  // -1（默认值）
if (age) std::cout << *age;     // 安全访问

💡 举一反三：optional 解决了函数返回值"可能不存在"的语义问题。比返回 -1/NULL 更安全，比抛异常更轻量。value_or() 提供默认值，has_value() 检查是否有值。

语法规则：std::string::find / rfind / find_first_of

std::string s = "hello world hello";
size_t p1 = s.find("world");      // 6（首次出现）
size_t p2 = s.rfind("hello");     // 12（最后一次出现）
size_t p3 = s.find_first_of("aeiou"); // 1（第一个元音）
size_t p4 = s.find("xyz");        // std::string::npos（未找到）

💡 举一反三：find 族函数返回 size_t 位置。未找到返回 std::string::npos（-1 的无符号形式）。务必检查 npos，否则访问 npos 位置会越界。C++23 新增 contains() 方法。

语法规则：std::string::substr(pos, count)

std::string s = "Hello World";
std::string sub = s.substr(0, 5);    // "Hello"
std::string rest = s.substr(6);      // "World"（省略 count 到末尾）
// 抛异常：s.substr(100)  // std::out_of_range

💡 举一反三：substr 返回新字符串，是拷贝操作不是视图。pos 超出范围抛异常。大量字符串操作时考虑用 string_view 替代（零拷贝）。

语法规则：static 成员函数

class Math {
public:
    static int add(int a, int b) { return a + b; }
    static const double PI;
};
const double Math::PI = 3.14159; // static 成员变量类外定义

int sum = Math::add(3, 4); // 直接通过类名调用

💡 举一反三：static 成员函数属于类而非对象，没有 this 指针。适合工具函数、工厂方法、单例获取。不能访问非 static 成员。
