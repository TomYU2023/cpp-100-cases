案例 72：std::tuple 元组操作

深入学习 tuple 的创建、访问、拼接和遍历技巧。

核心知识点：tuple、tie、tuple_cat、get、结构化绑定

完整代码（含逐行注释）

// 案例 72：std::tuple 元组操作
// tuple 的创建、解包、拼接和遍历
#include <iostream>
#include <tuple>
#include <string>
#include <type_traits>
using namespace std;

// 打印 tuple 的辅助函数（编译期递归）
template <size_t Index = 0, typename... Types>
void printTuple(const tuple<Types...>& t) {
    if constexpr (Index < sizeof...(Types)) {
        if constexpr (Index > 0) cout << ", ";
        cout << get<Index>(t);
        printTuple<Index + 1>(t);
    }
}

template <typename... Types>
void print(const string& label, const tuple<Types...>& t) {
    cout << label << "(";
    printTuple(t);
    cout << ")" << endl;
}

int main() {
    // 1. 创建 tuple
    auto t1 = make_tuple(42, 3.14, string("hello"));
    tuple<int, string, bool> t2(25, "world", true);

    print("t1", t1);
    print("t2", t2);

    // 2. 访问元素
    cout << "\nt1<0>: " << get<0>(t1) << endl;
    cout << "t1<1>: " << get<1>(t1) << endl;
    cout << "t1<2>: " << get<2>(t1) << endl;

    // 按类型访问（仅当类型唯一时可用）
    cout << "t2中的string: " << get<string>(t2) << endl;

    // 3. tie 解包
    int a;
    double b;
    string c;
    tie(a, b, c) = t1;
    cout << "\ntie 解包: " << a << ", " << b << ", " << c << endl;

    // tie 忽略某些值
    int x;
    tie(x, ignore, ignore) = t1;
    cout << "只取第一个: " << x << endl;

    // 4. tuple_cat 拼接
    auto t3 = tuple_cat(t1, make_tuple('A', 99));
    print("\n拼接后 t3", t3);

    // 5. 获取 tuple 大小
    cout << "\nt1 元素数: " << tuple_size<decltype(t1)>::value << endl;

    // 6. 交换
    tuple<int, int> ta(1, 2), tb(3, 4);
    ta.swap(tb);
    print("交换后 ta", ta);

    return 0;
}

t2中的string: world

交换后 ta(3, 4)
