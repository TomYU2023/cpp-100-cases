案例 38：mutex+lock_guard

使用 mutex 和 lock_guard 保护共享资源，实现线程安全的计数器。

核心知识点：mutex、lock_guard、临界区、死锁预防

完整代码（含逐行注释）

// 案例 38：mutex + lock_guard
// 使用互斥锁保护共享数据
#include <iostream>
#include <thread>
#include <mutex>
#include <vector>
using namespace std;

// 线程安全的计数器
class ThreadSafeCounter {
private:
    mutable mutex mtx; // mutable 允许在 const 函数中加锁
    int value;

public:
    ThreadSafeCounter() : value(0) {}

    // 递增
    void increment() {
        lock_guard<mutex> lock(mtx); // 构造时自动加锁
        value++;
        // lock 在离开作用域时自动解锁
    }

    // 递减
    void decrement() {
        lock_guard<mutex> lock(mtx);
        value--;
    }

    // 读取（const 但需要加锁，所以 mtx 声明为 mutable）
    int get() const {
        lock_guard<mutex> lock(mtx);
        return value;
    }
};

// 不安全版本（用于对比）
class UnsafeCounter {
private:
    int value;
public:
    UnsafeCounter() : value(0) {}
    void increment() { value++; } // 非原子操作！
    int get() const { return value; }
};

int main() {
    const int NUM_THREADS = 10;
    const int ITERATIONS = 100000;

    // 测试线程安全版本
    {
        ThreadSafeCounter safeCounter;
        vector<thread> threads;

        for (int i = 0; i < NUM_THREADS; i++) {
            threads.emplace_back([&safeCounter, ITERATIONS]() {
                for (int j = 0; j < ITERATIONS; j++) {
                    safeCounter.increment();
                }
            });
        }
        for (auto& t : threads) t.join();

        cout << "线程安全计数器: " << safeCounter.get()
             << " (期望: " << NUM_THREADS * ITERATIONS << ")" << endl;
    }

    // 测试不安全版本
    {
        UnsafeCounter unsafeCounter;
        vector<thread> threads;

        for (int i = 0; i < NUM_THREADS; i++) {
            threads.emplace_back([&unsafeCounter, ITERATIONS]() {
                for (int j = 0; j < ITERATIONS; j++) {
                    unsafeCounter.increment();
                }
            });
        }
        for (auto& t : threads) t.join();

        cout << "非安全计数器: " << unsafeCounter.get()
             << " (期望: " << NUM_THREADS * ITERATIONS << ")" << endl;
    }

    return 0;
}

非安全计数器: 186283 (期望: 1000000)
