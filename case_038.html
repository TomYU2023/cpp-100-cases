10 class Counter {

22 };

24 // 线程安全的容器（读写锁版）
25 class ThreadSafeVector {
26 mutable shared_mutex rw_mtx_; 读写锁
27 vector<string> data_;
28 public:
29 void add(const string& item) {
30 unique_lock<shared_mutex> lock(rw_mtx_); 写锁（独占）
31 data_.push_back(item);
32 }
33 string get(size_t idx) const {
34 shared_lock<shared_mutex> lock(rw_mtx_); 读锁（共享）
35 return idx < data_.size() ? data_[idx] : "";
36 }
37 size_t size() const {
38 shared_lock<shared_mutex> lock(rw_mtx_);
39 return data_.size();
40 }
41 };

新语法讲解

lock_guard / unique_lock / scoped_lock

lock_guard：最简单的 RAII 锁，构造时 lock，析构时 unlock，不可手动解锁。unique_lock：更灵活，支持 defer（延迟加锁）、try_lock、手动 unlock，开销略大于 lock_guard。scoped_lock（C++17）：可同时锁多个 mutex，避免死锁。

举一反三

// 推荐选择
lock_guard lg(m);  // 简单场景，性能最优
unique_lock ul(m); // 需要条件变量或手动控制
scoped_lock lk(m1, m2);  // 需要同时锁多个

mutable 与 const 方法加锁

锁操作修改了 mutex 的内部状态，但 const 方法不允许修改成员变量。解决方案：把 mutex 声明为 mutable，允许在 const 方法中修改。这是 mutex 用 mutable 的唯一正当场景。

举一反三

class C {
    mutable mutex mtx_;
    int data_;
public:
    int get() const {
        lock_guard lg(mtx_); // 不加 mutable 编译不过
        return data_;
    }
};

举一反三

递归锁 recursive_mutex

recursive_mutex rm;
void f() { lock_guard lg(rm); g(); } // f 中调用 g
void g() { lock_guard lg(rm); /* OK */ } // 同一线程可多次加锁

同一线程可重复加锁，但尽量不用，说明设计有问题

timed_mutex 超时加锁

timed_mutex tm;
if(tm.try_lock_for(100ms)) {
  // 成功加锁
  tm.unlock();
} else {
  // 超时，做别的事
}

避免无限等待，提高响应性

死锁的四个必要条件

// 1. 互斥
// 2. 持有并等待
// 3. 不可抢占
// 4. 循环等待
// 解决办法：统一加锁顺序，或用 scoped_lock
scoped_lock lock(mtx1, mtx2); // 原子加锁

破坏任意一个条件就能预防死锁
