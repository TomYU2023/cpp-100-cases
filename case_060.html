案例 60：协程 generator

使用 C++20 协程实现一个生成器，惰性生成数列。

核心知识点：C++20 协程、co_yield、generator 模式

完整代码（含逐行注释）

// 案例 60：协程 Generator
// C++20 协程实现惰性生成器
#include <iostream>
#include <coroutine>
#include <optional>
using namespace std;

// 简易 Generator（惰性生成序列）
template <typename T>
class Generator {
public:
    struct promise_type;
    using Handle = coroutine_handle<promise_type>;

    struct promise_type {
        optional<T> current_value;

        Generator get_return_object() {
            return Generator(Handle::from_promise(*this));
        }

        suspend_always initial_suspend() { return {}; }
        suspend_always final_suspend() noexcept { return {}; }

        // co_yield value 时调用
        suspend_always yield_value(T value) {
            current_value = value;
            return {};
        }

        void return_void() {}
        void unhandled_exception() { terminate(); }
    };

private:
    Handle handle;

public:
    explicit Generator(Handle h) : handle(h) {}
    ~Generator() { if (handle) handle.destroy(); }

    // 禁止拷贝
    Generator(const Generator&) = delete;
    Generator& operator=(const Generator&) = delete;

    // 支持移动
    Generator(Generator&& other) noexcept : handle(other.handle) {
        other.handle = nullptr;
    }

    // 获取下一个值
    optional<T> next() {
        if (!handle || handle.done()) return nullopt;
        handle.resume(); // 恢复协程执行
        if (handle.done()) return nullopt;
        return handle.promise().current_value;
    }
};

// 生成斐波那契数列的协程
Generator<long long> fibonacci(int n) {
    long long a = 0, b = 1;
    for (int i = 0; i < n; i++) {
        co_yield a;          // 暂停并返回值
        long long next = a + b;
        a = b;
        b = next;
    }
}

// 生成自然数序列
Generator<int> range(int start, int end, int step = 1) {
    for (int i = start; i < end; i += step) {
        co_yield i;
    }
}

int main() {
    // 测试斐波那契生成器
    cout << "斐波那契数列前 10 项:" << endl;
    auto fib = fibonacci(10);
    while (auto val = fib.next()) {
        cout << *val << " ";
    }
    cout << endl;

    // 测试范围生成器
    cout << "\nrange(5, 15, 2):" << endl;
    auto r = range(5, 15, 2);
    while (auto val = r.next()) {
        cout << *val << " ";
    }
    cout << endl;

    return 0;
}

  fib(2) = 1
  fib(3) = 2

  fib(5) = 5

偶数平方生成器（范围 1-20）：
  2^2=4  4^2=16  6^2=36  8^2=64  10^2=100
  12^2=144  14^2=196  16^2=256  18^2=324  20^2=400

Generator 与普通函数对比：
  普通函数：一次返回全部结果（需要 vector 存储）
  Generator：按需逐个产出（内存占用恒定）
