案例 60：协程 generator

使用 C++20 协程实现一个生成器，惰性生成数列。

核心知识点：C++20 协程、co_yield、generator 模式

完整代码（含逐行注释）

// 案例 60：协程 Generator
// C++20 协程实现惰性生成器
#include <iostream>
#include <coroutine>
#include <optional>
using namespace std;

// 简易 Generator（惰性生成序列）
template <typename T>
class Generator {
public:
    struct promise_type;
    using Handle = coroutine_handle<promise_type>;

    struct promise_type {
        optional<T> current_value;

        Generator get_return_object() {
            return Generator(Handle::from_promise(*this));
        }

        suspend_always initial_suspend() { return {}; }
        suspend_always final_suspend() noexcept { return {}; }

        // co_yield value 时调用
        suspend_always yield_value(T value) {
            current_value = value;
            return {};
        }

        void return_void() {}
        void unhandled_exception() { terminate(); }
    };

private:
    Handle handle;

public:
    explicit Generator(Handle h) : handle(h) {}
    ~Generator() { if (handle) handle.destroy(); }

    // 禁止拷贝
    Generator(const Generator&) = delete;
    Generator& operator=(const Generator&) = delete;

    // 支持移动
    Generator(Generator&& other) noexcept : handle(other.handle) {
        other.handle = nullptr;
    }

    // 获取下一个值
    optional<T> next() {
        if (!handle || handle.done()) return nullopt;
        handle.resume(); // 恢复协程执行
        if (handle.done()) return nullopt;
        return handle.promise().current_value;
    }
};

// 生成斐波那契数列的协程
Generator<long long> fibonacci(int n) {
    long long a = 0, b = 1;
    for (int i = 0; i < n; i++) {
        co_yield a;          // 暂停并返回值
        long long next = a + b;
        a = b;
        b = next;
    }
}

// 生成自然数序列
Generator<int> range(int start, int end, int step = 1) {
    for (int i = start; i < end; i += step) {
        co_yield i;
    }
}

int main() {
    // 测试斐波那契生成器
    cout << "斐波那契数列前 10 项:" << endl;
    auto fib = fibonacci(10);
    while (auto val = fib.next()) {
        cout << *val << " ";
    }
    cout << endl;

    // 测试范围生成器
    cout << "\nrange(5, 15, 2):" << endl;
    auto r = range(5, 15, 2);
    while (auto val = r.next()) {
        cout << *val << " ";
    }
    cout << endl;

    return 0;
}

语法规则：C++20 协程基础

#include <coroutine>
// co_yield: 暂停并产生值
// co_await: 等待异步操作
// co_return: 返回最终值
// 协程在暂停点保存状态，恢复后继续执行

举一反三：C++20 协程是无栈协程，编译器将函数体转换为状态机。三个关键字：co_yield（生成）、co_await（等待）、co_return（返回）。需实现 promise_type。

语法规则：Generator 实现要点

struct promise_type {
    suspend_always yield_value(T v) { curr = v; return {}; }
    // initial_suspend: 是否开始时就暂停
    // final_suspend: 结束后是否暂停
};

举一反三：optimistic_suspend 返回 suspend_always 表示协程需要手动 resume() 才开始。suspend_never 表示立即开始。final_suspend 设为 suspend_always 防止协程自动销毁。

语法规则：协程与迭代器对比

// 迭代器：需手动维护状态
// 协程 Generator：状态由编译器自动管理
// 优势：代码简洁，惰性求值，无限序列
// 劣势：学习曲线陡峭，调试困难

举一反三：Generator 模式的协程特别适合惰性序列（斐波那契、排列、文件行读取等）。Python 的 yield 和 C# 的 yield return 也是类似概念。
