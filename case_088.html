案例 88：简易 Git 实现

从零实现 init/add/commit/log 四个命令，理解 Git 的 blob/tree/commit 对象模型和哈希寻址机制。

核心知识点：Git 对象模型、std::hash、filesystem 遍历、istreambuf_iterator 文件读取、内容寻址存储

完整代码（含逐行注释）

// 案例 88：简易 Git 实现 —— init / add / commit / log
// 从零实现 Git 的核心机制：blob 对象、树对象、提交链、哈希寻址。
// 帮助理解 Git 底层数据模型的本质，而非仅会使用命令行。

#include <iostream>
#include <fstream>
#include <sstream>
#include <string>
#include <map>
#include <vector>
#include <filesystem>
#include <iomanip>
#include <ctime>
#include <functional>

namespace fs = std::filesystem;

// ─── 简易 SHA-1（教学用，非密码学安全）──────────
// 真实的 Git 使用 SHA-1 哈希命名对象。这里用简单哈希模拟。
std::string simpleHash(const std::string& data) {
    std::hash<std::string> hasher;
    size_t h = hasher(data);
    std::stringstream ss;
    ss << std::hex << std::setfill('0') << std::setw(16) << h;
    return ss.str();
}

const std::string OBJECTS = ".mygit/objects/";

// ─── 写入一个对象到 .mygit/objects/ ──────────────
void writeObject(const std::string& type, const std::string& data) {
    std::string header = type + " " + std::to_string(data.size()) + "\0";
    std::string full = header + data;
    std::string hash = simpleHash(full);
    fs::create_directories(OBJECTS + hash.substr(0, 2));
    std::ofstream out(OBJECTS + hash.substr(0,2) + "/" + hash.substr(2),
        std::ios::binary);
    out << full;
    std::cout << "  写入对象 " << type << " " << hash << std::endl;
}

// ─── Git init ────────────────────────────────────
void gitInit() {
    fs::create_directories(".mygit/objects");
    fs::create_directories(".mygit/refs/heads");
    std::ofstream head(".mygit/HEAD");
    head << "ref: refs/heads/master\n";  // 当前分支指针
    std::cout << "已初始化空仓库于 .mygit/" << std::endl;
}

// ─── Git add：将文件内容存储为 blob ─────────────
void gitAdd(const std::string& filename) {
    if (!fs::exists(filename)) {
        std::cerr << "文件不存在: " << filename << std::endl;
        return;
    }
    std::ifstream in(filename, std::ios::binary);
    std::string content((std::istreambuf_iterator<char>(in)),
                         std::istreambuf_iterator<char>());
    writeObject("blob", content);
    std::cout << "已暂存: " << filename << std::endl;
}

// ─── Git commit：创建树和提交对象 ────────────────
void gitCommit(const std::string& message) {
    // 简化：直接将文件列表写入树对象
    std::string treeData;
    for (const auto& entry : fs::directory_iterator(".")) {
        if (!entry.is_regular_file()) continue;
        std::string name = entry.path().filename().string();
        if (name == ".mygit") continue;  // 忽略仓库目录
        // 读取文件，生成 blob 哈希
        std::ifstream in(entry.path(), std::ios::binary);
        std::string content((std::istreambuf_iterator<char>(in)),
                             std::istreambuf_iterator<char>());
        std::string blobHash = simpleHash("blob " +
            std::to_string(content.size()) + "\0" + content);
        treeData += "100644 blob " + blobHash + " " + name + "\n";
    }

    // 写入树对象
    writeObject("tree", treeData);

    // 构造提交对象
    std::string treeHash = simpleHash("tree " +
        std::to_string(treeData.size()) + "\0" + treeData);

    // 读取父提交
    std::string parentHash;
    std::ifstream refFile(".mygit/refs/heads/master");
    if (refFile) std::getline(refFile, parentHash);

    time_t now = time(0);
    std::string commitData = "tree " + treeHash + "\n";
    if (!parentHash.empty()) commitData += "parent " + parentHash + "\n";
    commitData += "\n" + message + "\n";

    // 写入提交对象
    std::string header = "commit " + std::to_string(commitData.size()) + "\0";
    std::string full = header + commitData;
    std::string commitHash = simpleHash(full);

    fs::create_directories(OBJECTS + commitHash.substr(0, 2));
    std::ofstream out(OBJECTS + commitHash.substr(0,2) + "/" +
        commitHash.substr(2), std::ios::binary);
    out << full;

    // 更新分支引用
    std::ofstream masterRef(".mygit/refs/heads/master");
    masterRef << commitHash;

    std::cout << "[" << commitHash.substr(0, 7) << "] " << message << std::endl;
}

// ─── Git log：遍历提交链 ─────────────────────────
void gitLog() {
    std::ifstream refFile(".mygit/refs/heads/master");
    if (!refFile) { std::cout << "尚无提交。" << std::endl; return; }
    std::string hash;
    std::getline(refFile, hash);
    int n = 0;
    while (!hash.empty()) {
        // 读取提交对象
        std::string objPath = OBJECTS + hash.substr(0,2) + "/" + hash.substr(2);
        std::ifstream in(objPath, std::ios::binary);
        if (!in) break;
        std::string content((std::istreambuf_iterator<char>(in)),
                             std::istreambuf_iterator<char>());
        // 跳过 header（type size\0）
        auto nullPos = content.find('\0');
        if (nullPos == std::string::npos) break;
        std::string body = content.substr(nullPos + 1);

        std::cout << "commit " << hash.substr(0, 7) << std::endl;
        // 提取摘要（第一行非空的行）
        size_t msgStart = body.find("\n\n");
        if (msgStart != std::string::npos) {
            std::string msg = body.substr(msgStart + 2);
            size_t nl = msg.find('\n');
            if (nl != std::string::npos) msg = msg.substr(0, nl);
            std::cout << "    " << msg << std::endl;
        }

        // 查找父提交
        size_t parentPos = body.find("parent ");
        if (parentPos == std::string::npos) break;
        size_t nl = body.find('\n', parentPos);
        hash = body.substr(parentPos + 7, nl - parentPos - 7);
        std::cout << std::endl;
        n++;
    }
}

// ─── CLI ─────────────────────────────────────────
int main(int argc, char* argv[]) {
    if (argc < 2) {
        std::cout << "用法: git init | add <file> | commit <msg> | log" << std::endl;
        return 1;
    }
    std::string cmd = argv[1];

    if (cmd == "init") gitInit();
    else if (cmd == "add" && argc > 2) gitAdd(argv[2]);
    else if (cmd == "commit" && argc > 2) gitCommit(argv[2]);
    else if (cmd == "log") gitLog();
    else std::cout << "未知命令。" << std::endl;

    return 0;
}

语法规则：std::hash 标准哈希函数

#include <functional>
std::hash<std::string> hasher;
size_t h = hasher("hello");  // 将字符串映射为 size_t
// std::hash<T> 对 std::string、int、指针等有特化实现

💡 举一反三：std::hash 是 C++ 标准库为每种类型提供的哈希函数。unordered_map 内部就用它。注意：同一程序多次运行，同一输入的 hash 值可能不同（实现定义），不要用于持久化。Git 用的是 SHA-1，这里用 hash 仅为教学演示。

语法规则：std::istreambuf_iterator 一次性读取文件

#include <fstream>
#include <string>
std::ifstream in("file.txt", std::ios::binary);
std::string content(
    std::istreambuf_iterator<char>(in),
    std::istreambuf_iterator<char>());  // 默认构造 = 流结束哨兵

💡 举一反三：这是将整个文件读入 string 的最简洁方式。istreambuf_iterator 逐字符读取（包括空白符），效率高于逐行 getline 拼接。binary 模式防止 \r\n 转为 \n。

语法规则：std::filesystem 目录遍历

#include <filesystem>
namespace fs = std::filesystem;
for (const auto& entry : fs::directory_iterator(".")) {
    if (entry.is_regular_file())
        std::cout << entry.path().filename() << std::endl;
}

💡 举一反三：C++17 引入的 filesystem 库提供了跨平台的目录操作。directory_iterator 遍历目录，recursive_directory_iterator 递归遍历子目录。相比平台 API，filesystem 写法一致、可移植。
