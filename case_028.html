案例 28：find_if/count_if

使用 find_if 和 count_if 算法查找满足条件的元素，练习谓词函数的多种写法。

核心知识点：find_if、count_if、谓词（函数对象、Lambda）

完整代码（含逐行注释）

// 案例 28：find_if / count_if
// 使用谓词查找和统计满足条件的元素
#include <iostream>
#include <vector>
#include <algorithm>
#include <string>
using namespace std;

// 函数对象：判断是否及格
struct IsPass {
    int passLine;
    IsPass(int line) : passLine(line) {}
    bool operator()(int score) const {
        return score >= passLine;
    }
};

int main() {
    vector<int> scores = {45, 78, 92, 60, 55, 88, 73, 49, 95, 61};

    // 1. count_if：统计及格人数（使用 Lambda）
    int pass = count_if(scores.begin(), scores.end(),
                        [](int s) { return s >= 60; });
    cout << "总人数: " << scores.size() << endl;
    cout << "及格人数(>=60): " << pass << endl;
    cout << "不及格人数: " << scores.size() - pass << endl;

    // 2. count_if：统计优秀人数（使用函数对象）
    IsPass excellent(85);
    int exc = count_if(scores.begin(), scores.end(), excellent);
    cout << "优秀人数(>=85): " << exc << endl;

    // 3. find_if：找到第一个不及格的学生
    auto it = find_if(scores.begin(), scores.end(),
                      [](int s) { return s < 60; });
    if (it != scores.end()) {
        cout << "第一个不及格分数: " << *it
             << " (位置: " << (it - scores.begin()) << ")" << endl;
    }

    // 4. 找出所有优秀分数的位置
    cout << "优秀分数位置: ";
    auto pos = scores.begin();
    while ((pos = find_if(pos, scores.end(),
            [](int s) { return s >= 85; })) != scores.end()) {
        cout << (pos - scores.begin()) << "(" << *pos << ") ";
        pos++; // 移到下一个继续搜索
    }
    cout << endl;

    return 0;
}

语法规则：find_if / count_if

#include <algorithm>
auto it = find_if(v.begin(), v.end(), [](int x) { return x < 0; });
int n = count_if(v.begin(), v.end(), [](int x) { return x > 10; });

举一反三：_if 后缀版本接受谓词而非值。find 查找等于给定值的元素，find_if 按条件查找。还有 find_if_not（C++11 起）取反版本。

语法规则：谓词的三种写法

// 1. Lambda（最常用）
[](int s) { return s >= 60; }
// 2. 函数指针
bool isPass(int s) { return s >= 60; }
// 3. 函数对象（可携状态）
struct Check { int n; bool operator()(int x) { return x > n; } };

举一反三：Lambda 最简洁；函数对象可携带状态适合复杂逻辑；函数指针适合复用已有函数。C++20 支持 ranges::find_if 直接接受容器。

语法规则：循环中 find_if 查找所有匹配

auto it = v.begin();
while ((it = find_if(it, v.end(), pred)) != v.end()) {
    // 处理 *it
    ++it; // 移到下一个继续
}

举一反三：这是找到所有满足条件元素的惯用法。每次 find_if 从上次找到的位置 +1 开始。如果不 ++it 会无限循环找到同一个元素。
