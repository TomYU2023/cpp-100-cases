案例 97：std::apply tuple 调用

使用 apply 将 tuple 解包为函数参数，实现运行时动态参数调用。

核心知识点：apply、tuple 解包、变参转发、通用调用

完整代码（含逐行注释）

// 案例 97：std::apply tuple 调用
// 将 tuple 元素展开为函数参数
#include <iostream>
#include <tuple>
#include <functional>
#include <string>
#include <vector>
using namespace std;

// 接受 3 个参数的函数
void printPerson(const string& name, int age, double height) {
    cout << "姓名: " << name
         << ", 年龄: " << age
         << ", 身高: " << height << "m" << endl;
}

int sum(int a, int b, int c, int d) {
    return a + b + c + d;
}

int main() {
    // 1. 基本用法：tuple 展开为函数参数
    auto person = make_tuple("Alice", 25, 1.68);
    apply(printPerson, person);

    auto nums = make_tuple(1, 2, 3, 4);
    cout << "sum = " << apply(sum, nums) << endl;

    // 2. apply 配合 Lambda
    auto data = make_tuple(10, 20, 30);

    // 展开 tuple 到 Lambda 参数
    auto result = apply([](int x, int y, int z) {
        return x * y + z;
    }, data);
    cout << "10*20+30 = " << result << endl;

    // 3. 动态调用：根据条件构建参数
    vector<tuple<string, int, double>> people = {
        {"Bob", 30, 1.75},
        {"Charlie", 22, 1.82},
        {"David", 28, 1.70}
    };

    cout << "\n=== 批量调用 ===" << endl;
    for (const auto& p : people) {
        apply(printPerson, p);
    }

    // 4. apply + invoke 的组合（元编程常见）
    // 从 tuple 中调用对象的成员函数
    struct Task {
        void execute(int id, const string& msg) {
            cout << "Task " << id << ": " << msg << endl;
        }
    };

    Task task;
    auto params = make_tuple(42, "处理完成");
    apply([&task](auto&&... args) {
        task.execute(forward<decltype(args)>(args)...);
    }, params);

    return 0;
}

sum = 10
10*20+30 = 230

姓名: Charlie, 年龄: 22, 身高: 1.82m
姓名: David, 年龄: 28, 身高: 1.7m
Task 42: 处理完成
