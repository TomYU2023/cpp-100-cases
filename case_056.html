案例 56：cout格式化全攻略

系统梳理 iomanip 的所有格式化操作符，制表、对齐、精度控制。

核心知识点：iomanip、setw、setprecision、格式标志

完整代码（含逐行注释）

// 案例 56：cout 格式化全攻略
// 系统学习输出格式化控制
#include <iostream>
#include <iomanip>
#include <string>
using namespace std;

int main() {
    cout << "=== 1. 宽度控制 (setw) ===" << endl;
    cout << setw(10) << "Name" << setw(8) << "Age" << setw(12) << "Score" << endl;
    cout << setw(10) << "Alice" << setw(8) << 25 << setw(12) << 95.5 << endl;
    cout << setw(10) << "Bob" << setw(8) << 30 << setw(12) << 88.0 << endl;

    cout << "\n=== 2. 对齐方式 ===" << endl;
    cout << "左对齐: " << left << setw(15) << "Hello" << "|" << endl;
    cout << "右对齐: " << right << setw(15) << "Hello" << "|" << endl;
    cout << "内部对齐: " << internal << setw(15) << -42 << "|" << endl;

    cout << "\n=== 3. 精度控制 ===" << endl;
    double pi = 3.14159265358979;
    cout << "默认: " << pi << endl;
    cout << "precision(3): " << setprecision(3) << pi << endl;
    cout << "fixed + prec(3): " << fixed << setprecision(3) << pi << endl;
    cout << "scientific + prec(3): " << scientific << setprecision(3) << pi << endl;

    cout << "\n=== 4. 进制表示 ===" << endl;
    int num = 255;
    cout << "十进制: " << dec << num << endl;
    cout << "十六进制: " << hex << showbase << num << endl;
    cout << "八进制: " << oct << showbase << num << endl;
    cout << dec << noshowbase; // 恢复默认

    cout << "\n=== 5. 布尔值显示 ===" << endl;
    cout << "默认: " << true << " " << false << endl;
    cout << "boolalpha: " << boolalpha << true << " " << false << endl;

    cout << "\n=== 6. 填充字符 ===" << endl;
    cout << setfill('*') << setw(15) << right << 42 << endl;
    cout << setfill(' '); // 恢复默认

    cout << "\n=== 7. 显示正号和小数点 ===" << endl;
    cout << showpos << 42 << " " << -42 << endl;
    cout << showpoint << 10.0 << endl;
    cout << noshowpos << noshowpoint;

    return 0;
}

语法规则：setw 宽度控制

cout << setw(10) << "Name" << setw(8) << 25;
// setw 只对紧接着的一次输出生效
// 默认右对齐，左对齐用 cout << left;

举一反三：setw 是唯一只对一次输出生效的 manipulator（其他都是永久性的）。设置宽度后必须紧接着输出，且每次输出后自动重置为 0。

语法规则：setprecision 精度

cout << fixed << setprecision(3) << pi;
// fixed: 固定小数位数
// scientific: 科学计数法
// defaultfloat: 默认（自动选择）

举一反三：C++11 后引入 defaultfloat 恢复默认。setprecision 控制的是有效数字位数（defaultfloat）或小数位数（fixed/scientific）。fixed 对小数值友好，scientific 对大数值友好。

语法规则：进制操纵符

cout << hex << showbase << 255; // 0xff
cout << oct << 255;            // 377
cout << dec; // 恢复十进制
// showbase 添加前缀 0x/0

举一反三：hex/oct/dec 是永久性设置。showbase 影响 hex(0x) 和 oct(0)。noshowbase 取消前缀。setbase(n) (n=8,10,16) 一次性设置，受 支持。
