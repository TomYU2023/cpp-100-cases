案例 74：std::gcd/lcm 数学工具

使用 C++17 的 gcd 和 lcm 简化数学计算，以及相关的数值算法。

核心知识点：gcd、lcm、数值算法、欧几里得算法

完整代码（含逐行注释）

// 案例 74：std::gcd / lcm 数学工具
// 最大公约数和最小公倍数
#include <iostream>
#include <numeric>
#include <vector>
#include <string>
using namespace std;

// 分数简化
struct Fraction {
    long long num;   // 分子
    long long den;   // 分母

    Fraction(long long n, long long d) : num(n), den(d) {
        simplify();
    }

    void simplify() {
        if (den == 0) return;
        // 处理负号：保持分母为正
        if (den < 0) { num = -num; den = -den; }
        long long g = gcd(abs(num), abs(den));
        num /= g;
        den /= g;
    }

    string str() const {
        if (den == 1) return to_string(num);
        return to_string(num) + "/" + to_string(den);
    }
};

int main() {
    // 1. gcd 基本用法
    cout << "=== GCD (最大公约数) ===" << endl;
    cout << "gcd(12, 18) = " << gcd(12, 18) << endl;
    cout << "gcd(48, 180) = " << gcd(48, 180) << endl;
    cout << "gcd(17, 13) = " << gcd(17, 13) << endl; // 互质 → 1
    cout << "gcd(0, 5) = " << gcd(0, 5) << endl;     // 5

    // 2. lcm 基本用法
    cout << "\n=== LCM (最小公倍数) ===" << endl;
    cout << "lcm(12, 18) = " << lcm(12, 18) << endl;
    cout << "lcm(4, 6) = " << lcm(4, 6) << endl;
    cout << "lcm(7, 9) = " << lcm(7, 9) << endl;     // 7*9=63

    // 3. 分数简化
    cout << "\n=== 分数简化 ===" << endl;
    Fraction f1(6, 8);    // 6/8 = 3/4
    Fraction f2(15, -25); // -15/25 = -3/5
    Fraction f3(42, 56);  // 42/56 = 3/4
    cout << "6/8 = " << f1.str() << endl;
    cout << "15/(-25) = " << f2.str() << endl;
    cout << "42/56 = " << f3.str() << endl;

    // 4. 批量计算多个数的 gcd
    cout << "\n=== 多个数的 GCD ===" << endl;
    vector<long long> nums = {120, 180, 240, 300};
    long long g = nums[0];
    for (size_t i = 1; i < nums.size(); i++) {
        g = gcd(g, nums[i]);
    }
    cout << "gcd(120, 180, 240, 300) = " << g << endl;

    return 0;
}

语法规则：gcd 和 lcm

#include <numeric>
int g = gcd(12, 18); // 6
int l = lcm(12, 18); // 36
// C++17 标准，头文件 <numeric>

举一反三：gcd 使用欧几里得算法（辗转相除法）。lcm(a,b) = a * b / gcd(a,b)。两个函数返回非负值。C++17 前可用 __gcd（GCC 扩展）或手写。

语法规则：gcd 的数学性质

gcd(a, 0) = |a|
gcd(0, 0) = 0
gcd(a, b) = gcd(b, a mod b)
// 可扩展到负数：gcd(-12, 18) = 6

举一反三：gcd 结合律使批量计算简单（fold）。gcd 在密码学（RSA）、离散数学、分数运算中广泛使用。lcm 在周期问题（两个齿轮对齐时间）中有用。

语法规则：分数简化模式

void simplify() {
    long long g = gcd(abs(num), abs(den));
    num /= g; den /= g;
    if (den < 0) { num = -num; den = -den; }
}

举一反三：化简为最简分数是 gcd 的经典应用。先取绝对值再求 gcd。约定分母为正、互质。分子 0 时 gcd(0, den) = den，化简结果为 0/1。
