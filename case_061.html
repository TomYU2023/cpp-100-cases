案例 61：线程池(手写)

从零实现一个固定大小的线程池，支持任务提交和结果获取。

核心知识点：线程池架构、任务队列、condition_variable、future

完整代码（含逐行注释）

// 案例 61：线程池（手写）
// 固定大小的线程池，支持任务提交和获取返回值
#include <iostream>
#include <thread>
#include <mutex>
#include <condition_variable>
#include <queue>
#include <vector>
#include <functional>
#include <future>
#include <type_traits>
using namespace std;

class ThreadPool {
private:
    vector<thread> workers;               // 工作线程
    queue<function<void()>> tasks;        // 任务队列
    mutex queueMutex;                     // 队列锁
    condition_variable condition;         // 条件变量
    bool stop;                            // 停止标志

public:
    // 构造函数：启动指定数量的工作线程
    explicit ThreadPool(size_t numThreads) : stop(false) {
        for (size_t i = 0; i < numThreads; i++) {
            workers.emplace_back([this]() {
                // 工作线程的主循环
                while (true) {
                    function<void()> task;

                    {
                        unique_lock<mutex> lock(queueMutex);

                        // 等待任务或停止信号
                        condition.wait(lock, [this]() {
                            return stop || !tasks.empty();
                        });

                        // 停止且队列为空时退出
                        if (stop && tasks.empty()) return;

                        // 取出一个任务
                        task = move(tasks.front());
                        tasks.pop();
                    }

                    // 在锁外执行任务
                    task();
                }
            });
        }
    }

    // 提交任务并获取 future
    template <typename F, typename... Args>
    auto submit(F&& f, Args&&... args)
        -> future<typename result_of<F(Args...)>::type>
    {
        using ReturnType = typename result_of<F(Args...)>::type;

        // 用 packaged_task 包装任务
        auto taskPtr = make_shared<packaged_task<ReturnType()>>(
            bind(forward<F>(f), forward<Args>(args)...)
        );

        future<ReturnType> result = taskPtr->get_future();

        {
            unique_lock<mutex> lock(queueMutex);
            if (stop) throw runtime_error("ThreadPool 已停止");

            // 将 packaged_task 包装为 void() 函数入队
            tasks.emplace([taskPtr]() { (*taskPtr)(); });
        }
        condition.notify_one();

        return result;
    }

    // 析构函数：等待所有任务完成
    ~ThreadPool() {
        {
            unique_lock<mutex> lock(queueMutex);
            stop = true;
        }
        condition.notify_all();
        for (thread& worker : workers) {
            worker.join();
        }
    }
};

int main() {
    ThreadPool pool(4); // 4 个工作线程
    vector<future<int>> results;

    // 提交 10 个任务
    for (int i = 0; i < 10; i++) {
        results.push_back(pool.submit([i]() {
            // 模拟耗时计算
            this_thread::sleep_for(chrono::milliseconds(100));
            return i * i;
        }));
    }

    // 获取结果
    for (int i = 0; i < (int)results.size(); i++) {
        cout << "任务 " << i << " 结果: " << results[i].get() << endl;
    }

    return 0;
}

语法规则：线程池架构

class ThreadPool {
    vector<thread> workers;        // 线程池
    queue<function<void()>> tasks; // 任务队列
    mutex mtx;                     // 保护队列
    condition_variable cv;         // 通知机制
};

举一反三：线程池避免频繁创建/销毁线程的开销。核心组件：工作线程集合 + 线程安全任务队列 + 同步机制。典型应用：Web服务器、数据库连接池、异步任务执行。

语法规则：packaged_task + future 返回结果

using ReturnType = result_of<F(Args...)>::type;
auto task = make_shared<packaged_task<ReturnType()>>(f);
future<ReturnType> result = task->get_future();
// packaged_task 包装可调用对象，future 获取结果

举一反三：packaged_task 是可移动的可调用对象包装器，其 get_future() 返回关联 future。任务执行完 future 就绪。C++17 可用 invoke_result 替代 result_of。

语法规则：工作线程主循环模式

while (true) {
    unique_lock<mutex> lock(mtx);
    cv.wait(lock, []{ return stop || !tasks.empty(); });
    if (stop && tasks.empty()) return;
    auto task = move(tasks.front());
    tasks.pop();
    lock.unlock(); // 尽早释放锁
    task();        // 在锁外执行
}

举一反三：工作线程在循环中等待任务。先获取锁检查队列，有任务则取出后在锁外执行（避免长时间持锁）。stop 标志 + 空队列作为退出条件。
