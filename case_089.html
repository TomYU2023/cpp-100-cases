案例 89：std::iota 递增填充

使用 iota 生成连续递增序列，结合其他算法快速初始化容器。

核心知识点：iota、序列生成、ranges::iota、容器初始化

完整代码（含逐行注释）

// 案例 89：std::iota 递增填充
// 从初值开始生成连续递增序列
#include <iostream>
#include <numeric>
#include <vector>
#include <algorithm>
#include <string>
#include <map>
using namespace std;

int main() {
    // 1. 基础：生成 1 到 N 的序列
    vector<int> v(10);
    iota(v.begin(), v.end(), 1); // 从 1 开始
    cout << "1 到 10: ";
    for (int x : v) cout << x << " ";
    cout << endl;

    // 2. 从任意值开始
    vector<int> v2(8);
    iota(v2.begin(), v2.end(), 100); // 从 100 开始
    cout << "100 到 107: ";
    for (int x : v2) cout << x << " ";
    cout << endl;

    // 3. 递减序列（用反向迭代器）
    vector<int> v3(10);
    iota(v3.rbegin(), v3.rend(), 1); // 反向填充
    cout << "10 到 1: ";
    for (int x : v3) cout << x << " ";
    cout << endl;

    // 4. 生成步长不为 1 的序列
    // iota 只支持步长 1，这里用 generate
    vector<int> v4(10);
    int counter = 0;
    generate(v4.begin(), v4.end(), [&counter]() {
        return counter++ * 2; // 0, 2, 4, 6, ...
    });
    cout << "步长为 2: ";
    for (int x : v4) cout << x << " ";
    cout << endl;

    // 5. 实际应用：生成索引映射表
    vector<int> indices(100);
    iota(indices.begin(), indices.end(), 0); // 0-99

    // 按某些规则重新排列索引...
    // 例如：把所有索引按 i%10 分组排序
    sort(indices.begin(), indices.end(),
         [](int a, int b) {
             if (a % 10 != b % 10) return a % 10 < b % 10;
             return a < b;
         });

    cout << "\n按 i%10 排序的索引 (前 20 个): ";
    for (int i = 0; i < 20; i++) cout << indices[i] << " ";
    cout << endl;

    return 0;
}

语法规则：std::iota

#include <numeric>
iota(v.begin(), v.end(), start);
// 依次填充: start, start+1, start+2, ...
// 对迭代器范围内的每个元素调用 ++value 并赋值

举一反三：iota 是最简单的序列生成器。名字来自 APL 语言的 ι（iota）符号。C++11 引入。C++20 的 ranges::iota 更灵活（支持无穷序列）。

语法规则：反向 iota

iota(v.rbegin(), v.rend(), 1);
// 从末尾向开头方向填充
// 结果: 10, 9, 8, ..., 1
// rbegin/rend 改变填充方向

举一反三：rbegin() 从最后一个元素开始。每次 ++value 但方向是从后往前。这是生成递减序列的巧妙用法（无需额外整理）。

语法规则：iota vs generate

// iota: 步长固定为 1，简单场景
// generate: 任意生成逻辑，灵活但稍复杂
// 推荐：简单序列用 iota，复杂用 generate
generate(v.begin(), v.end(), [n=0]() mutable { return n++ * 3; });

举一反三：iota 语义清晰（递增填充）。generate 适合非线性的生成逻辑（如斐波那契、随机、映射公式）。两者结合：iota 生成索引，再 transform 为实际值。
