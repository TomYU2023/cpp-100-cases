案例 10：复数类(运算符重载)

定义一个复数类，重载 +、-、*、<< 等运算符，深入理解运算符重载的语法和应用。

核心知识点：运算符重载、友元函数、const 成员函数

完整代码（含逐行注释）

// 案例 10：复数类（运算符重载）
// 重载 +、-、*、==、<< 运算符
#include <iostream>
using namespace std;

class Complex {
private:
    double real; // 实部
    double imag; // 虚部

public:
    Complex(double r = 0, double i = 0) : real(r), imag(i) {}

    // 成员函数重载 + 运算符
    Complex operator+(const Complex& other) const {
        return Complex(real + other.real, imag + other.imag);
    }

    // 成员函数重载 - 运算符
    Complex operator-(const Complex& other) const {
        return Complex(real - other.real, imag - other.imag);
    }

    // 成员函数重载 * 运算符：(a+bi)(c+di) = (ac-bd)+(ad+bc)i
    Complex operator*(const Complex& other) const {
        return Complex(
            real * other.real - imag * other.imag,
            real * other.imag + imag * other.real
        );
    }

    // 重载 == 运算符
    bool operator==(const Complex& other) const {
        return real == other.real && imag == other.imag;
    }

    // 友元函数重载 << 输出运算符
    friend ostream& operator<<(ostream& os, const Complex& c) {
        os << c.real;
        if (c.imag >= 0) os << " + " << c.imag << "i";
        else             os << " - " << -c.imag << "i";
        return os;
    }
};

int main() {
    Complex a(3, 2);  // 3 + 2i
    Complex b(1, 4);  // 1 + 4i

    Complex sum = a + b;    // 调用 operator+
    Complex diff = a - b;   // 调用 operator-
    Complex prod = a * b;   // 调用 operator*

    cout << "a = " << a << endl;
    cout << "b = " << b << endl;
    cout << "a + b = " << sum << endl;
    cout << "a - b = " << diff << endl;
    cout << "a * b = " << prod << endl;
    cout << "a == b ? " << (a == b ? "true" : "false") << endl;

    return 0;
}

语法规则：成员函数重载二元运算符

Complex operator+(const Complex& other) const {
    return Complex(real + other.real, imag + other.imag);
}

举一反三：成员函数重载时左操作数是 this。一般 +、-、*、/ 返回新对象（const 函数，不改原对象），+=、-= 返回引用（修改自身）。

语法规则：友元函数重载 <<

friend ostream& operator<<(ostream& os, const Complex& c) {
    os << c.real << " + " << c.imag << "i";
    return os;
}

举一反三：<< 的第一个操作数是 ostream，不能作为成员函数重载，必须用友元或全局函数。cin 的 >> 同理。返回 ostream& 支持链式调用。

语法规则：默认参数构造函数

Complex(double r = 0, double i = 0) : real(r), imag(i) {}

举一反三：默认参数让一个构造函数覆盖多种调用方式：Complex()，Complex(3)，Complex(3,2)。默认参数只能从右向左设置。
