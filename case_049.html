案例 49：工厂模式

实现简单工厂和工厂方法模式，动态创建不同类型的对象。

核心知识点：工厂模式、虚函数、智能指针

完整代码（含逐行注释）

// 案例 49：工厂模式
// 用工厂方法创建不同类型的形状
#include <iostream>
#include <memory>
#include <string>
#include <vector>
using namespace std;

// 抽象产品
class Shape {
public:
    virtual void draw() const = 0;
    virtual unique_ptr<Shape> clone() const = 0;
    virtual ~Shape() = default;
};

// 具体产品：Circle
class Circle : public Shape {
    double radius;
public:
    explicit Circle(double r = 1.0) : radius(r) {}
    void draw() const override {
        cout << "绘制圆形，半径=" << radius
             << "，面积=" << (3.14159 * radius * radius) << endl;
    }
    unique_ptr<Shape> clone() const override {
        return make_unique<Circle>(radius);
    }
};

// 具体产品：Rectangle
class Rectangle : public Shape {
    double width, height;
public:
    Rectangle(double w, double h) : width(w), height(h) {}
    void draw() const override {
        cout << "绘制矩形，宽=" << width
             << " 高=" << height
             << " 面积=" << (width * height) << endl;
    }
    unique_ptr<Shape> clone() const override {
        return make_unique<Rectangle>(width, height);
    }
};

// 具体产品：Triangle
class Triangle : public Shape {
    double base, height;
public:
    Triangle(double b, double h) : base(b), height(h) {}
    void draw() const override {
        cout << "绘制三角形，底=" << base
             << " 高=" << height
             << " 面积=" << (0.5 * base * height) << endl;
    }
    unique_ptr<Shape> clone() const override {
        return make_unique<Triangle>(base, height);
    }
};

// 简单工厂类
class ShapeFactory {
public:
    enum ShapeType { SHAPE_CIRCLE, SHAPE_RECTANGLE, SHAPE_TRIANGLE };

    // 工厂方法：根据类型创建对象
    static unique_ptr<Shape> createShape(ShapeType type) {
        switch (type) {
            case SHAPE_CIRCLE:
                return make_unique<Circle>(5.0);
            case SHAPE_RECTANGLE:
                return make_unique<Rectangle>(4.0, 3.0);
            case SHAPE_TRIANGLE:
                return make_unique<Triangle>(6.0, 4.0);
            default:
                return nullptr;
        }
    }

    // 按名称创建
    static unique_ptr<Shape> createByName(const string& name) {
        if (name == "circle")    return make_unique<Circle>(5.0);
        if (name == "rectangle") return make_unique<Rectangle>(4.0, 3.0);
        if (name == "triangle")  return make_unique<Triangle>(6.0, 4.0);
        return nullptr;
    }
};

int main() {
    vector<unique_ptr<Shape>> shapes;

    // 通过工厂创建各种形状
    shapes.push_back(ShapeFactory::createShape(ShapeFactory::SHAPE_CIRCLE));
    shapes.push_back(ShapeFactory::createShape(ShapeFactory::SHAPE_RECTANGLE));
    shapes.push_back(ShapeFactory::createShape(ShapeFactory::SHAPE_TRIANGLE));
    shapes.push_back(ShapeFactory::createByName("circle"));

    // 多态调用
    for (const auto& s : shapes) {
        s->draw();
    }

    return 0;
}
