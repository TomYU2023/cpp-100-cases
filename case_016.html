案例 16：异常处理实战

实现一个除法函数并处理除零异常，练习 try-catch 和栈展开机制。

核心知识点：try-catch、throw、栈展开、noexcept

完整代码（含逐行注释）

// 案例 16：异常处理实战
// 练习 try-catch、栈展开和资源安全
#include <iostream>
#include <stdexcept>
#include <string>
using namespace std;

// 自定义异常类
class DivByZeroException : public runtime_error {
public:
    DivByZeroException() : runtime_error("除数不能为零！") {}
};

// 安全的除法函数
double safeDivide(double a, double b) {
    if (b == 0) {
        throw DivByZeroException(); // 抛出自定义异常
    }
    return a / b;
}

// 资源管理类：演示栈展开时自动调用析构
class Resource {
    string name;
public:
    Resource(const string& n) : name(n) {
        cout << "Resource [" << name << "] 已获取" << endl;
    }
    ~Resource() {
        cout << "Resource [" << name << "] 已释放" << endl;
    }
};

int main() {
    double x = 10, y = 0;

    try {
        Resource res("文件句柄"); // 栈对象，异常时会自动析构
        cout << "开始计算 " << x << " / " << y << " ..." << endl;
        double result = safeDivide(x, y);
        cout << "结果 = " << result << endl;
    }
    catch (const DivByZeroException& e) {
        cerr << "自定义异常: " << e.what() << endl;
    }
    catch (const exception& e) {
        cerr << "标准异常: " << e.what() << endl;
    }
    catch (...) {
        cerr << "未知异常！" << endl;
    }

    cout << "程序继续执行..." << endl;
    return 0;
}

开始计算 10 / 0 ...
Resource [文件句柄] 已释放
程序继续执行...

[stderr]
