案例 42：彩色终端输出

使用 ANSI 转义序列实现彩色终端输出，封装成易用的工具类。

核心知识点：ANSI 转义码、命名空间封装、枚举类

完整代码（含逐行注释）

// 案例 42：彩色终端输出
// 使用 ANSI 转义序列实现终端彩色文字
#include <iostream>
#include <string>
using namespace std;

namespace Color {
    // ANSI 转义码
    const string RESET   = "\033[0m";
    const string BOLD    = "\033[1m";
    const string DIM     = "\033[2m";
    const string UNDERLINE = "\033[4m";

    // 前景色
    namespace FG {
        const string BLACK   = "\033[30m";
        const string RED     = "\033[31m";
        const string GREEN   = "\033[32m";
        const string YELLOW  = "\033[33m";
        const string BLUE    = "\033[34m";
        const string MAGENTA = "\033[35m";
        const string CYAN    = "\033[36m";
        const string WHITE   = "\033[37m";

        // 亮色版本
        const string BRIGHT_BLACK   = "\033[90m";
        const string BRIGHT_RED     = "\033[91m";
        const string BRIGHT_GREEN   = "\033[92m";
        const string BRIGHT_YELLOW  = "\033[93m";
        const string BRIGHT_BLUE    = "\033[94m";
        const string BRIGHT_MAGENTA = "\033[95m";
        const string BRIGHT_CYAN    = "\033[96m";
        const string BRIGHT_WHITE   = "\033[97m";
    }

    // 背景色
    namespace BG {
        const string RED     = "\033[41m";
        const string GREEN   = "\033[42m";
        const string YELLOW  = "\033[43m";
        const string BLUE    = "\033[44m";
        const string MAGENTA = "\033[45m";
        const string CYAN    = "\033[46m";
        const string WHITE   = "\033[47m";
    }
}

// 带颜色的输出函数
void printColored(const string& text, const string& color,
                   const string& style = "") {
    cout << style << color << text << Color::RESET;
}

// 日志函数
void logInfo(const string& msg) {
    printColored("[INFO]", Color::FG::GREEN, Color::BOLD);
    cout << " " << msg << endl;
}

void logWarn(const string& msg) {
    printColored("[WARN]", Color::FG::YELLOW, Color::BOLD);
    cout << " " << msg << endl;
}

void logError(const string& msg) {
    printColored("[ERROR]", Color::FG::RED, Color::BOLD);
    cout << " " << msg << endl;
}

void logDebug(const string& msg) {
    printColored("[DEBUG]", Color::FG::CYAN);
    cout << " " << msg << endl;
}

int main() {
    cout << "=== ANSI 彩色终端输出 ===" << endl << endl;

    // 前景色演示
    cout << "基本颜色: ";
    printColored("红色", Color::FG::RED);
    cout << " ";
    printColored("绿色", Color::FG::GREEN);
    cout << " ";
    printColored("蓝色", Color::FG::BLUE);
    cout << " ";
    printColored("黄色", Color::FG::YELLOW);
    cout << endl;

    // 样式组合
    printColored("粗体蓝色", Color::FG::BLUE, Color::BOLD);
    cout << " ";
    printColored("下划线青色", Color::FG::CYAN, Color::UNDERLINE);
    cout << endl << endl;

    // 日志输出演示
    logInfo("服务启动成功");
    logWarn("磁盘空间不足");
    logError("连接超时");
    logDebug("变量 x = 42");

    return 0;
}

语法规则：ANSI 转义序列

\033[31m 红色文字 \033[0m
// \033 是 ESC 字符（ASCII 27）
// [ 是 CSI (Control Sequence Introducer)
// m 表示设置图形属性

举一反三：ANSI 码在大多数现代终端模拟器上有效，Windows 10+ 通过启用虚拟终端序列（ENABLE_VIRTUAL_TERMINAL_PROCESSING）支持。颜色范围：30-37 前景，40-47 背景。

语法规则：命名空间组织常量

namespace Color {
    namespace FG {
        const string RED = "\033[31m";
    }
}
// 使用: Color::FG::RED

举一反三：嵌套命名空间组织相关常量，避免命名冲突。C++17 可用 namespace Color::FG { } 简化语法。enum class 也可用于颜色枚举，但需映射到字符串。

语法规则：输出后务必 RESET

cout << RED << "text" << RESET;
// 忘记 RESET 会导致后续输出保持红色

举一反三：RAII 模式封装（构造时输出颜色码，析构时输出 RESET）可避免忘记重置。或包装成函数内部自动重置。
