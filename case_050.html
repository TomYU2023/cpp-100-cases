案例 50：观察者模式

实现经典的观察者（发布-订阅）模式，理解松耦合的事件通知机制。

核心知识点：观察者模式、回调函数、function

完整代码（含逐行注释）

// 案例 50：观察者模式
// 使用 function 实现灵活的事件通知
#include <iostream>
#include <functional>
#include <vector>
#include <string>
#include <algorithm>
using namespace std;

// 事件携带的数据
struct StockData {
    string symbol;  // 股票代码
    double price;   // 最新价格
    double change;  // 涨跌百分比
};

// 被观察者：股票行情源
class StockMarket {
private:
    // 回调函数类型
    using Observer = function<void(const StockData&)>;
    vector<Observer> observers;

public:
    // 注册观察者
    void subscribe(Observer observer) {
        observers.push_back(observer);
    }

    // 取消注册
    void unsubscribe(const Observer* observer) {
        // 简化：不支持根据函数对象取消（实际可用 token 模式）
    }

    // 通知所有观察者
    void notifyAll(const StockData& data) {
        for (auto& obs : observers) {
            obs(data);
        }
    }

    // 模拟行情更新
    void updatePrice(const string& symbol, double price, double change) {
        StockData data{symbol, price, change};
        cout << "[行情] " << symbol << " 最新价=" << price
             << " 涨跌=" << change << "%" << endl;
        notifyAll(data);
    }
};

// 观察者1：控制台显示
class ConsoleDisplay {
    string name;
public:
    explicit ConsoleDisplay(const string& n) : name(n) {}
    void onPriceUpdate(const StockData& data) {
        cout << "  [" << name << "] 收到 "
             << data.symbol << " 价格更新: "
             << data.price << endl;
    }
};

// 观察者2：提醒服务
class AlertService {
    double threshold;
public:
    explicit AlertService(double t) : threshold(t) {}
    void onPriceUpdate(const StockData& data) {
        if (data.change >= threshold) {
            cout << "  [提醒] " << data.symbol
                 << " 大幅上涨 " << data.change << "%!" << endl;
        } else if (data.change <= -threshold) {
            cout << "  [提醒] " << data.symbol
                 << " 大幅下跌 " << data.change << "%!" << endl;
        }
    }
};

int main() {
    StockMarket market;

    // 创建观察者对象
    ConsoleDisplay display("显示屏1");
    AlertService alert(5.0); // 涨跌超 5% 触发提醒

    // 使用 bind 将成员函数绑定为回调
    market.subscribe(bind(&ConsoleDisplay::onPriceUpdate, &display,
                           placeholders::_1));
    market.subscribe(bind(&AlertService::onPriceUpdate, &alert,
                           placeholders::_1));

    // 使用 Lambda 作为观察者
    market.subscribe([](const StockData& d) {
        cout << "  [日志] 记录 " << d.symbol << " 价格变化" << endl;
    });

    // 模拟行情数据
    market.updatePrice("AAPL", 175.50, 2.3);
    market.updatePrice("GOOG", 140.20, -6.1);
    market.updatePrice("MSFT", 330.00, 1.5);

    return 0;
}

语法规则：function 作为回调

#include <functional>
using Observer = function<void(const StockData&)>;
vector<Observer> observers;
// function 可存储 Lambda、函数指针、bind 结果等

举一反三：function 是通用可调用对象包装器。类型擦除机制使其可存储任何满足签名的可调用对象。有轻微虚函数开销，性能敏感场景用 template。

语法规则：bind 绑定成员函数

using namespace placeholders;
bind(&Class::method, &obj, _1);
// _1 是占位符，表示回调时的第一个参数
// 成员函数需要对象指针作为第一个隐式参数

举一反三：bind 将成员函数和对象绑定生成可调用对象。C++14 后 Lambda 更直观：[&obj](auto x){ obj.method(x); }。_1、_2 等占位符对应回调参数位置。

语法规则：观察者模式要点

// 主题(subject)维护观察者列表
// 状态变化 -> 遍历列表 -> 调用回调
// 核心：松耦合，主题不需要知道观察者是谁

举一反三：观察者模式解耦了事件产生方和处理方。实际应用需注意：回调中的异常处理、避免在回调中修改观察者列表（迭代器失效）、观察者生命周期管理。
