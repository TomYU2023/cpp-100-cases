案例 99：高并发聊天服务器

用 epoll 边缘触发 + 线程池 + 非阻塞 IO 实现支持数万并发连接的实时聊天服务器，理解 C10K 问题解决方案。

核心知识点：epoll(EPOLLET)、非阻塞IO(fcntl)、线程池(条件变量)、SO_REUSEADDR、广播消息

完整代码（含逐行注释）

// 案例 99：高并发聊天服务器（epoll + 线程池）
// 使用 Linux epoll 多路复用 + 线程池，支持数万并发连接。
// 这是 C10K/C10M 级别性能的入门实践。
// 需在 Linux 编译：g++ -std=c++17 -pthread chat_server.cpp -o chat_server

#include <iostream>
#include <string>
#include <vector>
#include <thread>
#include <mutex>
#include <queue>
#include <functional>
#include <atomic>
#include <condition_variable>
#include <cstring>
#include <algorithm>

#include <sys/epoll.h>
#include <sys/socket.h>
#include <netinet/in.h>
#include <arpa/inet.h>
#include <unistd.h>
#include <fcntl.h>

const int MAX_EVENTS = 10000;
const int PORT = 8888;
const int THREAD_COUNT = 4;

// --- 设置非阻塞 ---
int setNonblocking(int fd) {
    int flags = fcntl(fd, F_GETFL, 0);
    return fcntl(fd, F_SETFL, flags | O_NONBLOCK);
}

// --- 线程池 ---
class ThreadPool {
    std::vector<std::thread> workers;
    std::queue<std::function<void()>> tasks;
    std::mutex mtx;
    std::condition_variable cv;
    std::atomic<bool> stop{false};

public:
    ThreadPool(int n) {
        for (int i = 0; i < n; i++)
            workers.emplace_back([this] {
                while (true) {
                    std::function<void()> task;
                    {
                        std::unique_lock<std::mutex> lock(mtx);
                        cv.wait(lock, [this] { return stop || !tasks.empty(); });
                        if (stop && tasks.empty()) return;
                        task = std::move(tasks.front());
                        tasks.pop();
                    }
                    task();
                }
            });
    }

    void submit(std::function<void()> f) {
        {
            std::lock_guard<std::mutex> lock(mtx);
            tasks.push(std::move(f));
        }
        cv.notify_one();
    }

    ~ThreadPool() {
        stop = true;
        cv.notify_all();
        for (auto& t : workers) t.join();
    }
};

// --- 客户端集合（线程安全）---
std::vector<int> clients;
std::mutex clientsMutex;

void broadcast(const std::string& msg, int senderFd) {
    std::lock_guard<std::mutex> lock(clientsMutex);
    for (int fd : clients) {
        if (fd != senderFd)
            send(fd, msg.c_str(), msg.size(), 0);
    }
}

// --- 处理客户端消息 ---
void handleMessage(int fd) {
    char buf[4096];
    int n = recv(fd, buf, sizeof(buf) - 1, 0);

    if (n <= 0) {
        // 客户端断开
        std::cout << "[断开] fd=" << fd << std::endl;
        close(fd);
        {
            std::lock_guard<std::mutex> lock(clientsMutex);
            auto it = std::find(clients.begin(), clients.end(), fd);
            if (it != clients.end()) clients.erase(it);
        }
        return;
    }

    buf[n] = '\0';
    std::string msg = "[客户端" + std::to_string(fd) + "] " + std::string(buf);
    std::cout << "[消息] " << msg;
    broadcast(msg, fd);
}

int main() {
    // --- 创建 socket ---
    int listenFd = socket(AF_INET, SOCK_STREAM, 0);
    if (listenFd < 0) { perror("socket"); return 1; }

    // SO_REUSEADDR 允许快速重启服务器
    int opt = 1;
    setsockopt(listenFd, SOL_SOCKET, SO_REUSEADDR, &opt, sizeof(opt));

    struct sockaddr_in addr{};
    addr.sin_family = AF_INET;
    addr.sin_addr.s_addr = INADDR_ANY;
    addr.sin_port = htons(PORT);

    bind(listenFd, (struct sockaddr*)&addr, sizeof(addr));
    listen(listenFd, SOMAXCONN);
    setNonblocking(listenFd);

    std::cout << "=== 高并发聊天服务器 ===\n";
    std::cout << "监听端口: " << PORT << " | 工作线程: " << THREAD_COUNT << "\n";

    // --- 创建 epoll ---
    int epollFd = epoll_create1(0);
    struct epoll_event ev, events[MAX_EVENTS];
    ev.events = EPOLLIN | EPOLLET;  // 边缘触发
    ev.data.fd = listenFd;
    epoll_ctl(epollFd, EPOLL_CTL_ADD, listenFd, &ev);

    ThreadPool pool(THREAD_COUNT);

    // --- 事件循环 ---
    while (true) {
        int n = epoll_wait(epollFd, events, MAX_EVENTS, -1);

        for (int i = 0; i < n; i++) {
            int fd = events[i].data.fd;

            if (fd == listenFd) {
                // 新连接
                while (true) {
                    struct sockaddr_in clientAddr{};
                    socklen_t len = sizeof(clientAddr);
                    int clientFd = accept(listenFd,
                        (struct sockaddr*)&clientAddr, &len);
                    if (clientFd < 0) break;

                    setNonblocking(clientFd);
                    ev.events = EPOLLIN | EPOLLET;
                    ev.data.fd = clientFd;
                    epoll_ctl(epollFd, EPOLL_CTL_ADD, clientFd, &ev);

                    {
                        std::lock_guard<std::mutex> lock(clientsMutex);
                        clients.push_back(clientFd);
                    }
                    std::cout << "[连接] fd=" << clientFd
                              << " 来自 " << inet_ntoa(clientAddr.sin_addr)
                              << " (当前 " << clients.size() << " 人)" << std::endl;
                }
            } else {
                // 客户端消息 -> 提交到线程池
                pool.submit([fd] { handleMessage(fd); });
            }
        }
    }

    close(epollFd);
    close(listenFd);
    return 0;
}

语法规则：epoll 多路复用（边缘触发 EPOLLET）

int epollFd = epoll_create1(0);
struct epoll_event ev, events[MAX_EVENTS];
ev.events = EPOLLIN | EPOLLET;  // 边缘触发：只在状态变化时通知
ev.data.fd = fd;
epoll_ctl(epollFd, EPOLL_CTL_ADD, fd, &ev);

while (true) {
    int n = epoll_wait(epollFd, events, MAX_EVENTS, -1);
    for (int i = 0; i < n; i++) {
        int fd = events[i].data.fd;
        // 处理 ready 的 fd
    }
}

举一反三：epoll 是 Linux 上 O(1) 复杂度的事件通知机制。EPOLLET（边缘触发）比 EPOLLLT（水平触发）性能更高，但要求 fd 设为非阻塞，且每次读到 EAGAIN 才算读空。select/poll 是 O(n) 遍历，连接上万时性能急剧下降。

语法规则：非阻塞 IO + fcntl

#include <fcntl.h>
int flags = fcntl(fd, F_GETFL, 0);
fcntl(fd, F_SETFL, flags | O_NONBLOCK);

// 现在 recv() 不会阻塞了：无数据立即返回 -1 + EAGAIN
int n = recv(fd, buf, size, 0);
if (n < 0 && errno == EAGAIN) { /* 暂时没数据，稍后再试 */ }

举一反三：非阻塞 IO + epoll 是高性能网络服务器的标准组合。没有非阻塞的话，一个慢客户端就能阻塞整个事件循环。边缘触发模式下必须用非阻塞，否则可能丢事件。

语法规则：ThreadPool 生产者-消费者模型

class ThreadPool {
    std::queue<std::function<void()>> tasks;  // 共享任务队列
    std::mutex mtx;
    std::condition_variable cv;
    
    void submit(std::function<void()> f) {
        { std::lock_guard lock(mtx); tasks.push(f); }
        cv.notify_one();  // 唤醒一个等待线程
    }
};

举一反三：这个线程池是「生产者-消费者」模型。epoll 线程生产任务入队，工作线程消费出队执行。进阶可以用无锁队列（moodycamel::ConcurrentQueue）减少锁竞争，或用 work-stealing（每个线程有自己的队列，空闲时偷别人的）。
