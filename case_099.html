案例 99：type_traits 类型萃取实战

综合运用 type_traits 实现编译期类型检查、转换和信息查询。

核心知识点：type_traits、类型转换、类型检查、组合萃取

完整代码（含逐行注释）

// 案例 99：type_traits 类型萃取实战
// 编译期类型检查和转换的综合应用
#include <iostream>
#include <type_traits>
#include <string>
#include <vector>
using namespace std;

// 1. 编译期确保类型符合要求
template <typename T>
class NumericArray {
    static_assert(is_arithmetic<T>::value,
                  "NumericArray 只接受算术类型");
    vector<T> data;
public:
    void push(const T& val) { data.push_back(val); }
    size_t size() const { return data.size(); }
};

// 2. 使用类型转换萃取
template <typename T>
auto getDecayedValue(T&& val) {
    using RawType = decay_t<T>;           // 去除引用和 CV 限定
    using SignedType = make_signed_t<RawType>; // 转为有符号
    cout << "原始类型: " << typeid(T).name() << endl;
    cout << "退化类型: " << typeid(RawType).name() << endl;
    cout << "有符号类型: " << typeid(SignedType).name() << endl;
}

// 3. 条件类型选择
template <typename T>
struct StorageType {
    // 小类型直接存值，大类型用指针
    using type = conditional_t<
        (sizeof(T) <= sizeof(void*)),
        T, T*>;
};

int main() {
    // 1. 类型检查
    cout << "=== 类型检查 ===" << endl;
    cout << "int 是算术类型: " << is_arithmetic<int>::value << endl;
    cout << "string 是算术类型: "
         << is_arithmetic<string>::value << endl;
    cout << "int 是整型: " << is_integral<int>::value << endl;

    // 2. 静态断言
    NumericArray<int> intArr;    // OK
    intArr.push(42);
    // NumericArray<string> strArr; // 编译错误！

    // 3. 类型转换萃取
    cout << "\n=== 类型转换 ===" << endl;
    getDecayedValue(42);

    // 4. 条件类型选择
    StorageType<int>::type smallVal = 42;       // 直接存储
    StorageType<array<int, 100>>::type largeVal; // 指针存储
    cout << "\nStorageType<int>: sizeof = "
         << sizeof(smallVal) << " (直接存储)" << endl;
    cout << "StorageType<array<int,100>>: sizeof = "
         << sizeof(largeVal) << " (指针)" << endl;

    return 0;
}
