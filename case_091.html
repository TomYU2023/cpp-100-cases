案例 91：模板特化与偏特化

理解函数模板完全特化、类模板全特化与偏特化的规则和应用场景。

核心知识点：全特化、偏特化、优先级、SFINAE 入门

完整代码（含逐行注释）

// 案例 91：模板特化与偏特化
// 为特定类型或类型族提供定制实现
#include <iostream>
#include <string>
#include <type_traits>
using namespace std;

// 1. 主模板
template <typename T>
struct TypeInfo {
    static const char* name() { return "未知类型"; }
};

// 2. 全特化（针对特定类型）
template <>
struct TypeInfo<int> {
    static const char* name() { return "整数 (int)"; }
};

template <>
struct TypeInfo<double> {
    static const char* name() { return "双精度浮点 (double)"; }
};

template <>
struct TypeInfo<string> {
    static const char* name() { return "字符串 (string)"; }
};

// 3. 偏特化：针对所有指针类型
template <typename T>
struct TypeInfo<T*> {
    static string name() {
        return string("指向 ") + TypeInfo<T>::name() + " 的指针";
    }
};

// 4. 偏特化：针对所有 const 类型
template <typename T>
struct TypeInfo<const T> {
    static string name() {
        return string("const ") + TypeInfo<T>::name();
    }
};

int main() {
    cout << "int: " << TypeInfo<int>::name() << endl;
    cout << "double: " << TypeInfo<double>::name() << endl;
    cout << "char: " << TypeInfo<char>::name() << endl;
    cout << "string: " << TypeInfo<string>::name() << endl;
    cout << "int*: " << TypeInfo<int*>::name() << endl;
    cout << "const double: " << TypeInfo<const double>::name() << endl;
    cout << "const int*: " << TypeInfo<const int*>::name() << endl;
    return 0;
}

运行结果

int: 整数 (int)
double: 双精度浮点 (double)
char: 未知类型
string: 字符串 (string)
int*: 指向 整数 (int) 的指针
const double: const 双精度浮点 (double)
const int*: 指向 const 整数 (int) 的指针
