案例 91：有栈协程库

基于 ucontext 实现用户态协程调度器，包含创建、yield、调度循环，理解 swapcontext 上下文切换的底层原理。

核心知识点：ucontext/swapcontext 上下文切换、有栈协程、std::function 类型擦除、用户态调度

完整代码（含逐行注释）

// 案例 91：有栈协程库
// 使用 ucontext 实现用户态协程调度，理解协程的上下文切换机制。
// 协程 = 可以暂停和恢复的函数，比线程更轻量（切换开销 10ns 级别）。

#include <iostream>
#include <vector>
#include <queue>
#include <functional>
#include <ucontext.h>
#include <cstring>

const int STACK_SIZE = 1024 * 1024;  // 每个协程 1MB 栈
const int MAX_COROUTINES = 100;

// ─── 协程状态 ────────────────────────────────────
enum class CoState { READY, RUNNING, SUSPENDED, DEAD };

struct Coroutine {
    int id;
    CoState state;
    ucontext_t ctx;            // 上下文（寄存器、栈指针等）
    std::function<void()> func; // 协程体
    char* stack;               // 独立栈空间

    Coroutine() : id(0), state(CoState::READY), stack(nullptr) {}
};

// ─── 调度器 ──────────────────────────────────────
std::vector<Coroutine*> coroutines;
std::queue<int> readyQueue;   // 就绪队列
int currentCo = -1;
ucontext_t mainCtx;           // 主协程的上下文

// ─── 协程入口包装 ────────────────────────────────
// 每个协程启动后先执行 func，然后标记为 DEAD 并让出
void coEntry(Coroutine* co) {
    co->func();          // 执行协程体
    co->state = CoState::DEAD;
    // 协程体执行完毕后切换回调度器的上下文（主协程）
    setcontext(&mainCtx);
}

// ─── 创建协程 ────────────────────────────────────
int createCoroutine(std::function<void()> f) {
    auto* co = new Coroutine();
    co->id = coroutines.size();
    co->func = std::move(f);  // 移动而非拷贝，避免不必要的函数对象复制
    co->stack = new char[STACK_SIZE];

    // 初始化 ucontext
    getcontext(&co->ctx);
    co->ctx.uc_link = nullptr;          // 无后续上下文（手动调度）
    co->ctx.uc_stack.ss_sp = co->stack; // 栈的起始地址
    co->ctx.uc_stack.ss_size = STACK_SIZE;
    co->ctx.uc_stack.ss_flags = 0;

    // 创建上下文：指定入口函数和参数
    makecontext(&co->ctx, (void(*)())coEntry, 1, co);

    coroutines.push_back(co);
    readyQueue.push(co->id);
    return co->id;
}

// ─── 让出 CPU（yield）───────────────────────────
void coYield() {
    int prev = currentCo;
    Coroutine* co = coroutines[prev];
    co->state = CoState::SUSPENDED;
    readyQueue.push(prev);   // 重新排到就绪队列末尾

    // 调度下一个就绪协程
    if (!readyQueue.empty()) {
        int next = readyQueue.front();
        readyQueue.pop();
        currentCo = next;
        coroutines[next]->state = CoState::RUNNING;
        // 保存当前上下文（prev），恢复 next 的上下文
        swapcontext(&coroutines[prev]->ctx, &coroutines[next]->ctx);
    }
}

// ─── 调度循环（单线程）───────────────────────────
void schedule() {
    while (!readyQueue.empty()) {
        int id = readyQueue.front();
        readyQueue.pop();
        currentCo = id;
        coroutines[id]->state = CoState::RUNNING;

        // 从 mainCtx 切换到协程 — 核心魔法在这一行
        swapcontext(&mainCtx, &coroutines[id]->ctx);

        // 协程 yield 或结束后回到这里
        if (coroutines[id]->state == CoState::DEAD) {
            delete[] coroutines[id]->stack;
            delete coroutines[id];
            coroutines[id] = nullptr;
        }
    }
}

// ─── 示例：3 个协程交替执行 ──────────────────────
int main() {
    // 协程 1：输出 3 次 A 并 yield
    createCoroutine([]() {
        for (int i = 0; i < 3; i++) {
            std::cout << "[协程1] A(" << i << ")" << std::endl;
            coYield();
        }
    });

    // 协程 2：输出 3 次 B 并 yield
    createCoroutine([]() {
        for (int i = 0; i < 3; i++) {
            std::cout << "[协程2] B(" << i << ")" << std::endl;
            coYield();
        }
    });

    // 协程 3：不 yield，一次性跑完
    createCoroutine([]() {
        std::cout << "[协程3] 一次跑完！" << std::endl;
    });

    std::cout << "==== 开始调度 ====" << std::endl;
    schedule();
    std::cout << "==== 调度结束 ====" << std::endl;

    return 0;
}

语法规则：ucontext 上下文切换

#include <ucontext.h>
ucontext_t ctx;
getcontext(&ctx);              // 保存当前上下文
ctx.uc_stack.ss_sp = stack_ptr;  // 设置栈
ctx.uc_stack.ss_size = SIZE;
makecontext(&ctx, func_ptr, argc, arg1, arg2, ...);

// 切换上下文
swapcontext(&from, &to);  // 保存 from，恢复 to
setcontext(&to);          // 仅恢复，不保存

💡 举一反三：ucontext 是 POSIX 标准的用户态上下文切换 API（已被标记为废弃但仍广泛可用）。swapcontext 保存当前寄存器到 from，从 to 恢复寄存器——这就是协程切换的本质。现代方案用 boost.context 或汇编手写。

语法规则：std::function 和 lambda 回调

#include <functional>
std::function<void()> f;
f = []() { std::cout << "Hello" << std::endl; };
f();  // 调用

// 传参时用 std::move 避免拷贝
void setTask(std::function<void()> task) {
    m_task = std::move(task);  // 移动语义，浅拷贝控制块
}

💡 举一反三：std::function 是一个类型擦除的通用回调容器，可以存储 lambda、函数指针、仿函数等。缺点是每次调用的虚函数开销（10~20ns），性能敏感场景用模板直接存具体类型。

语法规则：new/delete 数组形式

char* buf = new char[1024];    // 分配 1024 字节
buf[0] = 'a';
delete[] buf;                 // 必须用 delete[]，不能用 delete

💡 举一反三：new[] 返回的是数组指针，delete[] 会调用数组中每个元素的析构函数（需要从分配器读取元素数量）。用错会导致未定义行为（内存泄漏或崩溃）。协程的独立栈用 char 数组表示，无需析构，但语法上仍要用 delete[]。
