案例 95：std::bind 参数绑定

使用 bind 进行参数绑定和适配，理解占位符的使用。

核心知识点：bind、占位符、参数重排、ref/cref

完整代码（含逐行注释）

// 案例 95：std::bind 参数绑定
// 预绑定参数和重排参数顺序
#include <iostream>
#include <functional>
#include <vector>
#include <algorithm>
#include <string>
using namespace std;
using namespace placeholders;

// 一个接受多个参数的函数
void logMessage(const string& level, int code, const string& msg) {
    cout << "[" << level << "] " << code << ": " << msg << endl;
}

int add(int a, int b, int c) {
    return a + b + c;
}

bool isBetween(int value, int low, int high) {
    return value >= low && value <= high;
}

int main() {
    // 1. 绑定部分参数
    auto errorLog = bind(logMessage, "ERROR", _1, _2);
    auto warnLog  = bind(logMessage, "WARN",  _1, _2);

    errorLog(404, "页面未找到");
    warnLog(1001, "磁盘空间不足");

    // 2. 绑定全部参数
    auto hello = bind(logMessage, "INFO", 0, "系统启动完成");
    hello(); // 所有参数已绑定，无需传入

    // 3. 占位符重排参数顺序
    // add(a, b, c): 用 _2, _1, _3 交换前两个参数
    auto addSwapped = bind(add, _2, _1, _3);
    cout << "\naddSwapped(10, 20, 30) = "
         << addSwapped(10, 20, 30) << endl; // 20+10+30=60

    // 4. 固定某个参数的值
    auto addFive = bind(add, _1, 5, _2);
    cout << "addFive(10, 3) = " << addFive(10, 3) << endl; // 10+5+3=18

    // 5. 在算法中使用 bind
    vector<int> data = {1, 5, 10, 15, 20, 25, 30};

    // 统计在 [10, 20] 之间的元素
    auto inRange = bind(isBetween, _1, 10, 20);
    int count = count_if(data.begin(), data.end(), inRange);
    cout << "\n[10,20] 范围内的元素数: " << count << endl;

    // 6. ref / cref：绑定引用
    int counter = 0;
    auto increment = [](int& c) { c++; };

    // 对 bind，需要用 ref 传递引用
    auto incCounter = bind(increment, ref(counter));
    incCounter();
    incCounter();
    cout << "counter = " << counter << endl; // 2

    return 0;
}

[WARN] 1001: 磁盘空间不足
[INFO] 0: 系统启动完成

addFive(10, 3) = 18
