案例 82：简易脚本引擎（表达式求值）

用递归下降法手写一个数学表达式解析求值器，支持四则运算、括号、幂运算、三角函数和变量。

核心知识点：递归下降解析器、语法分析、std::stod、map 变量表、浮点比较

完整代码（含逐行注释）

#include <iostream>
#include <string>
#include <cctype>
#include <stdexcept>
#include <map>

// ========== 简易数学表达式求值引擎 ==========
// 支持：数字、+、-、*、/、()、变量（单字母）、sin/cos/sqrt
// 实现方式：递归下降解析器（Recursive Descent Parser）

class ExprEngine {
    const std::string& expr_; // 表达式字符串引用
    size_t pos_;              // 当前解析位置
    std::map<std::string, double> vars_; // 变量表

    // 跳过空格
    void skipSpace() {
        while (pos_ < expr_.size() && std::isspace(expr_[pos_])) pos_++;
        // std::isspace 判断空白字符（空格、制表符、换行等），定义在 <cctype>
    }

    // 查看当前字符（不移动位置）
    char peek() const {
        return pos_ < expr_.size() ? expr_[pos_] : '\0';
    }

    // 消费当前字符并返回
    char consume() {
        return pos_ < expr_.size() ? expr_[pos_++] : '\0';
    }

    // 匹配并消费指定字符
    bool match(char c) {
        skipSpace();
        if (peek() == c) { pos_++; return true; }
        return false;
    }

    // ===== 递归下降核心：每个非终结符一个函数 =====

    // expression = term { ('+'|'-') term }
    // 语法规则：递归下降解析器 — 每个语法规则对应一个函数
    // expression → term (('+'|'-') term)*
    // 运算符优先级通过函数调用层级实现：expression 调用 term，term 调用 factor
    double expression() {
        skipSpace();
        double result = term();  // 先解析第一个项
        while (true) {
            skipSpace();
            if (match('+')) result += term();       // 加法
            else if (match('-')) result -= term();   // 减法
            else break;
        }
        return result;
    }

    // term = factor { ('*'|'/') factor }
    double term() {
        skipSpace();
        double result = factor(); // 先解析第一个因子
        while (true) {
            skipSpace();
            if (match('*')) result *= factor();      // 乘法
            else if (match('/')) {                   // 除法
                double divisor = factor();
                if (divisor == 0) throw std::runtime_error("除零错误");
                result /= divisor;
            }
            else break;
        }
        return result;
    }

    // factor = [ '+' | '-' ] power
    double factor() {
        skipSpace();
        if (match('+')) return power();   // 正号
        if (match('-')) return -power();  // 负号
        return power();
    }

    // power = atom [ '^' factor ]  (右结合)
    double power() {
        double base = atom();
        skipSpace();
        if (match('^')) {
            double exp = power(); // 递归调用实现右结合
            // 语法规则：std::pow(base, exp) — 幂运算，定义在 <cmath>
            return std::pow(base, exp);
        }
        return base;
    }

    // atom = number | '(' expression ')' | function '(' expression ')' | variable
    double atom() {
        skipSpace();
        // 括号表达式
        if (match('(')) {
            double val = expression();
            if (!match(')')) throw std::runtime_error("缺少右括号");
            return val;
        }

        // 数字
        // 语法规则：std::isdigit(c) — 判断字符是否为数字 '0'-'9'
        if (std::isdigit(peek()) || peek() == '.') {
            size_t start = pos_;
            while (pos_ < expr_.size() &&
                   (std::isdigit(expr_[pos_]) || expr_[pos_] == '.')) pos_++;
            // std::stod：string to double，定义在 <string>
            return std::stod(expr_.substr(start, pos_ - start));
        }

        // 函数调用：sin/cos/sqrt
        if (std::isalpha(peek())) {
            std::string name;
            while (pos_ < expr_.size() && std::isalpha(expr_[pos_]))
                name += expr_[pos_++];
            // 语法规则：std::isalpha(c) — 判断字符是否为字母 A-Z/a-z
            skipSpace();
            if (match('(')) {
                double arg = expression();
                if (!match(')')) throw std::runtime_error("函数缺少右括号");
                if (name == "sin") return std::sin(arg);
                if (name == "cos") return std::cos(arg);
                if (name == "sqrt") return std::sqrt(arg);
                throw std::runtime_error("未知函数: " + name);
            }
            // 变量
            if (vars_.count(name)) return vars_[name];
            throw std::runtime_error("未定义变量: " + name);
        }

        throw std::runtime_error("意外的字符: " + std::string(1, peek()));
    }

public:
    ExprEngine(const std::string& expr) : expr_(expr), pos_(0) {}

    void setVar(const std::string& name, double val) { vars_[name] = val; }

    double eval() {
        double result = expression();
        skipSpace();
        if (pos_ != expr_.size()) {
            throw std::runtime_error("表达式末尾有多余字符");
        }
        return result;
    }
};

int main() {
    // 测试用例
    struct Test { const char* expr; double expected; };
    Test tests[] = {
        {"1 + 2 * 3", 7},
        {"(1 + 2) * 3", 9},
        {"-3 + 5", 2},
        {"2 ^ 3 ^ 2", 512},  // 右结合：2^(3^2)=2^9=512
        {"sin(0)", 0},
        {"sqrt(16) + 1", 5},
    };

    ExprEngine eng("");
    for (auto& t : tests) {
        eng = ExprEngine(t.expr);
        double result = eng.eval();
        std::cout << t.expr << " = " << result
                  << (std::abs(result - t.expected) < 1e-9 ? " ✓" : " ✗")
                  << "\n";
    }

    // 变量演示
    ExprEngine eng2("x * 2 + y");
    eng2.setVar("x", 10);
    eng2.setVar("y", 5);
    std::cout << "x*2+y (x=10,y=5) = " << eng2.eval() << "\n";

    std::cout << "\n核心知识点：\n"
              << "1. 递归下降解析器：每个语法规则对应一个函数\n"
              << "2. 运算符优先级通过函数调用层级实现\n"
              << "3. 左递归需改写为 while 循环（如 expression 函数）\n";
    return 0;
}

语法规则：递归下降解析器（Recursive Descent Parser）

// 文法规则：
// expr   = term { ('+'|'-') term }
// term   = factor { ('*'|'/') factor }
// factor = number | '(' expr ')' | func '(' expr ')'
// 每个规则对应一个函数，层层调用

💡 举一反三：递归下降是最直观的手写解析器方式。每个非终结符一个函数，左递归需要改写为循环。比 yacc/bison 更灵活，但效率较低。适合小语言和 DSL。

语法规则：std::stod / std::stoi / std::stoul

std::string s = "3.14159";
double d = std::stod(s);    // 3.14159
int i = std::stoi("42");   // 42
// 还有 stof(float), stol(long), stoll(long long), stoull

💡 举一反三：C++11 引入的字符串转数值函数族。比 atof/atoi 更安全：支持指定转换位置（第二个参数），失败抛异常（std::invalid_argument / std::out_of_range）。

语法规则：std::abs / std::fabs / 浮点比较

double a = 0.1 + 0.2;  // 0.30000000000000004
double b = 0.3;
// 浮点比较用容差：
bool eq = std::abs(a - b) < 1e-9;
// 不要用 a == b，浮点有精度误差！

💡 举一反三：浮点数不能用 == 直接比较。标准做法是判断差的绝对值是否小于一个很小的容差值（epsilon），通常取 1e-9 或 1e-12。

语法规则：std::map::count / find / contains(C++20)

std::map<std::string, int> m = {{"x",10}};
if (m.count("x")) { /* 存在 */ }     // count 返回 0 或 1
if (m.find("x") != m.end()) { }     // find 返回迭代器
if (m.contains("x")) { }            // C++20，最清晰

💡 举一反三：map 的键是唯一的，count() 只能返回 0 或 1。C++20 新增 contains() 语义最明确。C++17 以下用 count() 或 find() != end()。
