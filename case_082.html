案例 82：std::merge 原地归并

使用 merge 和 inplace_merge 合并有序序列，理解归并思想。

核心知识点：merge、inplace_merge、归并排序基础

完整代码（含逐行注释）

// 案例 82：std::merge 原地归并
// 合并两个有序序列
#include <iostream>
#include <algorithm>
#include <vector>
#include <string>
#include <list>
using namespace std;

int main() {
    // 1. merge: 合并两个有序序列
    vector<int> A = {1, 3, 5, 7, 9};
    vector<int> B = {2, 4, 6, 8, 10};
    vector<int> result;

    merge(A.begin(), A.end(), B.begin(), B.end(),
          back_inserter(result));

    cout << "A: ";
    for (int x : A) cout << x << " ";
    cout << "\nB: ";
    for (int x : B) cout << x << " ";
    cout << "\n合并后: ";
    for (int x : result) cout << x << " ";
    cout << endl;

    // 2. 合并不同容器
    cout << "\n=== 合并不同容器 ===" << endl;
    list<int> lstA = {10, 30, 50};
    list<int> lstB = {20, 40, 60, 70};
    vector<int> merged;
    merge(lstA.begin(), lstA.end(), lstB.begin(), lstB.end(),
          back_inserter(merged));
    cout << "不同容器合并: ";
    for (int x : merged) cout << x << " ";
    cout << endl;

    // 3. inplace_merge: 在一个序列内合并两个有序段
    // 序列分为 [1,3,5,7] 和 [2,4,6,8] 两段，整体有序化
    vector<int> inplace = {1, 3, 5, 7, 2, 4, 6, 8};
    cout << "\ninplace_merge 前: ";
    for (int x : inplace) cout << x << " ";
    cout << endl;

    auto middle = inplace.begin() + 4; // 前 4 个和后 4 个分别有序
    inplace_merge(inplace.begin(), middle, inplace.end());

    cout << "inplace_merge 后: ";
    for (int x : inplace) cout << x << " ";
    cout << endl;

    // 4. 展示归并排序的底层原理
    vector<int> data = {38, 27, 43, 3, 9, 82, 10};
    cout << "\n原始: ";
    for (int x : data) cout << x << " ";
    cout << endl;

    // 模拟归并排序的一步：先对各半排序，再 inplace_merge
    // 这里简化：先 sort 各半
    sort(data.begin(), data.begin() + 4); // [3, 27, 38, 43]
    sort(data.begin() + 4, data.end());   // [9, 10, 82]
    inplace_merge(data.begin(), data.begin() + 4, data.end());
    cout << "一次归并排序后: ";
    for (int x : data) cout << x << " ";
    cout << endl;

    return 0;
}

语法规则：std::merge

merge(A.begin(), A.end(), B.begin(), B.end(), output);
// 两个有序序列合并为一个有序序列
// 稳定排序：相等元素按 A 在前，B 在后
// O(N+M) 时间，需要额外输出空间

举一反三：merge 是归并排序的核心操作。输入必须有序，输出到新容器。稳定：相等元素保持原序列中的顺序。不同容器的迭代器也可混合使用。

语法规则：inplace_merge

inplace_merge(v.begin(), middle, v.end());
// 将 v[begin,middle) 和 v[middle,end) 两段合并
// 两段各自要求有序
// 通常需要 O(N) 额外空间（缓冲区）

举一反三：inplace_merge 在单容器内合并相邻有序段。需要额外的线性空间（通常用临时缓冲区）来实现稳定合并。是归并排序最后一步的标准操作。

语法规则：归并排序 vs 快速排序

// 归并排序: 稳定，O(N log N) 保证，需要 O(N) 额外空间
// 快速排序: 不稳定，O(N log N) 平均，原地
// 外排序（大文件）必须用归并策略
// 数据量小时插入排序可能更快

举一反三：std::stable_sort 通常使用归并排序（或 TimSort）。链表排序天然适合归并（不需要额外空间）。merge 也可用于两个有序文件的合并。
