案例 14：模板函数

编写通用模板函数，实现任意类型的最大值查找、数组打印、交换等功能，理解泛型编程基础。

核心知识点：函数模板、类型推导、模板特化

完整代码（含逐行注释）

// 案例 14：模板函数
// 练习函数模板的定义和使用
#include <iostream>
#include <string>
using namespace std;

// 模板函数：返回两个值中较大的那个
template <typename T>
T myMax(T a, T b) {
    return (a > b) ? a : b;
}

// 模板函数：打印数组
template <typename T>
void printArray(const T arr[], size_t size) {
    cout << "[ ";
    for (size_t i = 0; i < size; i++) {
        cout << arr[i];
        if (i < size - 1) cout << ", ";
    }
    cout << " ]" << endl;
}

// 模板函数：交换两个值（引用传递）
template <typename T>
void mySwap(T& a, T& b) {
    T temp = a;
    a = b;
    b = temp;
}

int main() {
    // 类型自动推导
    cout << "max(10, 20) = " << myMax(10, 20) << endl;
    cout << "max(3.14, 2.72) = " << myMax(3.14, 2.72) << endl;
    cout << "max(\"abc\", \"abd\") = " << myMax(string("abc"), string("abd")) << endl;

    // 打印数组
    int intArr[] = {1, 2, 3, 4, 5};
    double dblArr[] = {1.1, 2.2, 3.3};
    printArray(intArr, 5);
    printArray(dblArr, 3);

    // 交换
    int x = 10, y = 20;
    mySwap(x, y);
    cout << "交换后: x=" << x << ", y=" << y << endl;

    return 0;
}

运行结果

max(10, 20) = 20
max(3.14, 2.72) = 3.14
max("abc", "abd") = abd
[ 1, 2, 3, 4, 5 ]
[ 1.1, 2.2, 3.3 ]
交换后: x=20, y=10
