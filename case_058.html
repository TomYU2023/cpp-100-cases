案例 58：智能指针循环引用

演示 shared_ptr 循环引用导致内存泄漏的问题及用 weak_ptr 解决的方法。

核心知识点：循环引用、weak_ptr、use_count

完整代码（含逐行注释）

// 案例 58：智能指针循环引用
// 演示并解决 shared_ptr 循环引用问题
#include <iostream>
#include <memory>
#include <string>
using namespace std;

// 有循环引用问题的节点
struct BadNode {
    string name;
    shared_ptr<BadNode> partner; // 双方互相持有 shared_ptr
    BadNode(const string& n) : name(n) {
        cout << "BadNode[" << name << "] 创建" << endl;
    }
    ~BadNode() {
        cout << "BadNode[" << name << "] 销毁" << endl;
    }
};

// 使用 weak_ptr 解决循环引用
struct GoodNode {
    string name;
    shared_ptr<GoodNode> next;   // 强引用
    weak_ptr<GoodNode> prev;     // 弱引用，不增加引用计数
    GoodNode(const string& n) : name(n) {
        cout << "GoodNode[" << name << "] 创建" << endl;
    }
    ~GoodNode() {
        cout << "GoodNode[" << name << "] 销毁" << endl;
    }
};

// 父-子双向引用场景
struct Child; // 前向声明
struct Parent {
    string name;
    shared_ptr<Child> child;
    Parent(const string& n) : name(n) {
        cout << "Parent[" << name << "] 创建" << endl;
    }
    ~Parent() { cout << "Parent[" << name << "] 销毁" << endl; }
};

struct Child {
    string name;
    weak_ptr<Parent> parent;  // 使用 weak_ptr 打破循环
    Child(const string& n) : name(n) {
        cout << "Child[" << name << "] 创建" << endl;
    }
    ~Child() { cout << "Child[" << name << "] 销毁" << endl; }
};

int main() {
    cout << "=== 1. 循环引用导致泄漏 ===" << endl;
    {
        auto a = make_shared<BadNode>("A");
        auto b = make_shared<BadNode>("B");
        a->partner = b;  // a 引用 b
        b->partner = a;  // b 引用 a → 循环！
        cout << "a use_count = " << a.use_count() << endl; // 2
        cout << "b use_count = " << b.use_count() << endl; // 2
    } // a 和 b 离开作用域，但引用计数不会归零 → 内存泄漏！

    cout << "\n=== 2. weak_ptr 解决 ===" << endl;
    {
        auto a = make_shared<GoodNode>("A");
        auto b = make_shared<GoodNode>("B");
        a->next = b;   // a 强引用 b
        b->prev = a;   // b 弱引用 a（不增加计数）
        cout << "a use_count = " << a.use_count() << endl; // 1
        cout << "b use_count = " << b.use_count() << endl; // 2
    } // 正常析构

    cout << "\n=== 3. 父子关系 ===" << endl;
    {
        auto parent = make_shared<Parent>("P");
        auto child = make_shared<Child>("C");
        parent->child = child;  // parent 强引用 child
        child->parent = parent; // child 弱引用 parent
        cout << "parent use_count = " << parent.use_count() << endl; // 1
    } // 正常析构

    return 0;
}
