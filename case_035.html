案例 35：filesystem 遍历器

使用 C++17 filesystem 库遍历目录，统计文件类型和大小，练习文件系统操作。

核心知识点：filesystem、directory_iterator、path 操作

完整代码（含逐行注释）

// 案例 35：filesystem 遍历器
// 递归遍历指定目录，统计文件信息
#include <iostream>
#include <filesystem>
#include <string>
#include <map>
using namespace std;
namespace fs = filesystem;

int main() {
    // 要遍历的目录（可修改为实际路径）
    string targetPath = ".";

    if (!fs::exists(targetPath)) {
        cout << "路径不存在: " << targetPath << endl;
        return 1;
    }

    size_t totalFiles = 0;
    size_t totalSize = 0;
    // 按扩展名统计文件数量和大小
    map<string, pair<size_t, uintmax_t>> stats;

    // 递归遍历目录
    for (const auto& entry : fs::recursive_directory_iterator(targetPath)) {
        // 跳过目录
        if (entry.is_directory()) continue;

        // 获取文件信息
        auto path = entry.path();
        string ext = path.extension().string(); // 扩展名
        if (ext.empty()) ext = "(无扩展名)";

        uintmax_t size = entry.file_size(); // 文件大小

        stats[ext].first++;    // 计数
        stats[ext].second += size;
        totalFiles++;
        totalSize += size;
    }

    // 输出统计结果
    cout << "=== 目录遍历统计 ===" << endl;
    cout << "路径: " << fs::absolute(targetPath) << endl;
    cout << "总文件数: " << totalFiles << endl;
    cout << "总大小: " << totalSize << " bytes" << endl;
    cout << endl;

    cout << "按扩展名统计:" << endl;
    for (const auto& [ext, info] : stats) {
        cout << "  " << ext << ": "
             << info.first << " 个文件, "
             << info.second << " bytes" << endl;
    }

    return 0;
}
