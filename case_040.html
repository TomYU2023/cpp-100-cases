案例 40：future+async

使用 future 和 async 实现异步任务，获取线程返回值，理解异步编程模型。

核心知识点：async、future、promise、异步编程

完整代码（含逐行注释）

// 案例 40：future + async
// 异步执行任务并获取返回值
#include <iostream>
#include <future>
#include <vector>
#include <numeric>
#include <chrono>
using namespace std;

// 模拟耗时计算
int heavyComputation(int id, int workload) {
    cout << "任务 " << id << " 开始 (workload=" << workload << ")" << endl;

    // 模拟耗时
    this_thread::sleep_for(chrono::milliseconds(workload * 100));

    int result = 0;
    for (int i = 0; i < workload * 1000000; i++) {
        result += i;
    }

    cout << "任务 " << id << " 完成" << endl;
    return result;
}

int main() {
    // 1. async 异步启动任务
    cout << "=== async 异步任务 ===" << endl;
    auto f1 = async(launch::async, heavyComputation, 1, 5);
    auto f2 = async(launch::async, heavyComputation, 2, 3);
    auto f3 = async(launch::async, heavyComputation, 3, 4);

    // 主线程可以做其他事情...

    // get() 获取结果（会阻塞直到任务完成）
    int r1 = f1.get();
    int r2 = f2.get();
    int r3 = f3.get();

    cout << "结果: " << r1 << ", " << r2 << ", " << r3 << endl;

    // 2. async 延迟执行
    cout << "\n=== 延迟执行 (deferred) ===" << endl;
    auto f4 = async(launch::deferred, heavyComputation, 4, 2);
    cout << "任务 4 还没开始执行..." << endl;
    // 只有在调用 get() 时才真正开始执行
    int r4 = f4.get();
    cout << "任务 4 结果: " << r4 << endl;

    // 3. 使用 future 等待多个并行任务
    cout << "\n=== 并行处理向量 ===" << endl;
    vector<int> data(10000000);
    iota(data.begin(), data.end(), 0);

    auto sumFuture = async(launch::async, [&data]() {
        return accumulate(data.begin(), data.begin() + data.size() / 2, 0LL);
    });
    auto sumFuture2 = async(launch::async, [&data]() {
        return accumulate(data.begin() + data.size() / 2, data.end(), 0LL);
    });

    long long total = sumFuture.get() + sumFuture2.get();
    cout << "总和: " << total << endl;

    return 0;
}
