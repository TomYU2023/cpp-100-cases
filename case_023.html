案例 023

LRU 缓存

完整代码（逐行注释）

1 #include <iostream>
2 #include <unordered_map> 哈希表 O(1) 查找
3 #include <list> 双向链表 O(1) 移动/删除
4 #include <string>
5 using namespace std;
6
7 // LRU Cache: Least Recently Used
8 // get(key) 时把该节点移到链表头，表示刚被使用
9 // put(key,val) 时若满了就删链表尾（最久未用）
10
11 class LRUCache {
12 int cap_; 容量
13 list<pair<int, string>> items_; 双向链表：头=最近使用，尾=最久未用
14 unordered_map<int, list<pair<int,string>>::iterator> map_; key -> 链表迭代器
15 public:
16 LRUCache(int cap) : cap_(cap) {}
17
18 string get(int key) {
19 auto it = map_.find(key);
20 if (it == map_.end()) return ""; 不存在
21 // 移到链表头（splice 是 O(1)）
22 items_.splice(items_.begin(), items_, it->second); 把节点移到开头
23 return it->second->second;
24 }
25
26 void put(int key, const string& val) {
27 auto it = map_.find(key);
28 if (it != map_.end()) { key 已存在
29 it->second->second = val; 更新值
30 items_.splice(items_.begin(), items_, it->second); 移到头
31 return;
32 }
33 if (items_.size() >= cap_) { 需要淘汰
34 auto last = items_.back().first; 最久未用的 key
35 map_.erase(last); 从哈希表删除
36 items_.pop_back(); 从链表删除
37 }
38 items_.emplace_front(key, val); 新节点放头部
39 map_[key] = items_.begin(); 记录迭代器
40 }
41
42 void print() { 调试输出
43 cout << "LRU: ";
44 for (auto& [k, v] : items_)
45 cout << k << ":" << v << " -> ";
46 cout << endl;
47 }
48 };
49
50 int main() {
51 LRUCache cache(3); 容量为 3
52 cache.put(1, "one"); cache.put(2, "two"); cache.put(3, "three");
53 cache.print(); 3:three -> 2:two -> 1:one
54 cache.get(1); // 把 1 移到头部 访问后变最近
55 cache.put(4, "four"); // 淘汰 2 最久未用的是 2
56 cache.print(); 4:four -> 1:one -> 3:three
57 return 0;
58 }
复制代码

新语法讲解

list::splice O(1) 节点转移

splice(pos, other, it) 把 other 链表的 it 所指节点移动到 pos 之前，时间复杂度 O(1)。关键：它移动的是节点而非拷贝元素，所以哈希表中保存的迭代器不会失效。

举一反三

list a = {1,2,3}, b = {4,5};
a.splice(a.begin(), b, b.begin());
// a: 4,1,2,3  b: 5
// b 的第一个节点被移动到了 a 的头部

unordered_map + list 组合设计

哈希表负责 O(1) 查找；双向链表负责维护顺序。哈希表的值存链表迭代器，这样链表操作后仍能通过迭代器访问元素。这是 LRU / LFU 缓存的标准实现模式。

举一反三

// 关键类型
unordered_map::iterator> map_;
// splice 后迭代器不失效，所以安全

升级为 LFU（最不频繁使用）

struct Node { int key, val, freq; };
// 每次 get/put 时 freq++
// 淘汰时选 freq 最小的

线程安全的 LRU

class ThreadSafeLRU : public LRUCache {
    mutable mutex mtx_;
    string get(int k) { lock_guard lk(mtx_); return LRUCache::get(k); }
};

加互斥锁保护所有公共方法
