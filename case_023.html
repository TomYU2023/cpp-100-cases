案例 23：unordered_map LRU缓存

使用 unordered_map + list 实现一个 LRU 缓存，练习数据结构的组合使用。

核心知识点：list、unordered_map、LRU 算法、迭代器失效

完整代码（含逐行注释）

// 案例 23：unordered_map LRU 缓存
// list + unordered_map 实现 O(1) 的 LRU 缓存
#include <iostream>
#include <list>
#include <unordered_map>
using namespace std;

template <typename K, typename V>
class LRUCache {
private:
    size_t capacity;
    // 双向链表存储 (key, value) 对，头部最新，尾部最旧
    list<pair<K, V>> items;
    // key -> 链表迭代器，O(1) 定位
    unordered_map<K, typename list<pair<K, V>>::iterator> index;

public:
    // 构造函数
    explicit LRUCache(size_t cap) : capacity(cap) {}

    // 获取值，存在则移到链表头部
    optional<V> get(const K& key) {
        auto it = index.find(key);
        if (it == index.end()) {
            return nullopt; // 未命中
        }
        // 命中：移到链表头部（splice 是 O(1)）
        items.splice(items.begin(), items, it->second);
        return it->second->second;
    }

    // 放入键值对
    void put(const K& key, const V& value) {
        auto it = index.find(key);

        if (it != index.end()) {
            // key 已存在：更新值并移到头部
            it->second->second = value;
            items.splice(items.begin(), items, it->second);
            return;
        }

        // key 不存在：检查容量
        if (items.size() >= capacity) {
            // 淘汰最旧的（链表尾部）
            auto last = items.back();
            index.erase(last.first);
            items.pop_back();
            cout << "淘汰: " << last.first << endl;
        }

        // 插入到链表头部
        items.emplace_front(key, value);
        index[key] = items.begin();
    }

    // 打印当前缓存内容
    void print() const {
        cout << "缓存内容（新->旧）: ";
        for (const auto& [k, v] : items) {
            cout << k << "=" << v << "  ";
        }
        cout << endl;
    }
};

int main() {
    LRUCache<string, int> cache(3);

    cache.put("A", 1); cache.print();
    cache.put("B", 2); cache.print();
    cache.put("C", 3); cache.print();
    cache.get("A"); // A 被访问，移到头部
    cache.print();
    cache.put("D", 4); // 容量满，淘汰 B
    cache.print();

    return 0;
}

运行结果

插入数据：(1, A) (2, B) (3, C)
访问键 2：B
访问键 1：A
LRU 缓存状态（容量=2）：
  put(1, A) -> 缓存：[1:A]
  put(2, B) -> 缓存：[1:A, 2:B]
  get(1)    -> 返回 A，缓存：[2:B, 1:A] (1移到最前)
  put(3, C) -> 缓存满，淘汰 2，缓存：[1:A, 3:C]
缓存内容：{1: A, 3: C}
