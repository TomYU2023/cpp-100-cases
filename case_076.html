案例 76：std::reduce 并行归约

使用 C++17 reduce 实现可并行的归约操作，对比 accumulate。

核心知识点：reduce、并行策略、accumulate 对比

完整代码（含逐行注释）

// 案例 76：std::reduce 并行归约
// 合并操作，支持并行执行
#include <iostream>
#include <numeric>
#include <vector>
#include <execution>
#include <string>
#include <chrono>
using namespace std;
using namespace chrono;

int main() {
    // 准备大数据集
    const int N = 10000000;
    vector<long long> data(N);
    iota(data.begin(), data.end(), 1); // 1, 2, 3, ...

    cout << "数据规模: " << N << endl;

    // 1. 顺序 accumulate
    auto start = high_resolution_clock::now();
    long long sum1 = accumulate(data.begin(), data.end(), 0LL);
    auto end = high_resolution_clock::now();
    double ms1 = duration<double, milli>(end - start).count();
    cout << "accumulate (顺序): " << ms1 << " ms, 结果=" << sum1 << endl;

    // 2. 顺序 reduce
    start = high_resolution_clock::now();
    long long sum2 = reduce(execution::seq, data.begin(), data.end(), 0LL);
    end = high_resolution_clock::now();
    double ms2 = duration<double, milli>(end - start).count();
    cout << "reduce (顺序): " << ms2 << " ms, 结果=" << sum2 << endl;

    // 3. 并行 reduce
    start = high_resolution_clock::now();
    long long sum3 = reduce(execution::par, data.begin(), data.end(), 0LL);
    end = high_resolution_clock::now();
    double ms3 = duration<double, milli>(end - start).count();
    cout << "reduce (并行): " << ms3 << " ms, 结果=" << sum3 << endl;

    // 4. 自定义归约操作：求乘积
    vector<int> smallVec = {2, 3, 4, 5};
    int product = reduce(smallVec.begin(), smallVec.end(),
                         1, multiplies<int>());
    cout << "\n乘积: 2 * 3 * 4 * 5 = " << product << endl;

    // 5. transform_reduce：先映射再归约
    // 计算平方和: 1^2 + 2^2 + 3^2 + 4^2 + 5^2
    vector<int> vec = {1, 2, 3, 4, 5};
    int sumSquares = transform_reduce(
        vec.begin(), vec.end(),
        0,
        plus<>(),            // 归约操作
        [](int x) { return x * x; }); // 映射操作
    cout << "平方和: " << sumSquares << endl;

    return 0;
}
