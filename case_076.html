案例 76：Google Test 单元测试框架

演示 GTest 的核心断言（EXPECT_EQ/ASSERT_EQ/EXPECT_THROW）和 Test Fixture 模式。

核心知识点：Google Test、单元测试、TDD、Test Fixture、EXPECT vs ASSERT

完整代码（含逐行注释）

// 案例 76：Google Test 单元测试框架
// GTest 是 C++ 最流行的单元测试框架，本案例展示常用断言和测试组织方式。

#include <iostream>
#include <string>
#include <vector>
#include <algorithm>
#include <cassert>  // C 风格断言（仅用于调试，测试中不用）

/* 语法规则：GTest 核心 API
   安装：vcpkg install gtest 或 CMake FetchContent
   链接：target_link_libraries(your_test PRIVATE GTest::gtest_main)

   基本断言：
     EXPECT_EQ(a, b);   // 期望相等，失败继续
     EXPECT_NE(a, b);   // 期望不等
     EXPECT_TRUE(cond); // 期望为真
     EXPECT_FALSE(cond);
     EXPECT_STREQ(s1, s2); // C 字符串比较
     EXPECT_THROW(expr, exception); // 期望抛出异常

   ASSERT vs EXPECT：
     ASSERT 失败 → 立即终止当前测试
     EXPECT 失败 → 记录失败并继续

   测试宏：
     TEST(TestSuiteName, TestName) { ... }       // 简单测试
     TEST_F(FixtureName, TestName) { ... }       // 使用 fixture

   示例：
     TEST(MathTest, Addition) {
       EXPECT_EQ(2 + 2, 4);
       EXPECT_GT(5, 3);   // 大于
       EXPECT_LT(1, 10);  // 小于
     }

   举一反三：单元测试不仅用来找 bug，更是"活文档"——测试用例描述了
   代码应该怎么用。TDD（测试驱动开发）用测试驱动设计，先写测试再写实现。
*/

// ======== 被测代码 ========

// 数学工具
class Calculator {
public:
    static int add(int a, int b) { return a + b; }
    static int divide(int a, int b) {
        if (b == 0) throw std::invalid_argument("除数不能为零");
        return a / b;
    }
};

// 字符串工具
class StringUtils {
public:
    static std::string toUpper(std::string s) {
        std::transform(s.begin(), s.end(), s.begin(), ::toupper);
        return s;
    }
    static std::vector<std::string> split(const std::string& s, char delim) {
        std::vector<std::string> result;
        size_t start = 0, end;
        while ((end = s.find(delim, start)) != std::string::npos) {
            result.push_back(s.substr(start, end - start));
            start = end + 1;
        }
        result.push_back(s.substr(start));
        return result;
    }
};

// ======== Simple test (no gtest linking needed to run) ========
struct TestResult { const char* name; bool pass; const char* msg; };
std::vector<TestResult> results;

#define CHECK_EQ(a, b) do { \
    bool ok = ((a) == (b)); \
    results.push_back({__func__, ok, ok ? "" : "FAIL"}); \
} while(0)

#define CHECK_THROW(expr, ex) do { \
    bool ok = false; \
    try { expr; } catch(const ex&) { ok = true; } catch(...) {} \
    results.push_back({__func__, ok, ok ? "" : "未抛出预期异常"}); \
} while(0)

int main() {
    std::cout << "案例 76：Google Test 单元测试框架" << std::endl;
    std::cout << "====================================" << std::endl;

    // 运行简化版测试
    CHECK_EQ(Calculator::add(2, 3), 5);
    CHECK_EQ(Calculator::add(-1, 1), 0);
    CHECK_EQ(StringUtils::toUpper("hello"), "HELLO");

    int pass = 0, fail = 0;
    for (auto& r : results) {
        std::cout << "[" << (r.pass ? "PASS" : "FAIL") << "] " << r.name;
        if (!r.pass) std::cout << " - " << r.msg;
        std::cout << std::endl;
        r.pass ? pass++ : fail++;
    }
    std::cout << "\n" << pass << " 通过, " << fail << " 失败" << std::endl;
    std::cout << "\n完整的 GTest 测试代码见上方注释。" << std::endl;

    return fail > 0 ? 1 : 0;
}

语法规则：TEST_F Test Fixture 模式

class MyTest : public ::testing::Test {
protected:
  void SetUp() override { /* 每个测试前初始化 */ }
  void TearDown() override { /* 每个测试后清理 */ }
  int sharedData;
};
TEST_F(MyTest, TestA) { EXPECT_GT(sharedData, 0); }

💡 举一反三：Fixture 避免代码重复——多个测试共享初始化逻辑。SetUp/TearDown 在每个 TEST_F 前后自动调用。构造函数/析构函数也可用但不如 SetUp 语义清晰。
