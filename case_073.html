案例 73：std::clamp 值裁剪

使用 clamp 将值限制在指定范围内，结合其他数值算法处理数据。

核心知识点：clamp、min/max、数值范围限制

完整代码（含逐行注释）

// 案例 73：std::clamp 值裁剪
// 将值限制在最小和最大边界内
#include <iostream>
#include <algorithm>
#include <vector>
#include <string>
using namespace std;

int main() {
    // 1. 基本用法：将值裁剪到 [min, max]
    cout << "=== std::clamp ===" << endl;
    cout << "clamp(5, 0, 10) = " << clamp(5, 0, 10) << endl;   // 5
    cout << "clamp(-3, 0, 10) = " << clamp(-3, 0, 10) << endl; // 0
    cout << "clamp(15, 0, 10) = " << clamp(15, 0, 10) << endl; // 10

    // 2. 裁剪容器中所有值
    vector<int> values = {-5, 0, 25, 50, 75, 100, 125};
    cout << "\n原始数据: ";
    for (int v : values) cout << v << " ";
    cout << endl;

    // 裁剪到 [0, 100]
    cout << "裁剪到 [0, 100]: ";
    for (int v : values) {
        cout << clamp(v, 0, 100) << " ";
    }
    cout << endl;

    // 3. 用 clamp 实现进度条百分比
    cout << "\n=== 进度百分比 ===" << endl;
    vector<double> rawPercents = {-10.0, 0.0, 45.5, 89.3, 100.0, 120.0, 55.0};

    for (double raw : rawPercents) {
        double percent = clamp(raw, 0.0, 100.0);
        int barWidth = static_cast<int>(percent / 5); // 每 5% 一个字符
        cout << raw << "%"
             << " -> " << string(barWidth, '#')
             << " " << percent << "%" << endl;
    }

    // 4. 对比 min/max 写法
    cout << "\n=== clamp vs min+max ===" << endl;
    int val = 150;
    cout << "clamp(150, 0, 100) = " << clamp(val, 0, 100) << endl;
    cout << "min(max(150,0),100) = " << min(max(val, 0), 100) << endl;

    return 0;
}
