案例 92：SFINAE 与 enable_if

使用 enable_if 在编译期根据类型条件选择不同实现，理解替换失败不是错误。

核心知识点：SFINAE、enable_if、类型萃取、条件编译期选择

完整代码（含逐行注释）

// 案例 92：SFINAE 与 enable_if
// 编译期根据类型条件选择函数重载
#include <iostream>
#include <type_traits>
#include <string>
using namespace std;

// 方式 1: enable_if 作为返回值类型
template <typename T>
typename enable_if<is_integral<T>::value, void>::type
process(T value) {
    cout << "处理整数: " << value << " (加 10 = "
         << value + 10 << ")" << endl;
}

template <typename T>
typename enable_if<is_floating_point<T>::value, void>::type
process(T value) {
    cout << "处理浮点数: " << value << " (平方 = "
         << value * value << ")" << endl;
}

// 方式 2: enable_if 作为默认模板参数
template <typename T,
          typename = enable_if_t<is_arithmetic<T>::value>>
void printDouble(T value) {
    cout << "2 * " << value << " = " << value * 2 << endl;
}

// 方式 3: 用 enable_if 控制类模板
template <typename T, typename Enable = void>
class Container {
public:
    static void info() { cout << "通用容器" << endl; }
};

// 对整数类型的特化
template <typename T>
class Container<T, enable_if_t<is_integral<T>::value>> {
public:
    static void info() { cout << "整数容器（有优化）" << endl; }
};

int main() {
    // 1. 根据参数类型自动选择重载
    process(42);          // 整数版本
    process(3.14);        // 浮点版本

    // 2. 仅限算术类型调用
    printDouble(5);       // OK
    printDouble(3.5);     // OK
    // printDouble(string("hi")); // 编译错误！

    // 3. 类模板的 enable_if
    Container<int>::info();          // 整数容器
    Container<double>::info();       // 通用容器
    Container<string>::info();       // 通用容器

    return 0;
}

运行结果

处理整数: 42 (加 10 = 52)
处理浮点数: 3.14 (平方 = 9.8596)
2 * 5 = 10
2 * 3.5 = 7
整数容器（有优化）
通用容器
通用容器
