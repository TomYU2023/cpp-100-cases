案例 75：std::sample 随机抽样

使用 sample 进行随机抽样，练习随机数生成器的正确使用。

核心知识点：sample、random_device、mt19937、分布

完整代码（含逐行注释）

// 案例 75：std::sample 随机抽样
// 从大数据集中随机抽取样本
#include <iostream>
#include <algorithm>
#include <random>
#include <vector>
#include <string>
#include <iterator>
using namespace std;

int main() {
    // 1. 准备数据集
    vector<string> students = {
        "Alice", "Bob", "Charlie", "David", "Eve",
        "Frank", "Grace", "Henry", "Ivy", "Jack",
        "Kate", "Leo", "Mary", "Nick", "Olivia"
    };

    cout << "总人数: " << students.size() << endl;

    // 2. 使用 sample 随机抽取
    random_device rd;                     // 真随机种子
    mt19937 gen(rd());                    // 梅森旋转生成器

    const int SAMPLE_COUNT = 5;

    // 稳定抽样：保持被选中元素的原始顺序
    vector<string> sample1;
    sample(students.begin(), students.end(),
           back_inserter(sample1), SAMPLE_COUNT, gen);

    cout << "\n=== 稳定抽样 (保持顺序) ===" << endl;
    for (size_t i = 0; i < sample1.size(); i++) {
        cout << i + 1 << ". " << sample1[i] << endl;
    }

    // 3. 不放回抽样（默认）
    cout << "\n=== 第二次抽样 ===" << endl;
    vector<string> sample2;
    sample(students.begin(), students.end(),
           back_inserter(sample2), SAMPLE_COUNT, gen);

    for (size_t i = 0; i < sample2.size(); i++) {
        cout << i + 1 << ". " << sample2[i] << endl;
    }

    // 4. 从中奖池抽奖
    vector<int> pool(100);
    iota(pool.begin(), pool.end(), 1); // 1~100

    vector<int> winners;
    sample(pool.begin(), pool.end(),
           back_inserter(winners), 3, gen);

    cout << "\n=== 抽奖结果 (1-100) ===" << endl;
    for (size_t i = 0; i < winners.size(); i++) {
        cout << "第 " << i + 1 << " 名: " << winners[i] << " 号" << endl;
    }

    return 0;
}
