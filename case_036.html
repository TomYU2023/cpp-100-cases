案例 36：regex 日志解析

使用正则表达式解析日志文件，提取时间戳、日志级别和消息内容。

核心知识点：regex、smatch、regex_match/regex_search

完整代码（含逐行注释）

// 案例 36：regex 日志解析
// 使用正则表达式提取日志信息
#include <iostream>
#include <string>
#include <regex>
#include <vector>
using namespace std;

struct LogEntry {
    string timestamp;
    string level;   // INFO, WARN, ERROR
    string message;
};

int main() {
    // 模拟日志行
    vector<string> logLines = {
        "2024-03-15 10:30:45 [INFO] 服务启动成功",
        "2024-03-15 10:31:02 [WARN] 磁盘使用率超过 80%",
        "2024-03-15 10:32:18 [ERROR] 数据库连接超时",
        "2024-03-15 10:33:00 [INFO] 用户 admin 登录",
        "2024-03-15 10:35:00 [ERROR] 文件未找到: config.json"
    };

    // 正则表达式：捕获时间戳、级别和消息
    // (\\d{4}-\\d{2}-\\d{2} \\d{2}:\\d{2}:\\d{2}) \\[(INFO|WARN|ERROR)\\] (.+)
    regex logPattern(
        R"((\d{4}-\d{2}-\d{2} \d{2}:\d{2}:\d{2}) \[(INFO|WARN|ERROR)\] (.+))"
    );

    vector<LogEntry> entries;
    int errorCount = 0;

    for (const auto& line : logLines) {
        smatch match;
        if (regex_match(line, match, logPattern)) {
            LogEntry entry;
            entry.timestamp = match[1].str();
            entry.level = match[2].str();
            entry.message = match[3].str();
            entries.push_back(entry);

            if (entry.level == "ERROR") errorCount++;
        }
    }

    // 输出解析结果
    cout << "=== 日志解析结果 ===" << endl;
    for (const auto& e : entries) {
        cout << "[" << e.timestamp << "] "
             << e.level << " | " << e.message << endl;
    }
    cout << "\n共解析 " << entries.size() << " 条日志，"
         << errorCount << " 条 ERROR" << endl;

    // 提取所有错误消息
    cout << "\n=== 错误日志 ===" << endl;
    regex errorPattern(R"(ERROR\] (.+))");
    for (const auto& line : logLines) {
        smatch m;
        if (regex_search(line, m, errorPattern)) {
            cout << "  " << m[1].str() << endl;
        }
    }

    return 0;
}

语法规则：regex 匹配与捕获

#include <regex>
regex pat(R"((\d+)年)");
smatch m;
if (regex_search(s, m, pat)) {
    cout << m[1]; // 第一个捕获组
}

举一反三：R"(...)" 是原始字符串字面量，不需要转义反斜杠。smatch 是 string 匹配结果。m[0] 是完整匹配，m[1] 开始是捕获组。

语法规则：regex_match vs regex_search

regex_match(s, m, pat); // 整个字符串必须完全匹配
regex_search(s, m, pat); // 搜索子串匹配

举一反三：regex_match 检查整个字符串是否满足模式（如验证邮箱格式）。regex_search 查找任意位置的匹配（如提取日志中的 IP 地址）。

语法规则：regex_replace 替换

#include <regex>
string result = regex_replace(s, pat, "new");
// 将匹配部分替换为新字符串

举一反三：正则表达式性能开销较大，高频场景考虑手动解析或使用 C++17 的 boyer_moore_searcher。复杂正则可先 regex::optimize 后复用。
