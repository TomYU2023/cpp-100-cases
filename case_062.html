案例 62：内存池实现

阶段：工程能力锻造难度：⭐⭐⭐⭐练什么：内存管理、placement new
手写固定大小的内存池，预分配大块内存，按需切分小块分配。理解 malloc/free 的底层原理，消除频繁 new/delete 的开销。

完整代码与逐行注释

case_062.cpp
1#include <iostream>


2#include <vector>


3#include <cstdint>


💬 第3行：uintptr_t 类型


4 


5// 固定大小内存池


6template<size_t BlockSize, size_t BlockCount = 1024>


💬 第6行：模板参数：块大小 + 块数量


7class MemoryPool {


8    static_assert(BlockSize >= sizeof(void*),


💬 第8行：编译期断言：块至少能放下一个指针


9        "BlockSize must be >= sizeof(void*)");


10 


11    union Block {


💬 第11行：union：同一内存位置存储不同数据类型


12        char data[BlockSize];


💬 第12行：当块被使用时：存储用户数据


13        Block* next;


💬 第13行：当块空闲时：存放下一个空闲块指针


14    };


15 


16    std::vector<Block> pool;


💬 第16行：用 vector 管理大块内存，自动 RAII


17    Block* freeList = nullptr;


💬 第17行：空闲链表头指针


18 


19public:


20    MemoryPool() {


21        pool.resize(BlockCount);


💬 第21行：预分配所有块


22        // 初始化空闲链表：把所有块串起来


23        for (size_t i = 0; i < BlockCount - 1; ++i) {


24            pool[i].next = &pool[i + 1];


💬 第24行：每块 next 指向下一块


25        }


26        pool[BlockCount - 1].next = nullptr;


💬 第26行：最后一块 next 为 null


27        freeList = &pool[0];


💬 第27行：空闲链表指向第一块


28    }


29 


30    void* allocate() {


31        if (!freeList) throw std::bad_alloc();


💬 第31行：无空闲块 → 抛异常


32        void* ptr = freeList;


💬 第32行：取链表头部


33        freeList = freeList->next;


💬 第33行：头指针后移


34        return ptr;


💬 第34行：返回可用内存


35    }


36 


37    void deallocate(void* ptr) {


38        if (!ptr) return;


💬 第38行：空指针不处理


39        auto block = static_cast<Block*>(ptr);


💬 第39行：还原为 Block 类型


40        block->next = freeList;


💬 第40行：头插法归还到空闲链表


41        freeList = block;


42    }


43 


44    // 便捷接口：分配 + 构造


45    template<typename T, typename... Args>


46    T* create(Args&&... args) {


47        void* mem = allocate();


💬 第47行：从内存池分配


48        return new (mem) T(std::forward<Args>(args)...);


💬 第48行：placement new 在已分配内存构造对象


49    }


50 


51    // 便捷接口：析构 + 归还


52    template<typename T>


53    void destroy(T* ptr) {


54        if (ptr) {


55            ptr->~T();


💬 第55行：手动调用析构函数


56            deallocate(ptr);


💬 第56行：归还内存


57        }


58    }


59};


60 


61// 测试用的简单类


62struct Widget {


63    int id;


64    double value;


65    Widget(int i, double v) : id(i), value(v) {}


66    ~Widget() { std::cout << "Widget " << id << " destroyed\n"; }


67};


68 


69int main() {


70    MemoryPool<sizeof(Widget), 4> pool;


💬 第70行：创建能存放 4 个 Widget 的内存池


71 


72    auto w1 = pool.create<Widget>(1, 3.14);


💬 第72行：从内存池构造 Widget


73    auto w2 = pool.create<Widget>(2, 2.71);


74    std::cout << "w1: id=" << w1->id << ", val=" << w1->value << std::endl;


75 


76    pool.destroy(w1);


💬 第76行：析构并归还内存


77 


78    auto w3 = pool.create<Widget>(3, 1.41);


💬 第78行：复用 w1 归还的内存


79    std::cout << "w3: id=" << w3->id << std::endl;


80 


81    pool.destroy(w2);


82    pool.destroy(w3);


83    return 0;


💬 第83行：内存池析构时 vector 自动释放所有块


84}
📖 语法规则：placement new
new (address) Type(args); // 在已分配内存上构造对象

示例：

char buf[100];
new (buf) int(42);  // 在 buf 上构造一个 int
// 注意：placement new 不分配内存，只在已有内存上调用构造函数

💡 举一反三：

  • placement new 返回的就是传入的地址
  • placement new 构造的对象不能 delete，要手动调用析构：obj->~T()
  • 常用于内存池、嵌入式等场景
📖 语法规则：链表管理空闲块
将空闲块的首部用作链表节点的 next 指针

示例：

struct Block { Block* next; };
Block* freeList;
void* alloc() {
  Block* b = freeList;
  freeList = freeList->next;
  return b;
}

💡 举一反三：

  • 每个块至少 sizeof(void*) 大，否则放不下 next 指针
  • 这就是 malloc 底层 jemalloc/tcmalloc 的核心思想
  • 空闲链表是侵入式数据结构——数据本身承载指针
