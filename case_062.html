案例 62：无锁队列 SPSC

实现一个单生产者单消费者的无锁环形队列，理解内存序的实际应用。

核心知识点：环形缓冲区、原子操作、memory_order、无锁编程

完整代码（含逐行注释）

// 案例 62：无锁队列 SPSC
// 单生产者单消费者环形队列
#include <iostream>
#include <atomic>
#include <vector>
#include <thread>
#include <chrono>
using namespace std;

template <typename T, size_t Capacity>
class SPSCQueue {
private:
    // 环形缓冲区（多分配一个位置区分空和满）
    static constexpr size_t BufferSize = Capacity + 1;
    vector<T> buffer{BufferSize};
    atomic<size_t> writePos{0}; // 生产者写入位置
    atomic<size_t> readPos{0};  // 消费者读取位置

public:
    // 生产者：尝试入队
    bool push(const T& item) {
        size_t w = writePos.load(memory_order_relaxed);
        size_t next = (w + 1) % BufferSize;

        // 检查队列是否满（下一个位置是读位置）
        if (next == readPos.load(memory_order_acquire))
            return false; // 队列满

        buffer[w] = item;
        writePos.store(next, memory_order_release);
        return true;
    }

    // 消费者：尝试出队
    bool pop(T& item) {
        size_t r = readPos.load(memory_order_relaxed);

        // 检查队列是否空（读位置等于写位置）
        if (r == writePos.load(memory_order_acquire))
            return false; // 队列空

        item = buffer[r];
        readPos.store((r + 1) % BufferSize, memory_order_release);
        return true;
    }

    // SPMC：单生产者多消费者版本（仅统计，需额外同步）
    bool empty() const {
        return readPos.load(memory_order_acquire) ==
               writePos.load(memory_order_acquire);
    }
};

int main() {
    SPSCQueue<int, 8> queue;
    atomic<bool> done{false};

    // 生产者线程
    thread producer([&]() {
        for (int i = 1; i <= 20; i++) {
            while (!queue.push(i)) {
                // 队列满则自旋等待
                this_thread::yield();
            }
            cout << "生产: " << i << endl;
            this_thread::sleep_for(chrono::milliseconds(50));
        }
        done.store(true);
    });

    // 消费者线程
    thread consumer([&]() {
        int value;
        while (!done.load() || !queue.empty()) {
            if (queue.pop(value)) {
                cout << "  消费: " << value << endl;
                this_thread::sleep_for(chrono::milliseconds(80));
            } else {
                this_thread::yield();
            }
        }
    });

    producer.join();
    consumer.join();

    return 0;
}
