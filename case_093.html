案例 93：2D 游戏引擎基础

实现 ECS（实体组件系统）架构的游戏引擎核心：游戏循环、移动/输入/AI/渲染系统、终端精灵渲染。

核心知识点：ECS 架构、模板类组件存储、游戏循环 + delta time、std::chrono 帧率控制、ANSI 控制台渲染

完整代码（含逐行注释）

// 案例 93：2D 游戏引擎基础
// 实现游戏引擎的核心组件：游戏循环、实体组件系统（ECS）、精灵渲染、输入处理。
// ECS 是现代游戏引擎的主流架构（Unity DOTS、Bevy、EnTT）。

#include <iostream>
#include <vector>
#include <map>
#include <memory>
#include <functional>
#include <chrono>
#include <thread>
#include <algorithm>

// ─── ECS 框架（精简版）───────────────────────────
using Entity = int;
Entity nextEntity = 0;

// 组件：位置
struct Position { float x = 0, y = 0; };

// 组件：速度
struct Velocity { float dx = 0, dy = 0; };

// 组件：精灵（字符表示，控制台渲染用）
struct Sprite {
    char ch = '?';
    int color = 7;  // 颜色编号
};

// 组件：玩家控制标签
struct PlayerControlled {};

// 组件：AI 移动标签
struct AIMovement {
    float timer = 0;
    float interval = 2.0f;
};

// ─── 组件存储（Component Store）─────────────────
// 每个组件类型一个存储，用 map<Entity, T> 管理
template<typename T>
class ComponentStore {
    std::map<Entity, T> data;
public:
    void add(Entity e, const T& c) { data[e] = c; }
    void remove(Entity e) { data.erase(e); }
    T* get(Entity e) {
        auto it = data.find(e);
        return it != data.end() ? &it->second : nullptr;
    }
    bool has(Entity e) { return data.find(e) != data.end(); }

    // 遍历所有拥有该组件的实体
    void forEach(std::function<void(Entity, T&)> fn) {
        for (auto& [e, c] : data) fn(e, c);
    }
};

// ─── 世界（World）─ 管理所有组件存储 ────────────
struct World {
    ComponentStore<Position> positions;
    ComponentStore<Velocity> velocities;
    ComponentStore<Sprite> sprites;
    ComponentStore<PlayerControlled> players;
    ComponentStore<AIMovement> aiMovements;

    Entity createEntity() {
        return nextEntity++;
    }

    void destroyEntity(Entity e) {
        positions.remove(e);
        velocities.remove(e);
        sprites.remove(e);
        players.remove(e);
        aiMovements.remove(e);
    }
};

// ─── 系统（Systems）──────────────────────────────
// 移动系统：速度 → 位置
void movementSystem(World& world, float dt) {
    world.velocities.forEach([&](Entity e, Velocity& v) {
        auto* p = world.positions.get(e);
        if (p) {
            p->x += v.dx * dt;
            p->y += v.dy * dt;
        }
    });
}

// 输入系统：读取键盘并更新玩家速度
void inputSystem(World& world) {
    world.players.forEach([&](Entity e, PlayerControlled&) {
        auto* v = world.velocities.get(e);
        if (!v) return;
        v->dx = v->dy = 0;
        // 简化：随机模拟输入方向
        static int frame = 0;
        frame++;
        if (frame % 60 < 20) v->dx = 5;
        else if (frame % 60 < 40) v->dx = -5;
        else if (frame % 60 < 55) v->dy = 3;
    });
}

// AI 系统：定时改变敌人方向
void aiSystem(World& world, float dt) {
    world.aiMovements.forEach([&](Entity e, AIMovement& ai) {
        ai.timer -= dt;
        if (ai.timer <= 0) {
            ai.timer = ai.interval;
            auto* v = world.velocities.get(e);
            if (v) {
                v->dx = (rand() % 3 - 1) * 3.0f;  // -3, 0, 3
                v->dy = (rand() % 3 - 1) * 3.0f;
            }
        }
    });
}

// 渲染系统：输出精灵位置
void renderSystem(World& world) {
    std::cout << "\033[2J\033[H";  // 清屏 + 光标归位（ANSI 转义）
    std::cout << "=== 2D Game Engine Demo ===\n";

    // 构建 20x10 的网格
    char grid[10][20];
    for (int y = 0; y < 10; y++)
        for (int x = 0; x < 20; x++)
            grid[y][x] = '.';

    world.sprites.forEach([&](Entity e, Sprite& s) {
        auto* p = world.positions.get(e);
        if (!p) return;
        int gx = int(p->x), gy = int(p->y);
        if (gx >= 0 && gx < 20 && gy >= 0 && gy < 10)
            grid[gy][gx] = s.ch;
    });

    for (int y = 0; y < 10; y++) {
        for (int x = 0; x < 20; x++)
            std::cout << grid[y][x];
        std::cout << std::endl;
    }
}

// ─── 主循环 ──────────────────────────────────────
int main() {
    World world;
    srand((unsigned)time(0));

    // 创建玩家实体
    Entity player = world.createEntity();
    world.positions.add(player, {10, 5});
    world.velocities.add(player, {0, 0});
    world.sprites.add(player, {'@', 15});
    world.players.add(player, {});

    // 创建 3 个敌人实体
    for (int i = 0; i < 3; i++) {
        Entity enemy = world.createEntity();
        world.positions.add(enemy, {float(rand() % 20), float(rand() % 10)});
        world.velocities.add(enemy, {0, 0});
        world.sprites.add(enemy, {'E', 12});
        world.aiMovements.add(enemy, {float(rand() % 3 + 1), 2.0f});
    }

    // 游戏主循环
    const float dt = 0.1f;  // 10 FPS（终端渲染慢）
    int frame = 0;
    while (frame < 100) {
        inputSystem(world);
        aiSystem(world, dt);
        movementSystem(world, dt);
        renderSystem(world);

        std::this_thread::sleep_for(
            std::chrono::milliseconds(int(dt * 1000)));
        frame++;
    }

    std::cout << "\n演示结束。" << std::endl;
    return 0;
}

语法规则：ECS（Entity Component System）架构模式

// 传统 OOP 继承链（不推荐）：
class Player : public GameObject { /* 位置+速度+精灵... */ };
class Enemy : public GameObject { /* 位置+速度+精灵+AI... */ };
// → 钻石继承、重复代码、难以组合

// ECS 模式（推荐）：
// Entity = ID（纯数字）
// Component = 纯数据（Position, Velocity, Sprite...）
// System = 纯逻辑（MovementSystem, RenderSystem...）

💡 举一反三：ECS 的核心优势：(1) 组合优于继承 — 任意给实体添加组件组合；(2) 数据导向 — 组件连续存储，缓存友好；(3) 系统可并行 — 无数据竞争。现代的 EnTT 库在 C++ 中实现了生产级 ECS。

语法规则：模板类 ComponentStore

template<typename T>
class ComponentStore {
    std::map<Entity, T> data;
public:
    void add(Entity e, const T& c) { data[e] = c; }
    T* get(Entity e) {
        auto it = data.find(e);
        return it != data.end() ? &it->second : nullptr;
    }
    void forEach(std::function<void(Entity, T&)> fn) {
        for (auto& [e, c] : data) fn(e, c);
    }
};

💡 举一反三：模板让每种组件类型共享同一套增删查逻辑。生产级 ECS 使用稀疏集合（Sparse Set）或原型（Archetype）替代 map，以实现连续内存和 SIMD 优化。

语法规则：std::chrono 和 sleep_for 定时

#include <chrono>
#include <thread>
std::this_thread::sleep_for(std::chrono::milliseconds(100));
// 或者用时间点做帧率控制：
auto start = std::chrono::steady_clock::now();
// ... 帧逻辑 ...
auto elapsed = std::chrono::steady_clock::now() - start;
float dt = std::chrono::duration<float>(elapsed).count();

💡 举一反三：游戏循环的黄金法则：总是用 delta time（dt）乘速度，而不是假设固定帧率。sleep_for 在终端游戏中使用，因为不需要 60fps。真正的游戏引擎用垂直同步（VSync）或忙等（spin wait）。
