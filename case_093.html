案例 93：constexpr 编译期计算

利用 constexpr 在编译期完成计算，减少运行时开销，实现零成本抽象。

核心知识点：constexpr、编译期计算、constexpr if、consteval

完整代码（含逐行注释）

// 案例 93：constexpr 编译期计算
// 在编译期完成计算，零运行时开销
#include <iostream>
#include <array>
using namespace std;

// 1. constexpr 函数：编译期可求值
constexpr int factorial(int n) {
    int result = 1;
    for (int i = 2; i <= n; i++) {
        result *= i;
    }
    return result;
}

// 2. constexpr 构造函数和类
class Point {
private:
    int x_, y_;
public:
    constexpr Point(int x, int y) : x_(x), y_(y) {}
    constexpr int x() const { return x_; }
    constexpr int y() const { return y_; }
    constexpr int distanceSquared() const {
        return x_ * x_ + y_ * y_;
    }
};

// 3. 模板 + constexpr：编译期斐波那契
template <int N>
struct Fibonacci {
    static constexpr int value =
        Fibonacci<N-1>::value + Fibonacci<N-2>::value;
};

template <> struct Fibonacci<0> { static constexpr int value = 0; };
template <> struct Fibonacci<1> { static constexpr int value = 1; };

int main() {
    // 编译期计算阶乘
    constexpr int fact10 = factorial(10);
    cout << "10! = " << fact10 << " (编译期计算)" << endl;

    // 运行时调用同样有效
    int n = 5;
    cout << n << "! = " << factorial(n) << " (运行时计算)" << endl;

    // 编译期创建对象
    constexpr Point p(3, 4);
    constexpr int dist = p.distanceSquared();
    cout << "点(3,4) 距离平方 = " << dist << " (编译期计算)" << endl;

    // 编译期数组大小（必须编译期已知）
    constexpr int size = factorial(4); // 24
    array<int, size> arr{}; // 编译期确定大小
    cout << "编译期数组大小: " << arr.size() << endl;

    // 模板斐波那契
    cout << "\n斐波那契(10) = " << Fibonacci<10>::value
         << " (编译期计算)" << endl;

    return 0;
}

语法规则：constexpr 函数

constexpr int square(int x) {
    return x * x;
}
// 当参数是编译期常量时，结果在编译期计算
// 当参数是运行时变量时，结果在运行时计算

举一反三：C++11 的 constexpr 函数只能有 return 语句。C++14 允许循环、分支、局部变量。C++20 允许动态分配（编译期 new/delete）。consteval (C++20) 强制编译期求值。

语法规则：constexpr 类

class Point {
    constexpr Point(int x, int y) : x_(x), y_(y) {}
    constexpr int x() const { return x_; }
};
// 构造函数和成员函数均可 constexpr

举一反三：constexpr 类可在编译期创建和使用。成员函数标记 constexpr 使得编译期对象可以调用。析构函数也可 constexpr（C++20），支持编译期 RAII。

语法规则：编译期 vs 运行时

// 编译期计算：
constexpr int val = factorial(10); // O(0) 运行时开销
array<int, factorial(5)> arr;      // 编译期确定大小
// 运行时计算：
int x = readInput();
int y = factorial(x); // 正常的运行时代码

举一反三：constexpr 提供「零成本抽象」：同一份代码根据需要编译为编译期或运行时版本。减少 magic number，用有意义的 constexpr 函数和常量替代硬编码值。
