案例 52：RAII 文件句柄

实现一个 RAII 封装的文件句柄类，确保文件在任何情况下都能正确关闭。

核心知识点：RAII、析构函数、移动语义、异常安全

完整代码（含逐行注释）

// 案例 52：RAII 文件句柄
// 用 RAII 封装文件资源，保证自动关闭
#include <iostream>
#include <cstdio>    // FILE*, fopen, fclose, fread, fwrite
#include <string>
#include <stdexcept>
using namespace std;

// RAII 文件句柄封装
class FileHandle {
private:
    FILE* file;

public:
    // 构造函数：打开文件
    explicit FileHandle(const string& path, const string& mode) {
        file = fopen(path.c_str(), mode.c_str());
        if (!file) {
            throw runtime_error("无法打开文件: " + path);
        }
        cout << "文件已打开: " << path << endl;
    }

    // 禁止拷贝（文件句柄不可共享）
    FileHandle(const FileHandle&) = delete;
    FileHandle& operator=(const FileHandle&) = delete;

    // 移动构造：转移所有权
    FileHandle(FileHandle&& other) noexcept : file(other.file) {
        other.file = nullptr;
        cout << "文件句柄已移动" << endl;
    }

    // 移动赋值
    FileHandle& operator=(FileHandle&& other) noexcept {
        if (this != &other) {
            close();            // 先关闭自己的文件
            file = other.file;
            other.file = nullptr;
        }
        return *this;
    }

    // 析构函数：自动关闭
    ~FileHandle() {
        close();
    }

    // 写入数据
    void write(const string& data) {
        if (!file) throw runtime_error("文件未打开");
        fwrite(data.c_str(), 1, data.size(), file);
    }

    // 读取全部内容
    string readAll() {
        if (!file) throw runtime_error("文件未打开");
        fseek(file, 0, SEEK_END);
        long size = ftell(file);
        fseek(file, 0, SEEK_SET);
        string content(size, '\0');
        fread(&content[0], 1, size, file);
        return content;
    }

    // 显式关闭
    void close() {
        if (file) {
            fclose(file);
            cout << "文件已关闭" << endl;
            file = nullptr;
        }
    }

    // 检查是否打开
    bool isOpen() const { return file != nullptr; }
};

int main() {
    try {
        // 创建文件并写入
        FileHandle fh("raii_test.txt", "w+");
        fh.write("Hello, RAII!\nFile handle test.\n");

        // 回到文件开头读取
        string content = fh.readAll();
        cout << "文件内容:\n" << content;

        // 移动文件句柄
        FileHandle fh2 = move(fh);
        cout << "fh.isOpen() = " << fh.isOpen() << endl;  // false
        cout << "fh2.isOpen() = " << fh2.isOpen() << endl; // true

    } // fh2 在此自动关闭
    catch (const exception& e) {
        cerr << "错误: " << e.what() << endl;
    }

    return 0;
}

语法规则：RAII 核心原则

// 资源获取即初始化
// 构造时分配，析构时释放
class Resource {
    Resource() { acquire(); }
    ~Resource() { release(); }
};

举一反三：RAII 是 C++ 最核心的惯用法。资源可以是内存、文件、锁、套接字等。异常发生时栈展开自动调用析构，保证资源不泄漏。永远不会忘记释放。

语法规则：delete 拷贝 + 实现移动

FileHandle(const FileHandle&) = delete;
FileHandle(FileHandle&& other) noexcept : file(other.file) {
    other.file = nullptr;
}

举一反三：资源管理类通常禁止拷贝（唯一所有权），但支持移动（转移所有权）。noexcept 对 STL 容器很重要——只有 noexcept 的移动构造，vector 扩容才会使用移动而非拷贝。

语法规则：C FILE* vs C++ fstream

// C: FILE* f = fopen(...); fclose(f);
// C++: fstream f(...); // RAII 自动关闭
// 本案例封装 C API 演示 RAII 原理

举一反三：C++ 标准库的 fstream 已经是 RAII 的。C 的 FILE* 需要手动 close，封装成 RAII 类可避免遗漏关闭。类似地，任何 C API 的资源句柄都可 RAII 封装。
