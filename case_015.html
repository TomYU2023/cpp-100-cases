案例 15：模板类:通用栈

使用模板实现一个通用栈类，支持任意类型的压栈、出栈、查看栈顶操作。

核心知识点：类模板、模板成员函数、栈数据结构

完整代码（含逐行注释）

// 案例 15：模板类：通用栈
// 实现一个支持任意类型的栈
#include <iostream>
#include <stdexcept> // std::out_of_range
using namespace std;

// 类模板：通用栈
template <typename T>
class Stack {
private:
    T* data;          // 动态数组存储栈元素
    size_t capacity;  // 当前容量
    size_t topIndex;  // 栈顶索引（指向下一个空位）

    // 扩容
    void expand() {
        capacity = (capacity == 0) ? 4 : capacity * 2;
        T* newData = new T[capacity];
        for (size_t i = 0; i < topIndex; i++) {
            newData[i] = data[i];
        }
        delete[] data;
        data = newData;
    }

public:
    Stack() : data(nullptr), capacity(0), topIndex(0) {}
    ~Stack() { delete[] data; }

    // 压栈
    void push(const T& value) {
        if (topIndex == capacity) expand();
        data[topIndex++] = value;
    }

    // 出栈
    T pop() {
        if (topIndex == 0) {
            throw out_of_range("Stack is empty!");
        }
        return data[--topIndex];
    }

    // 查看栈顶（不弹出）
    T& top() {
        if (topIndex == 0) {
            throw out_of_range("Stack is empty!");
        }
        return data[topIndex - 1];
    }

    bool empty() const { return topIndex == 0; }
    size_t size() const { return topIndex; }
};

int main() {
    // 整数栈
    Stack<int> intStack;
    intStack.push(10);
    intStack.push(20);
    intStack.push(30);
    cout << "int top: " << intStack.top() << endl;
    cout << "pop: " << intStack.pop() << endl;

    // 字符串栈
    Stack<string> strStack;
    strStack.push("Hello");
    strStack.push("World");
    while (!strStack.empty()) {
        cout << strStack.pop() << " ";
    }
    cout << endl;

    return 0;
}

运行结果

int top: 30
pop: 30
World Hello
