案例 98：std::integer_sequence 序列生成

使用 index_sequence 在编译期生成整数序列，实现高效元编程技巧。

核心知识点：integer_sequence、index_sequence、编译期展开、tuple 遍历

完整代码（含逐行注释）

// 案例 98：std::integer_sequence 序列生成
// 编译期生成整数序列，用于模板展开
#include <iostream>
#include <tuple>
#include <utility>
#include <string>
#include <array>
using namespace std;

// 1. 使用 index_sequence 打印 tuple 的每个元素
template <typename Tuple, size_t... I>
void printTupleImpl(const Tuple& t, index_sequence<I...>) {
    // 折叠表达式展开: (操作, ...)
    ((cout << (I == 0 ? "" : ", ") << get<I>(t)), ...);
}

template <typename... Args>
void printTuple(const tuple<Args...>& t) {
    printTupleImpl(t, index_sequence_for<Args...>{});
}

// 2. 用 index_sequence 生成编译期数组
template <typename T, size_t N, size_t... I>
constexpr array<T, N> makeArrayImpl(T value, index_sequence<I...>) {
    return {{(static_cast<void>(I), value)...}};
}

template <typename T, size_t N>
constexpr array<T, N> makeArray(T value) {
    return makeArrayImpl<T, N>(value, make_index_sequence<N>{});
}

// 3. 对 tuple 的每个元素应用函数
template <typename Tuple, typename Func, size_t... I>
void forEachImpl(Tuple&& t, Func&& f, index_sequence<I...>) {
    (f(get<I>(forward<Tuple>(t))), ...);
}

template <typename Tuple, typename Func>
void forEach(Tuple&& t, Func&& f) {
    forEachImpl(forward<Tuple>(t), forward<Func>(f),
                make_index_sequence<
                    tuple_size_v<decay_t<Tuple>>>{});
}

int main() {
    // 1. 打印 tuple
    auto t = make_tuple(42, 3.14, string("hello"), 'A');
    cout << "tuple 内容: ";
    printTuple(t);
    cout << endl;

    // 2. 编译期数组
    constexpr auto arr = makeArray<int, 5>(42);
    cout << "编译期数组: ";
    for (auto x : arr) cout << x << " ";
    cout << endl;

    // 3. 遍历 tuple 的每个元素
    cout << "\n遍历 tuple:" << endl;
    forEach(t, [](const auto& val) {
        cout << "  元素: " << val << endl;
    });

    return 0;
}

运行结果

编译期数组: 42 42 42 42 42 

遍历 tuple:

  元素: 3.14
  元素: hello
  元素: A
