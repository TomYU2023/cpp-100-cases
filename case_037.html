案例 037

thread 并行求和

完整代码（逐行注释）

1 #include <iostream>
2 #include <thread> std::thread
3 #include <vector>
4 #include <numeric> accumulate
5 #include <mutex> 互斥锁
6 #include <chrono> 计时
7 using namespace std;
8 using namespace chrono;
9
10 // 无锁并行求和
11 long long parallelSum(const vector<int>& v, int numThreads) {
12 size_t chunkSize = (v.size() + numThreads - 1) / numThreads; 向上取整
13 vector<thread> threads;
14 vector<long long> partialSums(numThreads, 0); 每个线程的结果
15
16 for (int t = 0; t < numThreads; t++) {
17 size_t start = t * chunkSize;
18 size_t end = min(start + chunkSize, v.size());
19
20 threads.emplace_back([&, t, start, end]() { 创建线程
21 partialSums[t] = accumulate(v.begin() + start,
22 v.begin() + end, 0LL);
23 });
24 }
25
26 for (auto& th : threads) th.join(); 等待所有线程完成
27
28 return accumulate(partialSums.begin(), partialSums.end(), 0LL); 汇总
29 }
30
31 int main() {
32 const int N = 10000000; 一千万个元素
33 vector<int> data(N);
34 for (int i = 0; i < N; i++) data[i] = i % 100; 0..99 循环
35
36 // 单线程基准
37 auto t1 = high_resolution_clock::now();
38 auto sum1 = accumulate(data.begin(), data.end(), 0LL);
39 auto t2 = high_resolution_clock::now();
40 cout << "单线程: " << sum1 << " 耗时 "
41 << duration_cast<milliseconds>(t2 - t1).count() << " ms\n";
42
43 // 多线程版本
44 int numThreads = thread::hardware_concurrency(); 获取 CPU 核心数
45 cout << "使用 " << numThreads << " 个线程\n";
46
47 auto t3 = high_resolution_clock::now();
48 auto sum2 = parallelSum(data, numThreads);
49 auto t4 = high_resolution_clock::now();
50 cout << "多线程: " << sum2 << " 耗时 "
51 << duration_cast<milliseconds>(t4 - t3).count() << " ms\n";
52
53 return 0;
54 }

新语法讲解

thread 基础

thread 构造时传入一个可调用对象和参数，立即开始在新线程中执行。join() 等待线程完成，detach() 分离线程（让其独立运行）。必须在 thread 析构前调用 join() 或 detach()，否则 terminate。

举一反三

thread t1([](){ cout << "线程1"; });
thread t2(doWork, arg1, arg2); // 传参数
t1.join(); // 等待
// 错误: thread 析构时未 join/detach → std::terminate

hardware_concurrency

thread::hardware_concurrency() 返回系统支持的并发线程数（通常 = CPU 核心数，含超线程）。用于决定创建多少线程最合适。可能返回 0（无法确定时），此时应使用合理的默认值如 4。

举一反三

int n = thread::hardware_concurrency();
if (n == 0) n = 4; // 兜底
// 推荐: 计算密集型用 n 个线程
// IO 密集型可以更多（如 2*n）

举一反三

数据竞争（Data Race）演示

int counter = 0;
// 两个线程同时 counter++ 结果是未定义的
thread t1([&]{ for(int i=0;i<100000;i++) counter++; });
thread t2([&]{ for(int i=0;i<100000;i++) counter++; });
// counter 不一定等于 200000！

多线程写同一变量需要同步

jthread (C++20)

jthread t([](){ /* work */ });
// jthread 析构时自动 join，不会 terminate
// 还支持 stop_token 协作式取消

jthread 是 thread 的安全替代品

thread_local 存储周期

thread_local int tls = 0; // 每个线程有自己的副本
// 类似全局变量，但每个线程一份

线程局部变量不需要加锁
