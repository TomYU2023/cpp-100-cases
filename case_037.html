案例 37：thread 并行求和

使用多线程将大数组分块并行求和，理解线程创建和数据分块的模式。

核心知识点：std::thread、join、数据分块、原子变量

完整代码（含逐行注释）

// 案例 37：thread 并行求和
// 多线程分块计算数组总和
#include <iostream>
#include <vector>
#include <thread>
#include <numeric>
#include <chrono>
#include <functional>
using namespace std;

// 计算 [start, end) 区间的和
long long partialSum(const vector<int>& arr, size_t start, size_t end) {
    long long sum = 0;
    for (size_t i = start; i < end; i++) {
        sum += arr[i];
    }
    return sum;
}

int main() {
    const size_t SIZE = 100000000; // 1 亿个元素
    const unsigned int NUM_THREADS = thread::hardware_concurrency();

    // 初始化大数组
    cout << "初始化 " << SIZE << " 个元素的数组..." << endl;
    vector<int> arr(SIZE, 1); // 全部填充 1

    cout << "CPU 核心数: " << NUM_THREADS << endl;

    // --- 单线程求和 ---
    auto start = chrono::high_resolution_clock::now();
    long long serialSum = partialSum(arr, 0, SIZE);
    auto end = chrono::high_resolution_clock::now();
    auto serialTime = chrono::duration<double, milli>(end - start).count();
    cout << "单线程: " << serialSum << " 耗时 " << serialTime << " ms" << endl;

    // --- 多线程并行求和 ---
    start = chrono::high_resolution_clock::now();

    vector<thread> threads;
    vector<long long> results(NUM_THREADS, 0);
    // 每个线程处理的数据块大小
    size_t chunkSize = SIZE / NUM_THREADS;

    for (unsigned int i = 0; i < NUM_THREADS; i++) {
        size_t chunkStart = i * chunkSize;
        size_t chunkEnd = (i == NUM_THREADS - 1) ? SIZE : chunkStart + chunkSize;

        // 启动线程，结果存入 results[i]
        threads.emplace_back([&arr, chunkStart, chunkEnd, &results, i]() {
            results[i] = partialSum(arr, chunkStart, chunkEnd);
        });
    }

    // 等待所有线程完成
    for (auto& t : threads) t.join();

    // 汇总结果
    long long parallelSum = accumulate(results.begin(), results.end(), 0LL);

    end = chrono::high_resolution_clock::now();
    auto parallelTime = chrono::duration<double, milli>(end - start).count();
    cout << "多线程: " << parallelSum << " 耗时 " << parallelTime << " ms" << endl;

    if (serialSum == parallelSum) {
        cout << "结果正确！加速比: " << (serialTime / parallelTime) << "x" << endl;
    } else {
        cout << "结果不一致！" << endl;
    }

    return 0;
}
