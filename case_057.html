案例 57：fstream 二进制读写

使用 fstream 进行二进制文件的读写操作，包含 struct 序列化。

核心知识点：fstream 二进制模式、read/write、reinterpret_cast

完整代码（含逐行注释）

// 案例 57：fstream 二进制读写
// 结构体的二进制序列化
#include <iostream>
#include <fstream>
#include <string>
#include <vector>
using namespace std;

// 数据记录结构
#pragma pack(push, 1) // 禁用内存对齐，确保跨平台大小一致
struct Record {
    int id;
    char name[20];
    double score;
};
#pragma pack(pop)

// 写入二进制文件
void writeRecords(const string& filename, const vector<Record>& records) {
    ofstream out(filename, ios::binary);
    if (!out) {
        cerr << "无法创建文件: " << filename << endl;
        return;
    }

    // 写入记录数量
    size_t count = records.size();
    out.write(reinterpret_cast<const char*>(&count), sizeof(count));

    // 逐条写入记录
    for (const auto& rec : records) {
        out.write(reinterpret_cast<const char*>(&rec), sizeof(Record));
    }
    out.close();
    cout << "已写入 " << count << " 条记录" << endl;
}

// 读取二进制文件
vector<Record> readRecords(const string& filename) {
    vector<Record> records;
    ifstream in(filename, ios::binary);
    if (!in) {
        cerr << "无法打开文件: " << filename << endl;
        return records;
    }

    // 读取记录数量
    size_t count;
    in.read(reinterpret_cast<char*>(&count), sizeof(count));

    // 逐条读取
    for (size_t i = 0; i < count; i++) {
        Record rec;
        in.read(reinterpret_cast<char*>(&rec), sizeof(Record));
        records.push_back(rec);
    }
    in.close();

    cout << "已读取 " << records.size() << " 条记录" << endl;
    return records;
}

int main() {
    // 准备数据
    vector<Record> data = {
        {1, "Alice", 95.5},
        {2, "Bob", 88.0},
        {3, "Charlie", 92.3}
    };

    // 写入
    string filename = "records.bin";
    writeRecords(filename, data);

    // 读取
    auto loaded = readRecords(filename);

    // 打印
    cout << "\n=== 读取结果 ===" << endl;
    for (const auto& rec : loaded) {
        cout << "ID: " << rec.id
             << "  Name: " << rec.name
             << "  Score: " << rec.score << endl;
    }

    return 0;
}

语法规则：二进制文件模式 ios::binary

ofstream out("file.bin", ios::binary);
ifstream in("file.bin", ios::binary);
// 必须加 ios::binary，否则换行符会被转换（Windows: \n → \r\n）

举一反三：ios::binary 阻止文本模式的特殊处理。二进制模式保证字节原样写入/读取。忘记此标志会导致 Windows 上文件损坏或读取偏移。

语法规则：reinterpret_cast 字节转换

int x = 42;
out.write(reinterpret_cast<const char*>(&x), sizeof(x));
// 把任意类型指针转为 char*（字节流）

举一反三：reinterpret_cast 是最危险的转换，应该仅在 I/O 和底层操作中使用。直接写入 struct 字节需要注意 endianness 和 padding 差异（不同平台/编译器可能不兼容）。

语法规则：#pragma pack 控制对齐

#pragma pack(push, 1)
struct Record { int id; char name[20]; double score; };
#pragma pack(pop)
// pack(1) 禁用填充，保证 sizeof 精确

举一反三：默认对齐会使 struct 大小 > 成员大小之和（填充字节）。序列化时禁用对齐确保可移植性，但可能降低访问效率。更好的做法是逐字段序列化而非整块写入。
