案例 63：std::any 类型擦除

使用 std::any 存储任意类型的值，实现一个通用配置容器。

核心知识点：any、any_cast、类型擦除、has_value

完整代码（含逐行注释）

// 案例 63：std::any 类型擦除
// 通用配置容器，存储任意类型
#include <iostream>
#include <any>
#include <string>
#include <map>
#include <vector>
#include <typeinfo>
using namespace std;

// 通用配置管理器
class Config {
private:
    map<string, any> values;

public:
    // 设置配置值（任意类型）
    template <typename T>
    void set(const string& key, const T& value) {
        values[key] = value;
    }

    // 获取配置值
    template <typename T>
    T get(const string& key, const T& defaultVal = T{}) const {
        auto it = values.find(key);
        if (it != values.end()) {
            try {
                return any_cast<T>(it->second);
            } catch (const bad_any_cast&) {
                cerr << "类型不匹配: " << key << endl;
            }
        }
        return defaultVal;
    }

    // 检查键是否存在
    bool has(const string& key) const {
        return values.find(key) != values.end();
    }

    // 获取存储的类型名
    string typeOf(const string& key) const {
        auto it = values.find(key);
        if (it != values.end()) {
            return it->second.type().name();
        }
        return "(none)";
    }

    // 打印所有配置
    void print() const {
        cout << "=== 配置列表 ===" << endl;
        for (const auto& [key, val] : values) {
            cout << key << " (type: " << val.type().name() << ")" << endl;
        }
    }
};

int main() {
    Config cfg;

    // 存储不同类型的值
    cfg.set("host", string("localhost"));
    cfg.set("port", 8080);
    cfg.set("timeout", 30.5);
    cfg.set("enabled", true);
    cfg.set("retries", 3);

    cfg.print();

    // 读取不同类型的值
    cout << "\nhost: " << cfg.get<string>("host") << endl;
    cout << "port: " << cfg.get<int>("port") << endl;
    cout << "timeout: " << cfg.get<double>("timeout") << endl;
    cout << "enabled: " << boolalpha << cfg.get<bool>("enabled") << endl;
    cout << "missing (default): " << cfg.get<int>("missing", -1) << endl;

    return 0;
}
