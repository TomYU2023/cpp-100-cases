案例 63：std::any 类型擦除

使用 std::any 存储任意类型的值，实现一个通用配置容器。

核心知识点：any、any_cast、类型擦除、has_value

完整代码（含逐行注释）

// 案例 63：std::any 类型擦除
// 通用配置容器，存储任意类型
#include <iostream>
#include <any>
#include <string>
#include <map>
#include <vector>
#include <typeinfo>
using namespace std;

// 通用配置管理器
class Config {
private:
    map<string, any> values;

public:
    // 设置配置值（任意类型）
    template <typename T>
    void set(const string& key, const T& value) {
        values[key] = value;
    }

    // 获取配置值
    template <typename T>
    T get(const string& key, const T& defaultVal = T{}) const {
        auto it = values.find(key);
        if (it != values.end()) {
            try {
                return any_cast<T>(it->second);
            } catch (const bad_any_cast&) {
                cerr << "类型不匹配: " << key << endl;
            }
        }
        return defaultVal;
    }

    // 检查键是否存在
    bool has(const string& key) const {
        return values.find(key) != values.end();
    }

    // 获取存储的类型名
    string typeOf(const string& key) const {
        auto it = values.find(key);
        if (it != values.end()) {
            return it->second.type().name();
        }
        return "(none)";
    }

    // 打印所有配置
    void print() const {
        cout << "=== 配置列表 ===" << endl;
        for (const auto& [key, val] : values) {
            cout << key << " (type: " << val.type().name() << ")" << endl;
        }
    }
};

int main() {
    Config cfg;

    // 存储不同类型的值
    cfg.set("host", string("localhost"));
    cfg.set("port", 8080);
    cfg.set("timeout", 30.5);
    cfg.set("enabled", true);
    cfg.set("retries", 3);

    cfg.print();

    // 读取不同类型的值
    cout << "\nhost: " << cfg.get<string>("host") << endl;
    cout << "port: " << cfg.get<int>("port") << endl;
    cout << "timeout: " << cfg.get<double>("timeout") << endl;
    cout << "enabled: " << boolalpha << cfg.get<bool>("enabled") << endl;
    cout << "missing (default): " << cfg.get<int>("missing", -1) << endl;

    return 0;
}

语法规则：std::any 基本操作

#include <any>
any a = 42;              // 存 int
a = string("hello");    // 改为 string
a.has_value();          // 是否有值
int v = any_cast<int>(a); // 提取值（错误类型抛 bad_any_cast）

举一反三：any 可以存储任何可拷贝类型（类似于 void* + 类型信息）。比 union 安全（有类型检查），比 variant 灵活（不限定类型集合）。代价：堆分配 + RTTI。

语法规则：any_cast 类型安全提取

try {
    int v = any_cast<int>(a);
} catch (const bad_any_cast& e) {
    // 类型不匹配
}
// 也可用指针版本：if (int* p = any_cast<int>(&a))

举一反三：any_cast 分两种：引用版本（类型不匹配抛异常）和指针版本（返回 nullptr）。指针版本适合不确定类型的场景（如 if-else 链判断）。

语法规则：any.type() 运行时类型查询

cout << a.type().name(); // 返回 type_info
if (a.type() == typeid(int)) { /* ... */ }
// 具体 name 内容实现定义（GCC 返回简写，MSVC 返回完整名）

举一反三：type().name() 在不同编译器返回不同字符串（不可移植）。用 type() == typeid(T) 比较类型更可靠。any 依赖 RTTI，禁用 RTTI 的构建无法使用。
