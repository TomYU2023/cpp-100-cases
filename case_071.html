案例 71：std::function 回调系统

使用 function 构建灵活的事件回调系统，支持注册、注销和触发。

核心知识点：function、事件系统、回调注册、token 模式

完整代码（含逐行注释）

// 案例 71：std::function 回调系统
// 可注册和注销的事件回调管理器
#include <iostream>
#include <functional>
#include <map>
#include <string>
using namespace std;

// 事件管理器
template <typename... Args>
class EventManager {
public:
    using Callback = function<void(Args...)>;
    using Token = int;

private:
    map<Token, Callback> callbacks;
    Token nextToken = 1;

public:
    // 注册回调，返回 token
    Token subscribe(Callback cb) {
        Token token = nextToken++;
        callbacks[token] = move(cb);
        return token;
    }

    // 注销回调
    bool unsubscribe(Token token) {
        return callbacks.erase(token) > 0;
    }

    // 触发事件（通知所有回调）
    void emit(Args... args) {
        for (auto& [token, cb] : callbacks) {
            cb(args...); // 调用每个回调
        }
    }

    // 获取注册的回调数量
    size_t count() const { return callbacks.size(); }
};

// 简单的按钮类
class Button {
private:
    string label;
public:
    EventManager<> onClick; // 点击事件

    explicit Button(string l) : label(move(l)) {}

    void click() {
        cout << "按钮 [" << label << "] 被点击" << endl;
        onClick.emit();
    }
};

int main() {
    // 1. 带参数的事件
    EventManager<int, string> logger;

    auto token1 = logger.subscribe([](int code, const string& msg) {
        cout << "[控制台] 错误 " << code << ": " << msg << endl;
    });

    auto token2 = logger.subscribe([](int code, const string& msg) {
        cout << "[文件日志] " << code << " - " << msg << endl;
    });

    logger.emit(404, "页面未找到");
    logger.emit(500, "服务器内部错误");

    // 注销一个回调
    logger.unsubscribe(token1);
    cout << "\n注销后再次触发:" << endl;
    logger.emit(403, "禁止访问");

    // 2. 按钮点击事件
    cout << "\n=== 按钮事件 ===" << endl;
    Button btn("提交");

    btn.onClick.subscribe([]() {
        cout << "  处理: 验证表单..." << endl;
    });
    btn.onClick.subscribe([]() {
        cout << "  处理: 提交数据..." << endl;
    });
    btn.onClick.subscribe([]() {
        cout << "  处理: 显示成功提示" << endl;
    });

    btn.click();

    return 0;
}
