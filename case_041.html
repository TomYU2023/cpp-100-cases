案例 41：命令行参数解析器

实现一个命令行参数解析器，支持短选项、长选项和值参数，练习字符串处理。

核心知识点：argc/argv、string 操作、map 存储选项

完整代码（含逐行注释）

// 案例 41：命令行参数解析器
// 解析类似 --name=value 和 -f 的选项
#include <iostream>
#include <string>
#include <map>
#include <vector>
using namespace std;

class ArgParser {
private:
    map<string, string> options;   // --key=value
    vector<string> positional;     // 位置参数

public:
    // 构造函数中解析参数
    ArgParser(int argc, char* argv[]) {
        for (int i = 1; i < argc; i++) {
            string arg = argv[i];

            if (arg.size() >= 2 && arg[0] == '-') {
                // 找到 = 的位置
                size_t eqPos = arg.find('=');

                if (eqPos != string::npos) {
                    // --key=value 形式
                    string key = arg.substr(0, eqPos);
                    string val = arg.substr(eqPos + 1);
                    options[key] = val;
                } else {
                    // -f 或 --flag 形式（无值）
                    options[arg] = "true";
                }
            } else {
                positional.push_back(arg);
            }
        }
    }

    // 查询选项值
    string get(const string& key, const string& defaultVal = "") const {
        auto it = options.find(key);
        return (it != options.end()) ? it->second : defaultVal;
    }

    // 检查选项是否存在
    bool has(const string& key) const {
        return options.find(key) != options.end();
    }

    // 获取位置参数
    const vector<string>& getPositional() const { return positional; }

    // 打印所有选项
    void print() const {
        cout << "=== 解析结果 ===" << endl;
        for (const auto& [k, v] : options) {
            cout << "  " << k << " = " << v << endl;
        }
        cout << "位置参数: ";
        for (const auto& p : positional) cout << p << " ";
        cout << endl;
    }
};

int main() {
    // 模拟命令行参数
    const char* fakeArgv[] = {
        "program",
        "--name=Alice",
        "--age=25",
        "-v",
        "-d",
        "--output=result.txt",
        "input1.txt",
        "input2.txt"
    };
    int fakeArgc = sizeof(fakeArgv) / sizeof(fakeArgv[0]);

    ArgParser parser(fakeArgc, const_cast<char**>(fakeArgv));

    parser.print();

    cout << "\n--name: " << parser.get("--name") << endl;
    cout << "--age: " << parser.get("--age") << endl;
    cout << "verbose: " << (parser.has("-v") ? "ON" : "OFF") << endl;
    cout << "debug: " << (parser.has("-d") ? "ON" : "OFF") << endl;

    return 0;
}

语法规则：argc 和 argv

int main(int argc, char* argv[]) {
    // argc: 参数数量（包括程序名）
    // argv[0]: 程序名
    // argv[1..argc-1]: 实际参数
}

举一反三：argv 是 C 风格字符串数组。遍历时注意边界检查。参数可能包含空格（被 shell 用引号括起时）。Windows 下可用 GetCommandLineW 获取完整命令行。

语法规则：字符串查找与截取

string arg = "--name=value";
size_t pos = arg.find('=');         // 找分隔符
string key = arg.substr(0, pos);   // 从 0 开始取 pos 个字符
string val = arg.substr(pos + 1);  // 从 pos+1 到末尾

举一反三：find 返回 string::npos 表示未找到。substr(pos, count) 的第二参数可省略（取到末尾）。注意检查返回值避免越界。

语法规则：map 存配置选项

map<string, string> options;
options["--name"] = "Alice";
if (options.count("-v")) { /* 存在 */ }

举一反三：map 天然支持 key-value 查询。命令行参数通常用 unordered_map（不需要排序）。bool 型选项常用存在性判断。
