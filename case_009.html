案例 9：学生成绩管理系统(struct版)

使用结构体实现一个学生成绩管理系统，支持添加、查询、排序、统计功能。

核心知识点：struct 定义、vector 存储、sort 自定义排序

完整代码（含逐行注释）

// 案例 9：学生成绩管理系统 (struct 版)
// 使用结构体 + vector + lambda 排序实现管理功能
#include <iostream>
#include <vector>
#include <string>
#include <algorithm>
#include <iomanip>
using namespace std;

// 学生结构体
struct Student {
    string name;
    int math;
    int english;
    int cs;

    // 计算总分
    int total() const { return math + english + cs; }
    // 计算平均分
    double avg() const { return total() / 3.0; }
};

// 打印单个学生信息
void printStudent(const Student& s) {
    cout << left << setw(10) << s.name
         << setw(8) << s.math
         << setw(8) << s.english
         << setw(8) << s.cs
         << setw(8) << s.total()
         << fixed << setprecision(1) << s.avg() << endl;
}

int main() {
    // 初始化学生数据
    vector<Student> students = {
        {"张三", 85, 92, 78},
        {"李四", 90, 88, 95},
        {"王五", 72, 85, 80},
        {"赵六", 95, 91, 88}
    };

    // 按总分降序排序（lambda 表达式自定义比较规则）
    sort(students.begin(), students.end(),
         [](const Student& a, const Student& b) {
             return a.total() > b.total(); // 总分高的排前面
         });

    // 打印表头
    cout << left << setw(10) << "姓名"
         << setw(8) << "数学"
         << setw(8) << "英语"
         << setw(8) << "C++"
         << setw(8) << "总分"
         << "平均分" << endl;
    cout << string(50, '-') << endl;

    // 打印所有学生
    for (const auto& s : students) {
        printStudent(s);
    }

    return 0;
}

语法规则：struct 定义与成员函数

struct Student {
    string name;
    int score;
    int total() const { return score; }
};

举一反三：C++ 中 struct 与 class 几乎相同，仅默认访问权限不同（struct 默认 public）。const 成员函数承诺不修改成员变量。

语法规则：sort + lambda 自定义排序

sort(v.begin(), v.end(),
     [](const T& a, const T& b) { return a.key > b.key; });

举一反三：第三个参数 lambda 返回 true 表示 a 应排在 b 前面。也可传函数指针或函数对象。降序用 >，升序用 <。

语法规则：iomanip 格式化表格

#include <iomanip>
cout << left << setw(10) << name; // 左对齐，宽度 10

举一反三：left/right 控制对齐方向，setw 控制列宽，setprecision(n) 控制小数位数，fixed 固定小数格式。
