3 #include <optional> C++17 optional 可有可无的值

9 void demoVariant() {

11
12 v = 42; 存储 int
13 cout << get<int>(v) << endl; get<T> 获取值

23 cout << "int: " << *p << endl;
24 } else {
25 cout << "v 不是 int" << endl;
26 }

28
29 visit([](const auto& val) { 泛型 lambda：auto 自动匹配类型
30 using T = decay_t<decltype(val)>; decltype 获取类型，decay_t 去掉引用和 const
31 if constexpr (is_same_v<T, int>) 编译期判断类型
32 cout << "这是整数: " << val << endl;
33 else if constexpr (is_same_v<T, double>)
34 cout << "这是浮点数: " << val << endl;
35 else

37 }, v);
38 }

41 void demoOptional() {
42 optional<int> maybe; 默认构造为空，值不存在
43 if (!maybe) cout << \"maybe 为空\" << endl; 可以当 bool 判断
44
45 maybe = 100; 赋值后值存在
46 cout << \"值: \" << *maybe << endl; 解引用获取值
47 cout << \"值: \" << maybe.value() << endl; value() 会检查，空时抛异常
48 cout << \"值: \" << maybe.value_or(-1) << endl; value_or 提供默认值

51 auto findFirst = [](const vector<int>& v, int target) -> optional<size_t> { 返回索引或空

54 }
55 return nullopt; 明确表示没找到

新语法讲解

variant（类型安全联合体）

C++17 引入，替代传统 union。语法：variant<类型列表...> 变量;。用 get<T>(v) 获取，get_if<T>(&v) 安全获取，visit 对不同类型执行不同操作。
variant<int, string> v;
v = 1;          // int
v = "text";     // string，旧值被销毁
int i = get<int>(v); // 如果当前不是 int 会抛 bad_variant_access

optional（可有可无的值）

C++17 引入，表示值可能存在也可能不存在。比用 -1 或 nullptr 表示 missing 更安全、语义更明确。
optional<string> name;
name = "Alice";
cout << *name;       // 需确保有值
cout << name.value_or("未知"); // 总是安全

if constexpr（编译期条件）

C++17 引入，在编译期根据条件决定是否编译某段代码。不同于普通 if（运行时判断），不成立的分支不会被编译，允许写出对某种类型无效的代码。
if constexpr (is_integral_v<T>) { bitOp(x); }
else { stringOp(x); }
// 如果 T=int，else 分支不编译，避免 stringOp(int) 错误

举一反三

variant 实现表达式求值

代码：using Node=variant; visit + 递归求值

思路：AST 节点的类型安全表示

optional 实现配置读取

代码：optional readConfig(key) { if(exists) return val; else return nullopt; }

思路：取代返回空字符串的模糊语义

组合使用 variant+optional

代码：using ApiResult = variant;

思路：类型安全的 Result 类型，类似 Rust
