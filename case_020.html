案例 20：variant + optional

使用 C++17 的 std::variant 和 std::optional 实现安全的多类型返回和可选值处理。

核心知识点：variant、optional、visit、值语义

完整代码（含逐行注释）

// 案例 20：variant + optional
// C++17 类型安全的多类型联合和可选值
#include <iostream>
#include <variant>   // variant
#include <optional>  // optional
#include <string>
#include <map>
using namespace std;

// 函数可返回 int、double 或 string 三种类型之一
variant<int, double, string> parseValue(const string& input) {
    try {
        // 尝试转为 int
        size_t pos;
        int i = stoi(input, &pos);
        if (pos == input.size()) return i;
    } catch (...) {}

    try {
        // 尝试转为 double
        size_t pos;
        double d = stod(input, &pos);
        if (pos == input.size()) return d;
    } catch (...) {}

    // 最后当作字符串
    return input;
}

// 使用 optional 的安全查找
optional<string> findUser(const map<int, string>& users, int id) {
    auto it = users.find(id);
    if (it != users.end()) {
        return it->second;         // 找到了
    }
    return nullopt;                 // 没找到
}

int main() {
    // === variant 用法 ===
    vector<string> inputs = {"42", "3.14", "hello"};

    for (const auto& s : inputs) {
        auto val = parseValue(s);
        // visit 模式匹配：根据当前实际类型执行对应 Lambda
        visit([](const auto& v) {
            using T = decay_t<decltype(v)>;
            if constexpr (is_same_v<T, int>)
                cout << "整数: " << v << endl;
            else if constexpr (is_same_v<T, double>)
                cout << "浮点: " << v << endl;
            else
                cout << "字符串: " << v << endl;
        }, val);
    }

    // === optional 用法 ===
    map<int, string> users = {{1, "Alice"}, {2, "Bob"}};

    auto result = findUser(users, 1);
    if (result.has_value()) {          // 或 if (result)
        cout << "找到用户: " << result.value() << endl;
    }

    // 使用 value_or 提供默认值
    auto noResult = findUser(users, 99);
    cout << "用户99: " << noResult.value_or("未知用户") << endl;

    return 0;
}

存储字符串：Hello C++17
存储浮点数：3.14

optional 示例：

  key "K3" not found, using default value: -1
