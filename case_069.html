案例 69：std::array 性能对比

对比 array、vector 和 C 数组在性能上的差异，理解栈分配 vs 堆分配。

核心知识点：array、栈分配、编译期大小、性能基准

完整代码（含逐行注释）

// 案例 69：std::array 性能对比
// 对比 array、vector 和 C 数组的访问性能
#include <iostream>
#include <array>
#include <vector>
#include <chrono>
#include <numeric>
using namespace std;
using namespace chrono;

constexpr int SIZE = 10000000;

// 计时辅助函数
template <typename F>
double measure(F&& func, const string& name) {
    auto start = high_resolution_clock::now();
    func();
    auto end = high_resolution_clock::now();
    double ms = duration<double, milli>(end - start).count();
    cout << name << ": " << ms << " ms" << endl;
    return ms;
}

int main() {
    cout << "=== 容器性能对比 (N=" << SIZE << ") ===" << endl;

    // 1. C 风格数组
    int* cArr = new int[SIZE];
    iota(cArr, cArr + SIZE, 0);

    // 2. std::array（假设 SIZE 较小的版本）
    // array 必须在编译期确定大小，用 1000 做演示
    constexpr int SMALL = 100000;
    array<int, SMALL> stdArr;
    iota(stdArr.begin(), stdArr.end(), 0);

    // 3. std::vector
    vector<int> vec(SIZE);
    iota(vec.begin(), vec.end(), 0);

    // 测试顺序访问
    cout << "\n--- 顺序求和 ---" << endl;

    measure([&]() {
        volatile long long sum = 0;
        for (int i = 0; i < SIZE; i++) sum += cArr[i];
    }, "C数组 (heap)");

    measure([&]() {
        volatile long long sum = 0;
        for (int i = 0; i < SMALL; i++) sum += stdArr[i];
    }, "std::array (stack) x10");

    measure([&]() {
        volatile long long sum = 0;
        for (int i = 0; i < SIZE; i++) sum += vec[i];
    }, "std::vector");

    // 测试随机访问
    cout << "\n--- 随机访问 ---" << endl;

    measure([&]() {
        volatile long long sum = 0;
        for (int i = 0; i < SIZE; i++) {
            int idx = (i * 1234567 + 7654321) % SIZE;
            sum += cArr[idx];
        }
    }, "C数组 随机");

    measure([&]() {
        volatile long long sum = 0;
        for (int i = 0; i < SIZE; i++) {
            int idx = (i * 1234567 + 7654321) % SIZE;
            sum += vec[idx];
        }
    }, "vector 随机");

    // 对比 at() vs operator[]
    cout << "\n--- at() vs operator[] ---" << endl;

    measure([&]() {
        volatile long long sum = 0;
        for (int i = 0; i < SMALL; i++) sum += stdArr.at(i);
    }, "array::at() (带边界检查)");

    measure([&]() {
        volatile long long sum = 0;
        for (int i = 0; i < SMALL; i++) sum += stdArr[i];
    }, "array::operator[] (无检查)");

    delete[] cArr;
    return 0;
}

语法规则：array vs vector vs C数组

// array: 栈分配，编译期大小，无动态分配
array<int, 100> arr;
// vector: 堆分配，运行时大小，可动态增长
vector<int> vec(100);
// C数组: 需手动管理，无边界信息
int cArr[100];

举一反三：array 是 C 数组的现代替代，支持迭代器、size()、at() 等。无额外开销（sizeof(array) == sizeof(C数组)）。编译期确定大小是限制也是优势。

语法规则：operator[] vs at()

arr[5];    // 不检查边界，越界是 UB
arr.at(5); // 检查边界，越界抛 out_of_range
// at() 有 1~3% 的性能开销（分支预测良好的情况下）

举一反三：operator[] 用于确定不会越界的场景（如循环中已知范围）。at() 用于不确定的场景（如用户输入索引）。Debug 模式下编译器的 operator[] 可能也有检查。

语法规则：array 初始化和赋值

array<int, 5> a = {1, 2, 3, 4, 5}; // 列表初始化
a.fill(0);                              // 全部填充
auto b = a;                            // 值拷贝（而非指针拷贝）

举一反三：与 C 数组不同，array 可以直接赋值和拷贝。fill() 比 memset 类型安全。array 可以整体传给函数（按值或引用），保留大小信息。
