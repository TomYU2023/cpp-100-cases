案例 69：std::array 性能对比

对比 array、vector 和 C 数组在性能上的差异，理解栈分配 vs 堆分配。

核心知识点：array、栈分配、编译期大小、性能基准

完整代码（含逐行注释）

// 案例 69：std::array 性能对比
// 对比 array、vector 和 C 数组的访问性能
#include <iostream>
#include <array>
#include <vector>
#include <chrono>
#include <numeric>
using namespace std;
using namespace chrono;

constexpr int SIZE = 10000000;

// 计时辅助函数
template <typename F>
double measure(F&& func, const string& name) {
    auto start = high_resolution_clock::now();
    func();
    auto end = high_resolution_clock::now();
    double ms = duration<double, milli>(end - start).count();
    cout << name << ": " << ms << " ms" << endl;
    return ms;
}

int main() {
    cout << "=== 容器性能对比 (N=" << SIZE << ") ===" << endl;

    // 1. C 风格数组
    int* cArr = new int[SIZE];
    iota(cArr, cArr + SIZE, 0);

    // 2. std::array（假设 SIZE 较小的版本）
    // array 必须在编译期确定大小，用 1000 做演示
    constexpr int SMALL = 100000;
    array<int, SMALL> stdArr;
    iota(stdArr.begin(), stdArr.end(), 0);

    // 3. std::vector
    vector<int> vec(SIZE);
    iota(vec.begin(), vec.end(), 0);

    // 测试顺序访问
    cout << "\n--- 顺序求和 ---" << endl;

    measure([&]() {
        volatile long long sum = 0;
        for (int i = 0; i < SIZE; i++) sum += cArr[i];
    }, "C数组 (heap)");

    measure([&]() {
        volatile long long sum = 0;
        for (int i = 0; i < SMALL; i++) sum += stdArr[i];
    }, "std::array (stack) x10");

    measure([&]() {
        volatile long long sum = 0;
        for (int i = 0; i < SIZE; i++) sum += vec[i];
    }, "std::vector");

    // 测试随机访问
    cout << "\n--- 随机访问 ---" << endl;

    measure([&]() {
        volatile long long sum = 0;
        for (int i = 0; i < SIZE; i++) {
            int idx = (i * 1234567 + 7654321) % SIZE;
            sum += cArr[idx];
        }
    }, "C数组 随机");

    measure([&]() {
        volatile long long sum = 0;
        for (int i = 0; i < SIZE; i++) {
            int idx = (i * 1234567 + 7654321) % SIZE;
            sum += vec[idx];
        }
    }, "vector 随机");

    // 对比 at() vs operator[]
    cout << "\n--- at() vs operator[] ---" << endl;

    measure([&]() {
        volatile long long sum = 0;
        for (int i = 0; i < SMALL; i++) sum += stdArr.at(i);
    }, "array::at() (带边界检查)");

    measure([&]() {
        volatile long long sum = 0;
        for (int i = 0; i < SMALL; i++) sum += stdArr[i];
    }, "array::operator[] (无检查)");

    delete[] cArr;
    return 0;
}
