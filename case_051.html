案例 51：策略模式

实现策略模式，将排序算法封装为可替换的策略对象，理解行为参数化。

核心知识点：策略模式、函数对象、模板策略

完整代码（含逐行注释）

// 案例 51：策略模式
// 将排序策略封装为可替换的函数对象
#include <iostream>
#include <vector>
#include <algorithm>
#include <string>
#include <functional>
using namespace std;

// 策略接口：排序策略
template <typename T>
using SortStrategy = function<void(vector<T>&)>;

// 具体策略：升序排序
template <typename T>
void ascendingSort(vector<T>& data) {
    sort(data.begin(), data.end());
}

// 具体策略：降序排序
template <typename T>
void descendingSort(vector<T>& data) {
    sort(data.begin(), data.end(), greater<T>());
}

// 具体策略：按绝对值排序（数值类型）
void absoluteSort(vector<int>& data) {
    sort(data.begin(), data.end(),
         [](int a, int b) { return abs(a) < abs(b); });
}

// 上下文：使用策略的排序器
template <typename T>
class Sorter {
private:
    vector<T> data;
    SortStrategy<T> strategy;

public:
    Sorter(vector<T> d, SortStrategy<T> s)
        : data(move(d)), strategy(s) {}

    // 运行时切换策略
    void setStrategy(SortStrategy<T> s) { strategy = s; }

    // 执行排序
    void execute() {
        cout << "排序前: ";
        for (const auto& x : data) cout << x << " ";
        cout << endl;

        strategy(data);

        cout << "排序后: ";
        for (const auto& x : data) cout << x << " ";
        cout << endl;
    }
};

int main() {
    vector<int> nums = {3, -1, 4, -5, 2, -9, 7};

    // 使用升序策略
    cout << "=== 升序 ===" << endl;
    Sorter<int> sorter(nums, ascendingSort<int>);
    sorter.execute();

    // 切换为降序策略
    cout << "\n=== 降序 ===" << endl;
    sorter.setStrategy(descendingSort<int>);
    sorter.execute();

    // 切换为绝对值排序
    cout << "\n=== 绝对值排序 ===" << endl;
    sorter.setStrategy(absoluteSort);
    sorter.execute();

    return 0;
}

运行结果

排序前: 3 -1 4 -5 2 -9 7 
排序后: -9 -5 -1 2 3 4 7 

=== 降序 ===
排序前: -9 -5 -1 2 3 4 7 
排序后: 7 4 3 2 -1 -5 -9 

=== 绝对值排序 ===
排序前: 7 4 3 2 -1 -5 -9 
排序后: -1 2 3 4 -5 7 -9
