案例 90：简易 HTTP 反向代理

实现基于 epoll 的非阻塞反向代理服务器，转发客户端请求到后端服务器，展示高性能网络编程的核心架构。

核心知识点：epoll 多路复用、非阻塞 IO(fcntl)、socket 编程、HTTP 协议解析、字节序转换

完整代码（含逐行注释）

// 案例 90：简易 HTTP 反向代理
// 在客户端和后端服务器之间建立中间层，转发请求和响应。
// 展示非阻塞 IO、epoll 多路复用、HTTP 协议解析的核心概念。

// 注意：本案例使用 Linux 的 epoll API，在 WSL 或 Linux 环境下编译运行。
// 编译: g++ -std=c++17 proxy.cpp -o proxy

#include <iostream>
#include <string>
#include <cstring>
#include <map>
#include <vector>
#include <sstream>
#include <unistd.h>
#include <sys/socket.h>
#include <sys/epoll.h>
#include <netinet/in.h>
#include <arpa/inet.h>
#include <fcntl.h>
#include <errno.h>

const int PORT = 8080;              // 代理监听端口
const char* BACKEND_HOST = "127.0.0.1";  // 后端服务器地址
const int BACKEND_PORT = 9000;      // 后端服务器端口
const int MAX_EVENTS = 1024;
const int BUFFER_SIZE = 4096;

// ─── 设置非阻塞 ──────────────────────────────────
int setNonBlocking(int fd) {
    int flags = fcntl(fd, F_GETFL, 0);
    return fcntl(fd, F_SETFL, flags | O_NONBLOCK);
}

// HTTP 请求（简化版，仅 GET）
struct HttpRequest {
    std::string method, path, version;
    std::map<std::string, std::string> headers;
    std::string body;
};

// ─── 解析 HTTP 请求 ──────────────────────────────
HttpRequest parseRequest(const std::string& raw) {
    HttpRequest req;
    std::istringstream ss(raw);
    std::string line;

    // 请求行: GET /index.html HTTP/1.1
    std::getline(ss, line);
    std::istringstream lineStream(line);
    lineStream >> req.method >> req.path >> req.version;

    // 解析头部
    while (std::getline(ss, line) && line != "\r") {
        auto colon = line.find(':');
        if (colon != std::string::npos)
            req.headers[line.substr(0, colon)] = line.substr(colon + 2);
    }
    return req;
}

// ─── 连接到后端 ──────────────────────────────────
int connectBackend() {
    int fd = socket(AF_INET, SOCK_STREAM, 0);
    sockaddr_in addr = {};
    addr.sin_family = AF_INET;
    addr.sin_port = htons(BACKEND_PORT);
    inet_pton(AF_INET, BACKEND_HOST, &addr.sin_addr);
    if (connect(fd, (sockaddr*)&addr, sizeof(addr)) < 0) {
        close(fd);
        return -1;
    }
    return fd;
}

// ─── 创建 TCP 套接字并绑定 ──────────────────────
// 创建一个监听套接字，用于接收客户端的连接请求
int createListenSocket() {
    int fd = socket(AF_INET, SOCK_STREAM, 0);  // AF_INET=IPv4, SOCK_STREAM=TCP
    int opt = 1;
    setsockopt(fd, SOL_SOCKET, SO_REUSEADDR, &opt, sizeof(opt));

    sockaddr_in addr = {};
    addr.sin_family = AF_INET;
    addr.sin_addr.s_addr = INADDR_ANY;   // 监听所有网络接口
    addr.sin_port = htons(PORT);

    bind(fd, (sockaddr*)&addr, sizeof(addr));
    listen(fd, SOMAXCONN);  // 监听，最大连接队列
    return fd;
}

// ─── epoll 事件循环 ──────────────────────────────
int main() {
    int listenFd = createListenSocket();
    setNonBlocking(listenFd);

    // 创建 epoll 实例
    int epollFd = epoll_create1(0);  // 0 或 EPOLL_CLOEXEC

    // 注册监听套接字到 epoll
    epoll_event ev = {};
    ev.events = EPOLLIN;        // 关注可读事件
    ev.data.fd = listenFd;
    epoll_ctl(epollFd, EPOLL_CTL_ADD, listenFd, &ev);

    std::cout << "代理服务器运行在 0.0.0.0:" << PORT << std::endl;
    std::cout << "后端: " << BACKEND_HOST << ":" << BACKEND_PORT << std::endl;

    epoll_event events[MAX_EVENTS];  // 就绪事件数组

    while (true) {
        // 等待事件（-1 表示无限等待直到有事件）
        int nfds = epoll_wait(epollFd, events, MAX_EVENTS, -1);

        for (int i = 0; i < nfds; i++) {
            int fd = events[i].data.fd;

            // 事件 1：新客户端连接
            if (fd == listenFd) {
                sockaddr_in clientAddr;
                socklen_t len = sizeof(clientAddr);
                int clientFd = accept(listenFd, (sockaddr*)&clientAddr, &len);
                setNonBlocking(clientFd);

                // 连接到后端
                int backendFd = connectBackend();
                setNonBlocking(backendFd);

                // 注册客户端和后端到 epoll
                ev.events = EPOLLIN | EPOLLET;  // ET 边缘触发
                ev.data.fd = clientFd;
                epoll_ctl(epollFd, EPOLL_CTL_ADD, clientFd, &ev);

                ev.data.fd = backendFd;
                epoll_ctl(epollFd, EPOLL_CTL_ADD, backendFd, &ev);

                std::cout << "[连接] 新连接 fd=" << clientFd
                          << " → 后端 fd=" << backendFd << std::endl;
            }
            // 事件 2：客户端或后端发来数据
            else {
                char buf[BUFFER_SIZE];
                int n = recv(fd, buf, sizeof(buf) - 1, 0);

                if (n <= 0) {
                    // 连接关闭
                    std::cout << "[关闭] fd=" << fd << std::endl;
                    epoll_ctl(epollFd, EPOLL_CTL_DEL, fd, nullptr);
                    close(fd);
                } else {
                    buf[n] = '\0';
                    std::cout << "[数据] fd=" << fd << " (" << n << " 字节)" << std::endl;

                    // 简化：直接转发（生产环境需要关联 client↔backend 并管理状态机）
                    // 实际代理应维护 fd 对映射，此处仅展示核心架构
                }
            }
        }
    }

    close(epollFd);
    close(listenFd);
    return 0;
}

语法规则：epoll 多路复用模型

int epfd = epoll_create1(0);
epoll_event ev;
ev.events = EPOLLIN;        // 关注可读
ev.data.fd = sockfd;
epoll_ctl(epfd, EPOLL_CTL_ADD, sockfd, &ev);

// 事件循环
epoll_event events[MAX];
int n = epoll_wait(epfd, events, MAX, timeout);
// events[0..n-1] 是就绪的事件

💡 举一反三：epoll 是 Linux 上高性能网络编程的核心。相比 select/poll（O(n) 遍历所有 fd），epoll 的事件驱动机制是 O(1)。边缘触发（EPOLLET）要求一次读完所有数据，否则不会再通知，适合高吞吐场景。macOS 上用 kqueue，Windows 上用 IOCP。

语法规则：fcntl 设置非阻塞文件描述符

#include <fcntl.h>
int flags = fcntl(fd, F_GETFL, 0);
fcntl(fd, F_SETFL, flags | O_NONBLOCK);

💡 举一反三：非阻塞模式下，read/write/accept 在没有数据时立即返回 -1 并设 errno=EAGAIN/EWOULDBLOCK，而不是阻塞等待。这是异步 IO 的基础，必须配合 epoll/select 使用。

语法规则：inet_pton 和 htons 网络地址转换

#include <arpa/inet.h>
inet_pton(AF_INET, "127.0.0.1", &addr.sin_addr);  // 字符串 → 网络字节序
addr.sin_port = htons(8080);  // host to network short — 端口字节序转换

💡 举一反三：网络上传输的数据按大端序（网络字节序），而 x86 是小端序，htons/htonl 做转换。inet_pton 替代老旧的 inet_addr，支持 IPv4 和 IPv6。
